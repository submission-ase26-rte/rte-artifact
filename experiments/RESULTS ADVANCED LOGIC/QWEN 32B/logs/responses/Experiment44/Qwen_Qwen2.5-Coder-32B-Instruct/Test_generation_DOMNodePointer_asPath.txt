public class DOMNodePointer_asPathTestGenerated {

    @Mock
    private Node node;

    @Mock
    private NodePointer parent;

    @Mock
    private Locale locale;

    @Mock
    private NamespaceResolver namespaceResolver;

    private DOMNodePointer domNodePointer;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        domNodePointer = new DOMNodePointer(node, locale);
    }

    @AfterEach
    public void tearDown() {
        domNodePointer = null;
    }

    @Test
    public void testAsPathWithId() {
        domNodePointer.id = "testId";
        when(domNodePointer.escape("testId")).thenReturn("testId");
        assertEquals("id('testId')", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithElementNodeAndNoParent() {
        when(node.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        when(DOMNodePointer.getLocalName(node)).thenReturn("element");
        when(domNodePointer.getNamespaceURI()).thenReturn(null);
        when(domNodePointer.getRelativePositionByQName()).thenReturn(1);
        assertEquals("/element[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithElementNodeAndParentNotDOMNodePointer() {
        domNodePointer.parent = parent;
        when(node.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        when(DOMNodePointer.getLocalName(node)).thenReturn("element");
        when(domNodePointer.getNamespaceURI()).thenReturn(null);
        when(domNodePointer.getRelativePositionByQName()).thenReturn(1);
        when(parent.asPath()).thenReturn("/parent");
        assertEquals("/parent/element[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithElementNodeAndParentDOMNodePointer() {
        domNodePointer.parent = parent;
        when(parent instanceof DOMNodePointer).thenReturn(true);
        when(node.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        when(DOMNodePointer.getLocalName(node)).thenReturn("element");
        when(domNodePointer.getNamespaceURI()).thenReturn(null);
        when(domNodePointer.getRelativePositionByQName()).thenReturn(1);
        when(parent.asPath()).thenReturn("/parent");
        assertEquals("/parent/element[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithElementNodeAndNamespace() {
        when(node.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        when(DOMNodePointer.getLocalName(node)).thenReturn("element");
        when(domNodePointer.getNamespaceURI()).thenReturn("http://example.com");
        when(domNodePointer.getNamespaceResolver().getPrefix("http://example.com")).thenReturn("ns");
        when(domNodePointer.getRelativePositionByQName()).thenReturn(1);
        assertEquals("/ns:element[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithElementNodeAndNoPrefix() {
        when(node.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        when(DOMNodePointer.getLocalName(node)).thenReturn("element");
        when(domNodePointer.getNamespaceURI()).thenReturn("http://example.com");
        when(domNodePointer.getNamespaceResolver().getPrefix("http://example.com")).thenReturn(null);
        when(domNodePointer.getRelativePositionOfElement()).thenReturn(1);
        assertEquals("/node()[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithTextNode() {
        when(node.getNodeType()).thenReturn(Node.TEXT_NODE);
        when(domNodePointer.getRelativePositionOfTextNode()).thenReturn(1);
        assertEquals("/text()[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithProcessingInstructionNode() {
        when(node.getNodeType()).thenReturn(Node.PROCESSING_INSTRUCTION_NODE);
        ProcessingInstruction pi = mock(ProcessingInstruction.class);
        when((ProcessingInstruction) node).thenReturn(pi);
        when(pi.getTarget()).thenReturn("target");
        when(domNodePointer.getRelativePositionOfPI()).thenReturn(1);
        assertEquals("/processing-instruction('target')[1]", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithDocumentNode() {
        when(node.getNodeType()).thenReturn(Node.DOCUMENT_NODE);
        assertEquals("", domNodePointer.asPath());
    }

    @Test
    public void testAsPathWithUnknownNodeType() {
        when(node.getNodeType()).thenReturn(Node.ATTRIBUTE_NODE);
        assertEquals("", domNodePointer.asPath());
    }
}
