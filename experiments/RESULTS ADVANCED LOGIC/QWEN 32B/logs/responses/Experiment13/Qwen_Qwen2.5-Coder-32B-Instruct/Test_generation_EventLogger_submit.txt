public class EventLogger_submitTestGenerated {

    @Mock
    private CargoRepository cargoRepository;

    @Mock
    private LocationRepository locationRepository;

    @Mock
    private VoyageRepository voyageRepository;

    @Mock
    private ApplicationEvents applicationEvents;

    @Mock
    private FacesContext context;

    @InjectMocks
    private EventLogger eventLogger;

    private AutoCloseable closeable;

    @BeforeEach
    public void setUp() {
        closeable = MockitoAnnotations.openMocks(this);
        eventLogger.setTrackingId("XYZ");
        eventLogger.setLocation("USNYC");
        eventLogger.setEventType(HandlingEvent.Type.LOAD);
        eventLogger.setVoyageNumber("voy001");
        eventLogger.setCompletionTime(LocalDateTime.now());

        FacesContextFactory contextFactory = (FacesContextFactory) FactoryFinder.getFactory(FactoryFinder.FACES_CONTEXT_FACTORY);
        LifecycleFactory lifecycleFactory = (LifecycleFactory) FactoryFinder.getFactory(FactoryFinder.LIFECYCLE_FACTORY);
        Lifecycle lifecycle = lifecycleFactory.getLifecycle(LifecycleFactory.DEFAULT_LIFECYCLE);
        eventLogger.context = contextFactory.getFacesContext(null, null, null, lifecycle);
    }

    @AfterEach
    public void tearDown() throws Exception {
        closeable.close();
    }

    @Test
    public void testSubmitWithVoyage() {
        eventLogger.setEventType(HandlingEvent.Type.LOAD);
        eventLogger.setVoyageNumber("voy001");

        eventLogger.submit();

        HandlingEventRegistrationAttempt attempt = new HandlingEventRegistrationAttempt(
                LocalDateTime.now(), eventLogger.getCompletionTime(), new TrackingId("XYZ"),
                new VoyageNumber("voy001"), HandlingEvent.Type.LOAD, new UnLocode("USNYC"));

        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(attempt);
        assertTrue(eventLogger.context.getMessageList().size() == 1);
        assertEquals("Event submitted", eventLogger.context.getMessageList().get(0).getSummary());
    }

    @Test
    public void testSubmitWithoutVoyage() {
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setVoyageNumber(null);

        eventLogger.submit();

        HandlingEventRegistrationAttempt attempt = new HandlingEventRegistrationAttempt(
                LocalDateTime.now(), eventLogger.getCompletionTime(), new TrackingId("XYZ"),
                null, HandlingEvent.Type.RECEIVE, new UnLocode("USNYC"));

        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(attempt);
        assertTrue(eventLogger.context.getMessageList().size() == 1);
        assertEquals("Event submitted", eventLogger.context.getMessageList().get(0).getSummary());
    }

    @Test
    public void testSubmitWithNullTrackingId() {
        eventLogger.setTrackingId(null);

        assertThrows(NullPointerException.class, () -> {
            eventLogger.submit();
        });
    }

    @Test
    public void testSubmitWithNullLocation() {

    @Test

    }

    @Test

        eventLogger.setCompletionTime(null);

        assertThrows(NullPointerException.class, () -> {
            eventLogger.submit();
        });
    }
}
