        Method result = MethodLookupUtils.lookupMethod(String.class, "valueOf", new Object[0]);
        assertNull(result);
    }

    @Test
    public void testLookupMethodWithFirstParameterNull() {
        Method result = MethodLookupUtils.lookupMethod(String.class, "valueOf", new Object[]{null});
        assertNull(result);
    }

        Method result = MethodLookupUtils.lookupMethod(String.class, "valueOf", new Object[]{"123", 456});
