import org.junit.jupiter.api.BeforeEach;

import org.mockito.InjectMocks;

import org.mockito.MockitoAnnotations;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.any;
import static org.mockito.ArgumentMatchers.eq;
import static org.mockito.Mockito.*;

public class TradeSLSBBean_completeOrderTestGenerated {
    @Mock
    private EntityManager entityManager;
    @InjectMocks
    private TradeSLSBBean tradeSLSBBean;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    public void testCompleteOrder_OrderNotFound() {
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(null);
        assertThrows(EJBException.class, () -> tradeSLSBBean.completeOrder(1, false));
    }

    @Test
    public void testCompleteOrder_OrderAlreadyCompleted() {
        OrderDataBean order = new OrderDataBean();
        order.setOrderStatus("completed");
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(order);
        tradeSLSBBean.completeOrder(1, false);
        verify(entityManager, never()).merge(any(OrderDataBean.class));
    }

    @Test
    public void testCompleteOrder_OrderCancelled() {
        OrderDataBean order = new OrderDataBean();
        order.setOrderStatus("cancelled");
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(order);
        tradeSLSBBean.completeOrder(1, false);
        verify(entityManager, never()).merge(any(OrderDataBean.class));
    }

    @Test
    public void testCompleteOrder_OrderOpen() {
        OrderDataBean order = new OrderDataBean();
        order.setOrderStatus("open");
        order.setOrderType("buy");
        order.setQuantity(10);
        order.setPrice(new BigDecimal("100.00"));
        order.setOrderFee(TradeConfig.getOrderFee("buy"));
        order.setCompletionDate(new Timestamp(System.currentTimeMillis()));
        AccountDataBean account = new AccountDataBean();
        account.setBalance(new BigDecimal("1000.00"));
        order.setAccount(account);
        QuoteDataBean quote = new QuoteDataBean("SUNW", "Sun Microsystems", 100000, new BigDecimal("100.00"), new BigDecimal("100.00"), new BigDecimal("100.00"), new BigDecimal("100.00"), 0.0);
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(order);
        when(entityManager.find(QuoteDataBean.class, "SUNW")).thenReturn(quote);
        when(entityManager.find(AccountDataBean.class, account.getAccountID())).thenReturn(account);
        tradeSLSBBean.completeOrder(1, false);
        verify(entityManager).merge(order);
        verify(entityManager).merge(account);
    }

    @Test
    public void testCompleteOrder_OrderOpenSell() {
        OrderDataBean order = new OrderDataBean();
        order.setOrderStatus("open");
        order.setOrderType("sell");
        order.setQuantity(10);
        order.setPrice(new BigDecimal("100.00"));
        order.setOrderFee(TradeConfig.getOrderFee("sell"));
        order.setCompletionDate(new Timestamp(System.currentTimeMillis()));
        AccountDataBean account = new AccountDataBean();
        account.setBalance(new BigDecimal("1000.00"));
        order.setAccount(account);
        QuoteDataBean quote = new QuoteDataBean("SUNW", "Sun Microsystems", 100000, new BigDecimal("100.00"), new BigDecimal("100.00"), new BigDecimal("100.00"), new BigDecimal("100.00"), 0.0);
        HoldingDataBean holding = new HoldingDataBean();
        holding.setQuantity(10);
        holding.setPurchasePrice(new BigDecimal("100.00"));
        holding.setQuote(quote);
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(order);
        when(entityManager.find(QuoteDataBean.class, "SUNW")).thenReturn(quote);
        when(entityManager.find(AccountDataBean.class, account.getAccountID())).thenReturn(account);
        when(entityManager.find(HoldingDataBean.class, holding.getHoldingID())).thenReturn(holding);
        tradeSLSBBean.completeOrder(1, false);
        verify(entityManager).merge(order);
        verify(entityManager).merge(account);
        verify(entityManager).merge(holding);
    }
}
