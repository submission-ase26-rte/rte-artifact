import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.mockito.MockedStatic;

import javax.faces.context.ExternalContext;
import javax.servlet.http.HttpSession;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class TradeConfigJSF_updateConfigTestGenerated {

    @Mock
    private ExternalContext context;

    @Mock
    private TradeDirectDBUtils dbUtils;

    @InjectMocks
    private TradeConfigJSF tradeConfigJSF;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        tradeConfigJSF.setRuntimeModeList(new String[]{"RuntimeMode1", "RuntimeMode2"});
        tradeConfigJSF.setOrderProcessingModeList(new String[]{"OrderProcessingMode1", "OrderProcessingMode2"});
        tradeConfigJSF.setWebInterfaceList(new String[]{"WebInterface1", "WebInterface2"});
    }

    @AfterEach
    public void tearDown() {
        tradeConfigJSF = null;
    }

    @Test
    public void testUpdateConfigWithValidInputs() {
        tradeConfigJSF.setRuntimeMode("RuntimeMode1");
        tradeConfigJSF.setOrderProcessingMode("OrderProcessingMode1");
        tradeConfigJSF.setWebInterface("WebInterface1");
        tradeConfigJSF.setMaxUsers(100);
        tradeConfigJSF.setMaxQuotes(1000);
        tradeConfigJSF.setMarketSummaryInterval(60);
        tradeConfigJSF.setPrimIterations(10);
        tradeConfigJSF.setPublishQuotePriceChange(true);
        tradeConfigJSF.setListQuotePriceChangeFrequency(50);
        tradeConfigJSF.setLongRun(true);
        tradeConfigJSF.setDisplayOrderAlerts(true);

        tradeConfigJSF.updateConfig();

        assertEquals("RuntimeMode1", TradeConfig.getRunTimeModeNames()[TradeConfig.getRunTimeMode()]);
        assertEquals("OrderProcessingMode1", TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
        assertEquals("WebInterface1", TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
        assertEquals(100, TradeConfig.getMAX_USERS());
        assertEquals(1000, TradeConfig.getMAX_QUOTES());
        assertEquals(60, TradeConfig.getMarketSummaryInterval());
        assertEquals(10, TradeConfig.getPrimIterations());
        assertTrue(TradeConfig.getPublishQuotePriceChange());
        assertEquals(50, TradeConfig.getListQuotePriceChangeFrequency());
        assertTrue(TradeConfig.getLongRun());
        assertTrue(TradeConfig.getDisplayOrderAlerts());
        assertEquals("DayTrader Configuration Updated", tradeConfigJSF.getResult());
    }

    @Test
    public void testUpdateConfigWithInvalidOrderProcessingMode() {
        tradeConfigJSF.setOrderProcessingMode("InvalidMode");

        try (MockedStatic<Log> mockedLog = mockStatic(Log.class)) {
            tradeConfigJSF.updateConfig();

            mockedLog.verify(() -> Log.error(any(Exception.class), eq("TradeConfigJSF.updateConfig(..): minor exception caught"), eq("trying to set orderProcessing to InvalidMode"), eq("reverting to current value")));
        }

        assertEquals("OrderProcessingMode1", TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
    }

    @Test
    public void testUpdateConfigWithInvalidWebInterface() {
        tradeConfigJSF.setWebInterface("InvalidInterface");

        try (MockedStatic<Log> mockedLog = mockStatic(Log.class)) {
            tradeConfigJSF.updateConfig();

            mockedLog.verify(() -> Log.error(any(Exception.class), eq("TradeConfigJSF.updateConfig(..): minor exception caught"), eq("trying to set WebInterface to InvalidInterface"), eq("reverting to current value")));
        }

        assertEquals("WebInterface1", TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
    }

    @Test
    public void testUpdateConfigWithNullOrderProcessingMode() {
        tradeConfigJSF.setOrderProcessingMode(null);

        tradeConfigJSF.updateConfig();

        assertEquals("OrderProcessingMode1", TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
    }

    @Test
    public void testUpdateConfigWithNullWebInterface() {
        tradeConfigJSF.setWebInterface(null);

        assertEquals("WebInterface1", TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
    }
}
