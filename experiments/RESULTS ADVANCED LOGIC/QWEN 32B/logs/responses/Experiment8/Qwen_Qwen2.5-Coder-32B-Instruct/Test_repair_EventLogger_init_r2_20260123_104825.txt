@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class EventLogger_initTestGenerated {

    @BeforeEach

    }

    @Test
    public void testInitWithNoCargos() {
        when(cargoRepository.findAll()).thenReturn(new ArrayList<>());
        when(locationRepository.findAll()).thenReturn(new ArrayList<>());
        when(voyageRepository.findAll()).thenReturn(new ArrayList<>());

        eventLogger.init();

        assertTrue(eventLogger.getTrackingIds().isEmpty());
        assertTrue(eventLogger.getLocations().isEmpty());
        assertTrue(eventLogger.getVoyages().isEmpty());
    }

    @Test
    public void testInitWithCargosAndLocationsAndVoyages() {
        List<Cargo> cargos = new ArrayList<>();
        Cargo cargo1 = new Cargo(new TrackingId("XYZ"), new HandlingHistory(new ArrayList<>()));
        cargo1.getDelivery().setTransportStatus(TransportStatus.IN_PORT);
        cargos.add(cargo1);

        Cargo cargo2 = new Cargo(new TrackingId("ABC"), new HandlingHistory(new ArrayList<>()));
        cargo2.getDelivery().setTransportStatus(TransportStatus.CLAIMED);
        cargos.add(cargo2);

        List<Location> locations = new ArrayList<>();
        Location location1 = new Location(new UnLocode("USNYC"), "New York");
        locations.add(location1);

        List<Voyage> voyages = new ArrayList<>();
        Voyage voyage1 = new Voyage(new VoyageNumber("0123"), new ArrayList<>());
        voyages.add(voyage1);

        when(cargoRepository.findAll()).thenReturn(cargos);
        when(locationRepository.findAll()).thenReturn(locations);
        when(voyageRepository.findAll()).thenReturn(voyages);

        eventLogger.init();

        assertEquals(2, eventLogger.getTrackingIds().size());
        assertEquals("XYZ", ((SelectItem) eventLogger.getTrackingIds().get(0)).getValue());
        assertEquals("ABC", ((SelectItem) eventLogger.getTrackingIds().get(1)).getValue());

        assertEquals(1, eventLogger.getLocations().size());
        assertEquals("USNYC (USNYC)", ((SelectItem) eventLogger.getLocations().get(0)).getLabel());

        assertEquals(1, eventLogger.getVoyages().size());
        assertEquals("0123", ((SelectItem) eventLogger.getVoyages().get(0)).getValue());
    }

    @Test
