@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
@ExtendWith(MockitoExtension.class)
public class DefaultHandlingEventService_registerHandlingEventTestGenerated {
    @Mock
    private ApplicationEvents applicationEvents;
    @Mock
    private HandlingEventRepository handlingEventRepository;
    @Mock
    private HandlingEventFactory handlingEventFactory;
    @InjectMocks
    private DefaultHandlingEventService defaultHandlingEventService;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    public void testRegisterHandlingEventWithNullCompletionTime() {
        TrackingId trackingId = new TrackingId("XYZ");
        VoyageNumber voyageNumber = new VoyageNumber("V123");
        UnLocode unLocode = new UnLocode("USNYC");
        Type type = Type.LOAD;
        assertThrows(CannotCreateHandlingEventException.class, () -> 
            defaultHandlingEventService.registerHandlingEvent(null, trackingId, voyageNumber, unLocode, type)
        );
        verify(handlingEventRepository, never()).store(any());
        verify(applicationEvents, never()).cargoWasHandled(any());
    }

    @Test
    public void testRegisterHandlingEventWithNullTrackingId() {
        LocalDateTime completionTime = LocalDateTime.now();
        VoyageNumber voyageNumber = new VoyageNumber("V123");
        UnLocode unLocode = new UnLocode("USNYC");
        Type type = Type.LOAD;
        assertThrows(CannotCreateHandlingEventException.class, () -> 
            defaultHandlingEventService.registerHandlingEvent(completionTime, null, voyageNumber, unLocode, type)
        );
        verify(handlingEventRepository, never()).store(any());
        verify(applicationEvents, never()).cargoWasHandled(any());
    }

    @Test

    @Test
    public void testRegisterHandlingEventWithNullType() {
        LocalDateTime completionTime = LocalDateTime.now();
        TrackingId trackingId = new TrackingId("XYZ");
        VoyageNumber voyageNumber = new VoyageNumber("V123");
        UnLocode unLocode = new UnLocode("USNYC");
        assertThrows(CannotCreateHandlingEventException.class, () ->

        );
        verify(handlingEventRepository, never()).store(any());
        verify(applicationEvents, never()).cargoWasHandled(any());
    }

    @Test
    public void testRegisterHandlingEventSuccess() throws CannotCreateHandlingEventException {
        LocalDateTime completionTime = LocalDateTime.now();
        TrackingId trackingId = new TrackingId("XYZ");
        VoyageNumber voyageNumber = new VoyageNumber("V123");
        UnLocode unLocode = new UnLocode("USNYC");
        Type type = Type.LOAD;
        HandlingEvent event = mock(HandlingEvent.class);
        when(handlingEventFactory.createHandlingEvent(any(), eq(completionTime), eq(trackingId), eq(voyageNumber), eq(unLocode), eq(type)))
                .thenReturn(event);
        defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, type);
        verify(handlingEventRepository).store(event);
        verify(applicationEvents).cargoWasHandled(event);
    }

    @Test
    public void testRegisterHandlingEventWithLogger() throws CannotCreateHandlingEventException {
        LocalDateTime completionTime = LocalDateTime.now();
        TrackingId trackingId = new TrackingId("XYZ");
        VoyageNumber voyageNumber = new VoyageNumber("V123");
        UnLocode unLocode = new UnLocode("USNYC");
        Type type = Type.LOAD;
        HandlingEvent event = mock(HandlingEvent.class);
        when(handlingEventFactory.createHandlingEvent(any(), eq(completionTime), eq(trackingId), eq(voyageNumber), eq(unLocode), eq(type)))
                .thenReturn(event);
        java.util.logging.Logger loggerMock = mock(java.util.logging.Logger.class);
        ReflectionTestUtils.setField(defaultHandlingEventService, "logger", loggerMock);
        defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, type);
        verify(loggerMock).info("Registered handling event");
    }

}
