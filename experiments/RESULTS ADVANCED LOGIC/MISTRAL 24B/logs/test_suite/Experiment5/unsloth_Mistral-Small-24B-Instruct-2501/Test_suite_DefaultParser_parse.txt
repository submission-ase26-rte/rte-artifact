package org.apache.commons.cli;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Properties;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Timeout;
import org.apache.commons.cli.Option;
import org.apache.commons.cli.OptionGroup;
import org.apache.commons.cli.ParseException;
import org.apache.commons.cli.UnrecognizedOptionException;
import org.apache.commons.cli.MissingOptionException;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class DefaultParser_parseTestGenerated {

    private DefaultParser parser;
    private Options options;
    private Properties properties;

    @BeforeEach
    void setUp() {
        parser = new DefaultParser();
        options = new Options();
        properties = new Properties();
    }

    @Test
    void testParseWithValidOptionsAndArguments() throws ParseException {
        options.addOption("a", "optionA", true, "description");
        String[] arguments = {"-a", "valueA"};
        CommandLine cmd = parser.parse(options, arguments, properties, false);
        assertNotNull(cmd);
        assertEquals("valueA", cmd.getOptionValue("a"));
    }

    @Test
    void testParseWithInvalidOption() {
        String[] arguments = {"-b"};
        assertThrows(UnrecognizedOptionException.class, () -> parser.parse(options, arguments, properties, false));
    }

    @Test
    void testParseWithStopAtNonOptionTrue() throws ParseException {
        options.addOption("a", "optionA", true, "description");
        String[] arguments = {"-a", "valueA", "extra"};
        CommandLine cmd = parser.parse(options, arguments, properties, true);
        assertNotNull(cmd);
        assertEquals(Arrays.asList("extra"), cmd.getArgList());
    }

    @Test
    void testParseWithStopAtNonOptionFalse() {
        options.addOption("a", "optionA", true, "description");
        String[] arguments = {"-a", "valueA", "extra"};
        assertThrows(ParseException.class, () -> parser.parse(options, arguments, properties, false));
    }

    @Test
    void testParseWithNullArguments() throws ParseException {
        CommandLine cmd = parser.parse(options, null, properties, false);
        assertNotNull(cmd);
        assertTrue(cmd.getArgList().isEmpty());
    }

    @Test
    void testParseWithNullProperties() throws ParseException {
        options.addOption("a", "optionA", true, "description");
        String[] arguments = {"-a", "valueA"};
        CommandLine cmd = parser.parse(options, arguments, null, false);
        assertNotNull(cmd);
        assertEquals("valueA", cmd.getOptionValue("a"));
    }

    @Test
    void testParseWithProperties() throws ParseException {
        options.addOption("a", "optionA", true, "description");
        properties.setProperty("a", "valueFromProperties");
        CommandLine cmd = parser.parse(options, null, properties, false);
        assertNotNull(cmd);
        assertEquals("valueFromProperties", cmd.getOptionValue("a"));
    }

    @Test
    void testParseWithRequiredOptionMissing() {
        Option opt = new Option("a", "optionA", true, "description");
        opt.setRequired(true);
        options.addOption(opt);
        String[] arguments = {};
        assertThrows(MissingOptionException.class, () -> parser.parse(options, arguments, properties, false));
    }

    @Test
    void testParseWithOptionGroup() throws ParseException {
        OptionGroup group = new OptionGroup();
        group.addOption(new Option("a", "optionA", true, "description"));
        group.addOption(new Option("b", "optionB", true, "description"));
        options.addOptionGroup(group);
        String[] arguments = {"-a", "valueA"};
        CommandLine cmd = parser.parse(options, arguments, properties, false);
        assertNotNull(cmd);
        assertEquals("valueA", cmd.getOptionValue("a"));
    }

    @Test
    void testParseWithOptionGroupMissing() {
        OptionGroup group = new OptionGroup();
        group.addOption(new Option("a", "optionA", true, "description"));
        group.addOption(new Option("b", "optionB", true, "description"));
        options.addOptionGroup(group);
        group.setRequired(true);
        String[] arguments = {};
        assertThrows(MissingOptionException.class, () -> parser.parse(options, arguments, properties, false));
    }
}
