public static String crypt(final byte[] original, String salt) {
    if (salt == null) {
        salt = generateRandomSalt();
    } else if (!salt.matches(CRYPT_SALT_REGEX)) {
        throw new IllegalArgumentException("Invalid salt value: " + salt);
    }
    byte[] key = original;
    byte[] saltBytes = salt.getBytes(StandardCharsets.UTF_8);
    byte[] encrypted = new byte[key.length + saltBytes.length];
    System.arraycopy(key, 0, encrypted, 0, key.length);
    System.arraycopy(saltBytes, 0, encrypted, key.length, saltBytes.length);
    byte[] finalResult = desEncrypt(key, encrypted);
    return bytesToHex(finalResult);
}

private static String generateRandomSalt() {
    byte[] saltBytes = new byte[2];
    secureRandom.nextBytes(saltBytes);
    return new String(saltBytes, StandardCharsets.UTF_8);
}

private static byte[] desEncrypt(byte[] key, byte[] data) {
    int[] schedule = desSetKey(key);
    int[] result = body(schedule, 0, 0);
    intToFourBytes(result[0], data, 0);
    intToFourBytes(result[1], data, 4);
    return data;
}

private static String bytesToHex(byte[] bytes) {
    StringBuilder builder = new StringBuilder();
    for (byte b : bytes) {
        builder.append(String.format("%02x", b));
    }
    return builder.toString();
}
