    @BeforeEach

    @Test
    public void testCompleteOrder_SellOrder() throws Exception {
        OrderDataBean order = mock(OrderDataBean.class);
        AccountDataBean account = mock(AccountDataBean.class);
        QuoteDataBean quote = mock(QuoteDataBean.class);
        HoldingDataBean holding = mock(HoldingDataBean.class);
        BigDecimal price = BigDecimal.ONE;
        double quantity = 10;
        when(order.getHolding()).thenReturn(holding);
        when(order.getAccount()).thenReturn(account);
        when(order.getQuote()).thenReturn(quote);
        when(order.getPrice()).thenReturn(price);
        when(order.getQuantity()).thenReturn(quantity);
        when(order.isSell()).thenReturn(true);
        when(order.getOrderID()).thenReturn(1);
        when(order.isCompleted()).thenReturn(false);
        when(entityManager.find(OrderDataBean.class, 1)).thenReturn(order);
        try (MockedStatic<TradeConfig> mockedTradeConfig = mockStatic(TradeConfig.class)) {
            mockedTradeConfig.when(TradeConfig::getRandomPriceChangeFactor).thenReturn(BigDecimal.ONE);
            OrderDataBean result = tradeSLSBBean.completeOrder(1, false);
            assertNotNull(result);
            assertNull(result.getOrderStatus());
            assertNotNull(result.getCompletionDate());
            verify(order).setHolding(null);
            verify(order).setOrderStatus(null);
            verify(order).setCompletionDate(any(Timestamp.class));
            verify(entityManager).remove(holding);
            verify(entityManager).find(OrderDataBean.class, 1);
        }
    }
