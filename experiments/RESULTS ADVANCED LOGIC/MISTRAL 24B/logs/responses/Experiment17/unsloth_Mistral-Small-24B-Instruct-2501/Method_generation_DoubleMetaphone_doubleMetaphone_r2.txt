public String doubleMetaphone(String value, final boolean alternate) {
    if (value == null || value.trim().isEmpty()) {
        return null;
    }

    value = cleanInput(value);
    boolean slavoGermanic = isSlavoGermanic(value);
    int index = 0;
    StringBuilder result = alternate ? alternate : primary;

    while (index < value.length()) {
        char ch = value.charAt(index);
        switch (ch) {
            case 'A':
            case 'E':
            case 'I':
            case 'O':
            case 'U':
            case 'Y':
                index = handleAEIOUY(result, index);
                break;
            case 'B':
                result.append('P');
                if (charAt(value, index + 1) == 'B') {
                    index += 2;
                } else {
                    index++;
                }
                break;
            case 'C':
                index = handleC(value, result, index);
                break;
            case 'D':
                index = handleD(value, result, index);
                break;
            case 'F':
                if (charAt(value, index + 1) == 'F') {
                    result.append('F');
                    index += 2;
                } else {
                    result.append('F');
                    index++;
                }
                break;
            case 'G':
                index = handleG(value, result, index, slavoGermanic);
                break;
            case 'H':
                index = handleH(value, result, index);
                break;
            case 'J':
                index = handleJ(value, result, index, slavoGermanic);
                break;
            case 'K':
                if (charAt(value, index + 1) == 'K') {
                    result.append('K');
                    index += 2;
                } else {
                    result.append('K');
                    index++;
                }
                break;
            case 'L':
                index = handleL(value, result, index);
                break;
            case 'M':
                if (conditionM0(value, index)) {
                    index++;
                } else {
                    result.append('M');
                    index = charAt(value, index + 1) == 'M' ? index + 2 : index + 1;
                }
                break;
            case 'N':
                if (charAt(value, index + 1) == 'N') {
                    result.append('N');
                    index += 2;
                } else {
                    result.append('N');
                    index++;
                }
                break;
            case 'P':
                index = handleP(value, result, index);
                break;
            case 'Q':
                result.append("K");
                index = charAt(value, index + 1) == 'Q' ? index + 2 : index + 1;
                break;
            case 'R':
                index = handleR(value, result, index, slavoGermanic);
                break;
            case 'S':
                index = handleS(value, result, index, slavoGermanic);
                break;
            case 'T':
                index = handleT(value, result, index);
                break;
            case 'V':
                result.append('F');
                index = charAt(value, index + 1) == 'V' ? index + 2 : index + 1;
                break;
            case 'W':
                index = handleW(value, result, index);
                break;
            case 'X':
                index = handleX(value, result, index);
                break;
            case 'Z':
                index = handleZ(value, result, index, slavoGermanic);
                break;
            default:
                index++;
                break;
        }
    }

    if (alternate) {
        return alternate.toString();
    } else {
        return primary.toString();
    }
}
