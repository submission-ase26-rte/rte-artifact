public Pet getPet(String name, boolean ignoreNew) {
    if (name == null) {
        throw new NullPointerException("Pet name must not be null");
    }

    if (name.isEmpty()) {
        return null;
    }

    for (Pet pet : pets) {
        if (ignoreNew && pet.isNew()) {
            continue;
        }
        if (pet.getName().equalsIgnoreCase(name)) {
            return pet;
        }
    }
    return null;
}
