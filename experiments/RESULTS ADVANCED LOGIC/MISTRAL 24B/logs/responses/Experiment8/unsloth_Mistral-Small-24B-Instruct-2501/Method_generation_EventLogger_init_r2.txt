public void init() {
    List<Cargo> cargos = cargoRepository.findAll();
    List<Location> locations = locationRepository.findAll();
    List<Voyage> voyages = voyageRepository.findAll();

    trackingIds = new ArrayList<>();
    locations = new ArrayList<>();
    voyages = new ArrayList<>();

    for (Cargo cargo : cargos) {
        if (cargo.getItinerary() != null && !cargo.getItinerary().isEmpty()) {
            trackingIds.add(new SelectItem(cargo.getTrackingId().toString(), cargo.getTrackingId().toString()));
        }
    }

    for (Location location : locations) {
        locations.add(new SelectItem(location.getUnLocode().toString(), location.getName() + " (" + location.getUnLocode() + ")"));
    }

    for (Voyage voyage : voyages) {
        voyages.add(new SelectItem(voyage.getVoyageNumber().toString(), voyage.getVoyageNumber().toString()));
    }
}
