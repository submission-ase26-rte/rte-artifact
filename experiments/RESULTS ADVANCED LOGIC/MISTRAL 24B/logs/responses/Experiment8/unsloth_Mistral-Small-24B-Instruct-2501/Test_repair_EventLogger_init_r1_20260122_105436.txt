@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class EventLogger_initTestGenerated {

    @Mock
    private CargoRepository cargoRepository;

    @Mock
    private LocationRepository locationRepository;

    @Mock
    private VoyageRepository voyageRepository;

    @InjectMocks
    private EventLogger eventLogger;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        eventLogger = new EventLogger();
        ReflectionTestUtils.setField(eventLogger, "cargoRepository", cargoRepository);
        ReflectionTestUtils.setField(eventLogger, "locationRepository", locationRepository);
        ReflectionTestUtils.setField(eventLogger, "voyageRepository", voyageRepository);
    }

    @Test
    public void testInitWithValidData() {
        TrackingId trackingId1 = new TrackingId("trackingId1");
        TrackingId trackingId2 = new TrackingId("trackingId2");
        Cargo cargo1 = new Cargo(trackingId1, null);
        Cargo cargo2 = new Cargo(trackingId2, null);
        Itinerary itinerary1 = new Itinerary(new ArrayList<>());
        Itinerary itinerary2 = new Itinerary(new ArrayList<>());
        Delivery delivery1 = new Delivery(TransportStatus.DELIVERED, null,  null, null, null);
        Delivery delivery2 = new Delivery(TransportStatus.DELIVERED, null, null, null, null);

        ReflectionTestUtils.setField(cargo1, "itinerary", itinerary1);
        ReflectionTestUtils.setField(cargo1, "delivery", delivery1);
        ReflectionTestUtils.setField(cargo2, "itinerary", itinerary2);
        ReflectionTestUtils.setField(cargo2, "delivery", delivery2);

        List<Cargo> cargos = new ArrayList<>();
        cargos.add(cargo1);
        cargos.add(cargo2);

        UnLocode unLocode1 = new UnLocode("unLocode1");
        Location location1 = new Location(unLocode1, "Location1");

        UnLocode unLocode2 = new UnLocode("unLocode2");
        Location location2 = new Location(unLocode2, "Location2");

        List<Location> locations = new ArrayList<>();
        locations.add(location1);
        locations.add(location2);

        VoyageNumber voyageNumber1 = new VoyageNumber("voyageNumber1");
        Voyage voyage1 = new Voyage(voyageNumber1, new Schedule(new ArrayList<>()));
        VoyageNumber voyageNumber2 = new VoyageNumber("voyageNumber2");
        Voyage voyage2 = new Voyage(voyageNumber2, new Schedule(new ArrayList<>()));

        List<Voyage> voyages = new ArrayList<>();
        voyages.add(voyage1);
        voyages.add(voyage2);

        when(cargoRepository.findAll()).thenReturn(cargos);
        when(locationRepository.findAll()).thenReturn(locations);
        when(voyageRepository.findAll()).thenReturn(voyages);

        eventLogger.init();

        List<SelectItem> trackingIds = eventLogger.getTrackingIds();
        assertEquals(2, trackingIds.size());
        assertEquals("trackingId1", trackingIds.get(0).getValue().toString());
        assertEquals("trackingId2", trackingIds.get(1).getValue().toString());

        List<SelectItem> locationsList = eventLogger.getLocations();
        assertEquals(2, locationsList.size());
        assertEquals("Location1 (unLocode1)", locationsList.get(0).getLabel());
        assertEquals("Location2 (unLocode2)", locationsList.get(1).getLabel());

        List<SelectItem> voyagesList = eventLogger.getVoyages();
        assertEquals(2, voyagesList.size());
        assertEquals("voyageNumber1", voyagesList.get(0).getValue().toString());
        assertEquals("voyageNumber2", voyagesList.get(1).getValue().toString());
    }

    @Test
    public void testInitWithNoCargo() {
        when(cargoRepository.findAll()).thenReturn(new ArrayList<>());

        List<SelectItem> trackingIds = eventLogger.getTrackingIds();

    }
