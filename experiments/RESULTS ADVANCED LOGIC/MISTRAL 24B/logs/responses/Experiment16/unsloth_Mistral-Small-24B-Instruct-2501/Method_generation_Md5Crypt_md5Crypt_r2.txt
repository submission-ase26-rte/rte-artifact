private static String generateSalt(Random random) {
    return B64.getRandomSalt(8);
}

private static boolean isValidSalt(String salt, String prefix) {
    return salt.startsWith(prefix) && salt.length() == prefix.length() + 8;
}

private static String md5(String input, MessageDigest md5Digest) {
    md5Digest.reset();
    md5Digest.update(input.getBytes(StandardCharsets.UTF_8));
    return B64.encodeBase64String(md5Digest.digest());
}
