    if (txt == null || txt.isEmpty()) {
        return "";
    }

    StringBuilder metaphone = new StringBuilder();
    int length = txt.length();
    int prevIndex = 0;
    int currentIndex = 0;

    boolean prevWasVowel = false;

    while (currentIndex < length) {
        char currentChar = txt.charAt(currentIndex);
        boolean currentIsVowel = isVowel(metaphone, currentIndex);

        if (currentIsVowel) {
            metaphone.append(currentChar);
            prevWasVowel = true;
        } else {
            if (prevWasVowel) {
                prevWasVowel = false;
            }

            if (currentChar == 'C') {

                        currentIndex += 2;
