package org.apache.commons.compress.archivers.ar;

import static org.junit.jupiter.api.Assertions.*;

import java.io.InputStream;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class ArArchiveInputStream_matchesTestGenerated {

    private byte[] signature;

    @BeforeEach
    public void setUp() {
        signature = new byte[8];
    }

    @Test
    public void testMatchesValidSignature() {
        signature[0] = 0x21;
        signature[1] = 0x3c;
        signature[2] = 0x61;
        signature[3] = 0x72;
        signature[4] = 0x63;
        signature[5] = 0x68;
        signature[6] = 0x3e;
        signature[7] = 0x0a;

        boolean result = ArArchiveInputStream.matches(signature, 8);
        assertTrue(result);
    }

    @Test
    public void testMatchesInvalidSignature() {
        signature[0] = 0x22;
        signature[1] = 0x3c;
        signature[2] = 0x61;
        signature[3] = 0x72;
        signature[4] = 0x63;
        signature[5] = 0x68;
        signature[6] = 0x3e;
        signature[7] = 0x0a;

        boolean result = ArArchiveInputStream.matches(signature, 8);
        assertFalse(result);
    }

    @Test
    public void testMatchesLengthLessThanEight() {
        boolean result = ArArchiveInputStream.matches(signature, 7);
        assertFalse(result);
    }

    @Test
    public void testMatchesNullSignature() {
        try {
            ArArchiveInputStream.matches(null, 8);
            fail("Expected NullPointerException");
        } catch (NullPointerException e) {
            assertEquals("signature is marked non-null but got null", e.getMessage());
        }
    }

    @Test
    public void testMatchesSignatureLengthZero() {
        signature = new byte[0];
        boolean result = ArArchiveInputStream.matches(signature, 0);
        assertFalse(result);
    }

    @Test
    public void testMatchesSignatureLengthNegative() {
        assertThrows(IllegalArgumentException.class, () -> ArArchiveInputStream.matches(signature, -1));
    }
}
