            return ((Collection) object).toArray((Object[]) Array.newInstance(componentType, 0));
        }
        if (object.getClass().isArray()) {
            return Arrays.copyOf((Object[]) object, ((Object[]) object).length, (Class<?>) toType);
        }
    }

    if (ConvertUtils.canConvert(object, toType)) {
        return ConvertUtils.convert(object, toType);
    }

    if (toType == String.class) {
        return String.valueOf(object);
    }

    throw new JXPathTypeConversionException("Cannot convert " + object + " to " + toType);
}
