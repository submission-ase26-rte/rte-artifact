package org.eclipse.cargotracker.interfaces.handling.mobile;

import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.verify;
import static org.mockito.ArgumentMatchers.arg;

import jakarta.annotation.PostConstruct;
import jakarta.faces.application.FacesMessage;
import jakarta.faces.context.FacesContext;
import jakarta.inject.Inject;
import jakarta.inject.Named;
import java.time.LocalDateTime;
import java.util.List;
import org.eclipse.cargotracker.application.ApplicationEvents;
import org.eclipse.cargotracker.domain.model.cargo.Cargo;
import org.eclipse.cargotracker.domain.model.cargo.TrackingId;
import org.eclipse.cargotracker.domain.model.handling.HandlingEvent;
import org.eclipse.cargotracker.domain.model.location.Location;
import org.eclipse.cargotracker.domain.model.location.UnLocode;
import org.eclipse.cargotracker.domain.model.voyage.Voyage;
import org.eclipse.cargotracker.domain.model.voyage.VoyageNumber;
import org.eclipse.cargotracker.interfaces.handling.HandlingEventRegistrationAttempt;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mockito;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class EventLogger_submitTestGenerated {

  private EventLogger eventLogger;
  private ApplicationEvents applicationEvents;

  @BeforeEach
  public void setUp() {
    applicationEvents = mock(ApplicationEvents.class);
    eventLogger = new EventLogger();
  }

  @AfterEach
  public void tearDown() {
    Mockito.resetMocks(applicationEvents);
  }

  @Test
  public void testSubmitWithVoyage() {
    eventLogger.setTrackingId("12345");
    eventLogger.setLocation("loc1");
    eventLogger.setEventType(HandlingEvent.Type.LOAD);
    eventLogger.setVoyageNumber("voyage1");
    eventLogger.setCompletionTime(LocalDateTime.now());

    eventLogger.submit();

    verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
        arg -> arg.getTrackingId().getValue().equals("12345")
            && arg.getVoyage().getValue().equals("voyage1")
            && arg.getEventType() == HandlingEvent.Type.LOAD
            && arg.getLocation().getValue().equals("loc1"));
  }

  @Test
  public void testSubmitWithoutVoyage() {
    eventLogger.setTrackingId("12345");
    eventLogger.setLocation("loc1");
    eventLogger.setEventType(HandlingEvent.Type.CLAIM);
    eventLogger.setVoyageNumber(null);
    eventLogger.setCompletionTime(LocalDateTime.now());

    eventLogger.submit();

    verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
        arg -> arg.getTrackingId().getValue().equals("12345")
            && arg.getVoyage() == null
            && arg.getEventType() == HandlingEvent.Type.CLAIM
            && arg.getLocation().getValue().equals("loc1"));
  }

  @Test
  public void testSubmitWithNullValues() {
    eventLogger.setTrackingId(null);
    eventLogger.setLocation(null);
    eventLogger.setEventType(HandlingEvent.Type.UNLOAD);
    eventLogger.setVoyageNumber("voyage1");
    eventLogger.setCompletionTime(LocalDateTime.now());

    eventLogger.submit();

    verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
        arg -> arg.getTrackingId().getValue() == null
            && arg.getVoyage().getValue().equals("voyage1")
            && arg.getEventType() == HandlingEvent.Type.UNLOAD
            && arg.getLocation().getValue() == null);
  }
}
