package org.eclipse.cargotracker.application.internal;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;
import static org.mockito.Mockito.doNothing;
import static org.mockito.Mockito.when;

import java.time.LocalDateTime;
import java.util.logging.Logger;
import jakarta.inject.Inject;
import org.eclipse.cargotracker.application.ApplicationEvents;
import org.eclipse.cargotracker.domain.model.cargo.TrackingId;
import org.eclipse.cargotracker.domain.model.handling.CannotCreateHandlingEventException;
import org.eclipse.cargotracker.domain.model.handling.HandlingEvent;
import org.eclipse.cargotracker.domain.model.handling.HandlingEventFactory;
import org.eclipse.cargotracker.domain.model.handling.HandlingEventRepository;
import org.eclipse.cargotracker.domain.model.location.UnLocode;
import org.eclipse.cargotracker.domain.model.voyage.VoyageNumber;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mockito;
import org.mockito.junit.jupiter.MockitoExtension;

@MockitoExtension
public class DefaultHandlingEventService_registerHandlingEventTestGenerated {

    @Inject
    private ApplicationEvents applicationEvents;
    @Inject
    private HandlingEventRepository handlingEventRepository;
    @Inject
    private HandlingEventFactory handlingEventFactory;
    @Inject
    private Logger logger;
    private DefaultHandlingEventService defaultHandlingEventService;

    @BeforeEach
    public void setUp() {
        applicationEvents = Mockito.mock(ApplicationEvents.class);
        handlingEventRepository = Mockito.mock(HandlingEventRepository.class);
        handlingEventFactory = Mockito.mock(HandlingEventFactory.class);
        logger = Mockito.mock(Logger.class);
        defaultHandlingEventService = new DefaultHandlingEventService(applicationEvents, handlingEventRepository, handlingEventFactory, logger);
    }

    @AfterEach
    public void tearDown() {
        Mockito.resetMocks(applicationEvents, handlingEventRepository, handlingEventFactory, logger);
    }

    @Test
    public void registerHandlingEvent_success() throws CannotCreateHandlingEventException {
        LocalDateTime completionTime = LocalDateTime.now();
        TrackingId trackingId = new TrackingId("123");
        VoyageNumber voyageNumber = new VoyageNumber("Voyage1");
        UnLocode unLocode = new UnLocode("locode1");
        HandlingEvent.Type type = HandlingEvent.Type.LOAD;

        HandlingEvent event = Mockito.mock(HandlingEvent.class);
        when(handlingEventFactory.createHandlingEvent(Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any()))
                .thenReturn(event);

        defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, type);

        Mockito.verify(handlingEventRepository).store(event);
        Mockito.verify(applicationEvents).cargoWasHandled(event);
        Mockito.verify(logger).info("Registered handling event");
    }

    @Test
    public void registerHandlingEvent_cannotCreateHandlingEventException() {
        LocalDateTime completionTime = LocalDateTime.now();
        TrackingId trackingId = new TrackingId("123");
        VoyageNumber voyageNumber = new VoyageNumber("Voyage1");
        UnLocode unLocode = new UnLocode("locode1");
        HandlingEvent.Type type = HandlingEvent.Type.LOAD;

        when(handlingEventFactory.createHandlingEvent(Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any()))
                .thenThrow(CannotCreateHandlingEventException.class);

        assertThrows(CannotCreateHandlingEventException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, type);
        });
    }

    @Test
    public void registerHandlingEvent_nullInput() {
        LocalDateTime completionTime = null;
        TrackingId trackingId = new TrackingId("123");
        VoyageNumber voyageNumber = new VoyageNumber("Voyage1");
        UnLocode unLocode = new UnLocode("locode1");
        HandlingEvent.Type type = HandlingEvent.Type.LOAD;

        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, type);
        });
    }
}
