import java.io.InputStream;
import org.apache.commons.compress.archivers.ArchiveException;
import org.apache.commons.compress.archivers.ar.ArArchiveInputStream;
import org.apache.commons.compress.archivers.arj.ArjArchiveInputStream;
import org.apache.commons.compress.archivers.cpio.CpioArchiveInputStream;
import org.apache.commons.compress.archivers.tar.TarArchiveInputStream;
import org.apache.commons.compress.archivers.zip.ZipArchiveInputStream;
import org.apache.commons.compress.utils.IOUtils;

public static String detect(final InputStream in) throws ArchiveException {
    if (in == null) {
        throw new IllegalArgumentException("Input stream cannot be null");
    }
    try {
        byte[] signature = new byte[4];
        int read = in.read(signature);
        if (read < 0) {
            throw new IllegalArgumentException("No mark support from input stream");
        }
        if (ArArchiveInputStream.matches(signature, 4)) {
            return AR;
        }
        if (ArjArchiveInputStream.matches(signature, 4)) {
            return ARJ;
        }
        if (CpioArchiveInputStream.matches(signature, 4)) {
            return CPIO;
        }
        if (TarArchiveInputStream.matches(signature, 4)) {
            return TAR;
        }
        if (ZipArchiveInputStream.matches(signature, 4)) {
            return ZIP;
        }
        if (SevenZFile.matches(signature, 4)) {
            return SEVEN_Z;
        }
        throw new ArchiveException("Unknown archive format");
    } catch (IOException e) {
        throw new ArchiveException("IO Exception during archive detection", e);
    }
}
