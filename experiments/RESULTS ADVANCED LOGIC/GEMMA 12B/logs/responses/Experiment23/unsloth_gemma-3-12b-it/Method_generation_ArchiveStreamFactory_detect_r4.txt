import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.util.Arrays;

import org.apache.commons.compress.archivers.ArchiveException;
import org.apache.commons.compress.archivers.ArchiveStreamFactory;
import org.apache.commons.compress.archivers.ar.ArArchiveInputStream;
import org.apache.commons.compress.archivers.arj.ArjArchiveInputStream;
import org.apache.commons.compress.archivers.cpio.CpioArchiveInputStream;
import org.apache.commons.compress.archivers.tar.TarArchiveInputStream;
import org.apache.commons.compress.archivers.zip.ZipArchiveInputStream;
import org.apache.commons.compress.utils.IOUtils;

public static String detect(final InputStream in) throws ArchiveException {
        if (in == null) {
            throw new IllegalArgumentException("Input stream cannot be null");
        }
        try {
            byte[] signature = new byte[SIGNATURE_SIZE];
            int read = in.read(signature);
            if (read < 0) {
                throw new IllegalArgumentException("No mark support");
            }
            if (Arrays.equals(signature, ArArchiveInputStream.SUPPORTED_AR_SIGNATURE)) {
                return AR;
            }
            if (ArjArchiveInputStream.matches(signature, SIGNATURE_SIZE)) {
                return ARJ;
            }
            if (CpioArchiveInputStream.matches(signature, SIGNATURE_SIZE)) {
                return CPIO;
            }
            if (TarArchiveInputStream.matches(signature, SIGNATURE_SIZE)) {
                return TAR;
            }
            if (ZipArchiveInputStream.matches(signature, SIGNATURE_SIZE)) {
                return ZIP;
            }
            if (SevenZFile.matches(signature, SIGNATURE_SIZE)) {
                return SEVEN_Z;
            }
            if (DumpArchiveInputStream.matches(signature, SIGNATURE_SIZE)) {
                return DUMP;
            }
            throw new ArchiveException("Unknown archive format");
        } catch (IOException e) {
            throw new ArchiveException("IOException while detecting archive type", e);
        }
    }
