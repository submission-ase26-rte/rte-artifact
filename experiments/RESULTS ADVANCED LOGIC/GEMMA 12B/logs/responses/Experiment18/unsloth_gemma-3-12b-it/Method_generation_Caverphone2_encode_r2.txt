public String encode(final String source) {
        if (source == null) {
            throw new NullPointerException();
        }
        if (SoundexUtils.isEmpty(source)) {
            return "TEN_1";
        }
        return encode(source, 0);
    }

    private String encode(final String source, final int level) {
        if (SoundexUtils.isEmpty(source)) {
            return "TEN_1";
        }
        if (level > 2) {
            return source;
        }
        String s = source.toLowerCase();
        char[] chars = s.toCharArray();
        StringBuilder sb = new StringBuilder();
        for (char c : chars) {
            if (Character.isLetter(c)) {
                sb.append(c);
            }
        }
        String cleaned = sb.toString();

        if (SoundexUtils.isEmpty(cleaned)) {
            return "TEN_1";
        }

        String firstLetter = cleaned.substring(0, 1);
        String rest = cleaned.substring(1);

        StringBuilder encoded = new StringBuilder(firstLetter);

        for (char c : rest.toCharArray()) {
            char encodedChar = encodeChar(c);
            if (encodedChar != '0') {
                encoded.append(encodedChar);
            }
        }
        
        String encodedString = encoded.toString();

        if (encodedString.length() > 4) {
            encodedString = encodedString.substring(0, 4);
        }

        while (encodedString.length() < 4) {
            encodedString += "0";
        }

        return encodedString;
    }

    private char encodeChar(char c) {
        switch (c) {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                return '0';
            case 'b':
            case 'f':
            case 'p':
            case 'v':
                return '1';
            case 'c':
            case 'g':
            case 'j':
            case 'k':
            case 'q':
            case 's':
            case 'x':
            case 'z':
                return '2';
            case 'd':
            case 't':
                return '3';
            case 'l':
                return '4';
            case 'm':
            case 'n':
                return '5';
            case 'r':
                return '6';
            default:
                return '0';
        }
    }
