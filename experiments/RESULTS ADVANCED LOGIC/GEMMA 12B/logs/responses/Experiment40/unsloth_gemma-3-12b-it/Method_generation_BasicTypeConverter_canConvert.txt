public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        return toType.isAssignableFrom(null);
    }

    if (toType == Boolean.class) {
        return object instanceof Number;
    }

    if (toType == Number.class) {
        return object instanceof Boolean || object instanceof Number;
    }

    if (toType.isPrimitive()) {
        if (object instanceof Number) {
            return true;
        }
        if (object instanceof String) {
            try {
                Number number = ConvertUtils.toNumber((String) object, toType);
                return true;
            } catch (JXPathTypeConversionException e) {
                return false;
            }
        }
        return false;
    }

    if (toType.isArray()) {
        if (object instanceof Collection) {
            Collection collection = (Collection) object;
            if (collection.isEmpty()) {
                return false;
            }
            Object firstElement = collection.iterator().next();
            return firstElement.getClass().isAssignableFrom(toType.getComponentType());
        }
        if (object instanceof Object[]) {
            Object[] array = (Object[]) object;
            if (array.length == 0) {
                return false;
            }
            return array[0].getClass().isAssignableFrom(toType.getComponentType());
        }
        return false;
    }

    if (toType.isInterface()) {
        return toType.isAssignableFrom(object.getClass());
    }

    return toType.isAssignableFrom(object.getClass());
}
