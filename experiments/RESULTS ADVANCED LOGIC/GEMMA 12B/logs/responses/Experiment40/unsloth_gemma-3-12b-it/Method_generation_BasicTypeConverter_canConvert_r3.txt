public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        return toType.isAssignableFrom(null);
    }

    if (toType == boolean.class || toType == Boolean.class) {
        return object instanceof Number;
    }

    if (toType == Integer.class) {
        return object instanceof Integer || object instanceof Long || object instanceof Double || object instanceof String || object instanceof Boolean;
    }

    if (toType == Double.class) {
        return object instanceof Double || object instanceof String;
    }

    if (toType == java.util.concurrent.atomic.AtomicBoolean.class) {
        return object instanceof Boolean;
    }

    if (toType == BigDecimal.class) {
        return object instanceof String;
    }

    if (toType == BigInteger.class) {
        return object instanceof String;
    }

    if (object.getClass().isArray()) {
        if (toType == List.class || toType == Collection.class) {
            return true;
        }
        if (toType == String.class) {
            return true;
        }
        return false;
    }

    if (object instanceof Collection) {
        if (toType == List.class || toType == Collection.class) {
            return true;
        }
        return false;
    }

    if (object instanceof Pointer) {
        return toType == Integer.class;
    }

    if (object instanceof NodeSet) {
        return toType == Integer.class;
    }
    
    return false;
}
