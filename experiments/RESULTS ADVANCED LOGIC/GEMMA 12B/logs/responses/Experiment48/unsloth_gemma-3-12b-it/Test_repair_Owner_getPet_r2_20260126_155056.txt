class Owner_getPetTestGenerated {

    private Owner owner;
    private Pet pet1;
    private Pet pet2;

    @BeforeEach
    void setUp() {
        owner = new Owner();
        List<Pet> pets = new ArrayList<>();
        pet1 = new Pet();
        pet1.setName("Buddy");
        pet2 = new Pet();
        pet2.setName("buddy");
        pets.add(pet1);
        pets.add(pet2);
        ReflectionTestUtils.setField(owner, "pets", pets);
    }

    @Test
    void testGetPetExistingNameIgnoreCase() {
        Pet result = owner.getPet("buddy", false);
        assertNotNull(result);
        assertEquals("Buddy", result.getName());
    }

    @Test
    void testGetPetExistingNameIgnoreNewTrue() {
        Pet result = owner.getPet("buddy", true);
        assertNotNull(result);
        assertEquals("Buddy", result.getName());
    }

    @Test
    void testGetPetNewPetIgnoreNewFalse() {
        pet1.setBirthDate(LocalDate.now());
        Pet result = owner.getPet("Buddy", false);
        assertNotNull(result);
        assertEquals("Buddy", result.getName());
    }

    @Test
    void testGetPetNewPetIgnoreNewTrue() {
        pet1.setBirthDate(LocalDate.now());
        Pet result = owner.getPet("Buddy", true);
        assertNotNull(result);
        assertEquals("Buddy", result.getName());
    }

    @Test
    void testGetPetNonExistingName() {

        assertNull(result);
    }
