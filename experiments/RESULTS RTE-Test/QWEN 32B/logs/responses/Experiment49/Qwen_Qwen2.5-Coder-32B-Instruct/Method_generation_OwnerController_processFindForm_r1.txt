public String processFindForm(@RequestParam(defaultValue = "1") int page, Owner owner, BindingResult result, Model model) {
    String lastName = owner.getLastName();
    if (lastName == null || lastName.trim().isEmpty()) {
        result.rejectValue("lastName", "notFound", "not found");
        return "owners/findOwners";
    }
    Page<Owner> paginatedOwners = findPaginatedForOwnersLastName(page, lastName);
    if (paginatedOwners.getTotalElements() == 0) {
        result.rejectValue("lastName", "notFound", "not found");
        return "owners/findOwners";
    } else if (paginatedOwners.getTotalElements() == 1) {
        return "redirect:/owners/" + paginatedOwners.getContent().get(0).getId();
    } else {
        return addPaginationModel(page, model, paginatedOwners);
    }
}
