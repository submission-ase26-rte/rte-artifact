public static String md5Crypt(final byte[] keyBytes, final String salt, final String prefix, final Random random) {
    if (keyBytes == null) {
        throw new NullPointerException();
    }
    if (salt == null) {
        StringBuilder saltBuilder = new StringBuilder(prefix);
        for (int i = 0; i < 8; i++) {
            saltBuilder.append(B64.B64T_ARRAY[random.nextInt(64)]);
        }
        return md5Crypt(keyBytes, saltBuilder.toString(), prefix, random);
    }
    if (salt.length() < 2) {
        throw new IllegalArgumentException("Invalid salt value: " + salt);
    }
    StringBuilder saltBuilder = new StringBuilder(prefix);
    int saltLength = 0;
    for (int i = 0; i < salt.length() && salt.charAt(i) != '$'; i++) {
        if (saltLength < 8) {
            saltBuilder.append(salt.charAt(i));
            saltLength++;
        }
    }
    String saltValue = saltBuilder.toString();
    MessageDigest md = DigestUtils.getMd5Digest();
    md.update(keyBytes);
    md.update(saltValue.getBytes(StandardCharsets.UTF_8));
    byte[] hash = md.digest();
    md.reset();
    md.update(keyBytes);
    md.update(saltValue.getBytes(StandardCharsets.UTF_8));
    md.update(hash);
    for (int i = keyBytes.length; i > 0; i -= BLOCKSIZE) {
        md.update(hash, 0, i > BLOCKSIZE ? BLOCKSIZE : i);
    }
    for (int i = keyBytes.length; i > 0; i >>= 1) {
        if ((i & 1) == 1) {
            md.update(hash, 0, 1);
        } else {
            md.update(keyBytes, 0, 1);
        }
    }
    hash = md.digest();
    for (int i = 0; i < 1000; i++) {
        md.reset();
        if ((i & 1) == 1) {
            md.update(keyBytes);
        } else {
            md.update(hash);
        }
        if (i % 3 != 0) {
            md.update(saltValue.getBytes(StandardCharsets.UTF_8));
        }
        if (i % 7 != 0) {
            md.update(keyBytes);
        }
        if ((i & 1) == 1) {
            md.update(hash);
        } else {
            md.update(keyBytes);
        }
        hash = md.digest();
    }
    StringBuilder result = new StringBuilder(saltValue);
    B64.b64from24bit(hash[0], hash[6], hash[12], 4, result);
    B64.b64from24bit(hash[1], hash[7], hash[13], 4, result);
    B64.b64from24bit(hash[2], hash[8], hash[14], 4, result);
    B64.b64from24bit(hash[3], hash[9], hash[15], 4, result);
    B64.b64from24bit(hash[4], hash[10], hash[5], 4, result);
    B64.b64from24bit((byte) 0, (byte) 0, hash[11], 2, result);
    return result.toString();
}
