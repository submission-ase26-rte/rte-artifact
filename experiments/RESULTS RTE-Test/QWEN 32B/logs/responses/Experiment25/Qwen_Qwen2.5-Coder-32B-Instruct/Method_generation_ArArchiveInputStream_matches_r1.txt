public static boolean matches(final byte[] signature, final int length) {
    if (signature == null || length < 8) {
        return false;
    }
    return signature[0] == '!' && signature[1] == '<' && signature[2] == 'a' &&
           signature[3] == 'r' && signature[4] == 'c' && signature[5] == 'h' &&
           signature[6] == '>' && signature[7] == '\n';
}
