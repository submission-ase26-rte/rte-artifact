public static Options parsePattern(final String pattern) {
    if (pattern == null) {
        throw new NullPointerException("pattern cannot be null");
    }

    Options options = new Options();
    StringBuilder opt = new StringBuilder();
    boolean hasArg = false;
    boolean isRequired = false;
    Class<?> type = null;

    for (int i = 0; i < pattern.length(); i++) {
        char ch = pattern.charAt(i);
        if (isValueCode(ch)) {
            if (opt.length() > 0) {
                Option option = Option.builder(opt.toString())
                        .hasArg(hasArg)
                        .required(isRequired)
                        .type(type)
                        .build();
                options.addOption(option);
                opt.setLength(0);
                hasArg = false;
                isRequired = false;
                type = null;
            }
            if (ch == '!') {
                isRequired = true;
            } else {
                hasArg = true;
                type = getValueType(ch);
            }
        } else {
            opt.append(ch);
        }
    }

    if (opt.length() > 0) {
        Option option = Option.builder(opt.toString())
                .hasArg(hasArg)
                .required(isRequired)
                .type(type)
                .build();
        options.addOption(option);
    }

    return options;
}
