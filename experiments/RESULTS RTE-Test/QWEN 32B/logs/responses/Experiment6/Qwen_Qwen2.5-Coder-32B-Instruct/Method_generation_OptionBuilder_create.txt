public static Option create(final String opt) throws IllegalArgumentException {
    if (opt == null || opt.length() != 1) {
        throw new IllegalArgumentException("opt cannot be null or more than one character");
    }
    Option.Builder builder = new Option.Builder(opt);
    builder.setLongOpt(longOption);
    builder.setDescription(description);
    builder.setArgName(argName);
    builder.setRequired(required);
    builder.setArgs(argCount);
    builder.setType(type);
    builder.setOptionalArg(optionalArg);
    builder.setValueSeparator(valueSeparator);
    return builder.create();
}
