public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        return true;
    }
    if (toType.isInstance(object)) {
        return true;
    }
    if (toType.isAssignableFrom(object.getClass())) {
        return true;
    }
    if (object instanceof String) {
        if (toType == Integer.class || toType == Long.class || toType == Double.class || toType == Float.class || toType == Byte.class || toType == Short.class || toType == Character.class || toType == Boolean.class) {
            return true;
        }
    }
    if (object instanceof Number) {
        if (toType == String.class || toType == Boolean.class) {
            return true;
        }
    }
    if (object instanceof Boolean) {
        if (toType == String.class || toType == Integer.class || toType == Long.class || toType == Double.class || toType == Float.class || toType == Byte.class || toType == Short.class || toType == Character.class) {
            return true;
        }
    }
    if (object.getClass().isArray()) {
        if (toType.isArray() && toType.getComponentType().isAssignableFrom(object.getClass().getComponentType())) {
            return true;
        }
        if (Collection.class.isAssignableFrom(toType)) {
            return true;
        }
        if (toType.isAssignableFrom(object.getClass().getComponentType())) {
            return true;
        }
    }
    if (object instanceof Collection) {
        if (toType.isArray() && toType.getComponentType().isAssignableFrom(((Collection<?>) object).iterator().next().getClass())) {
            return true;
        }
        if (Collection.class.isAssignableFrom(toType)) {
            return true;
        }
        if (toType.isAssignableFrom(((Collection<?>) object).iterator().next().getClass())) {
            return true;
        }
    }
    if (object instanceof Pointer) {
        return canConvert(((Pointer) object).getValue(), toType);
    }
    if (object instanceof NodeSet) {
        return canConvert(((NodeSet) object).getValues().iterator().next(), toType);
    }
    if (ConvertUtils.lookup(toType) != null) {
        return true;
    }
    return false;
}
