package org.apache.commons.compress.archivers;

import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.io.InputStream;
import org.apache.commons.compress.archivers.ArchiveException;
import org.apache.commons.compress.archivers.ArchiveStreamFactory;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class ArchiveStreamFactory_detectTestGenerated {
    @Mock
    private InputStream mockInputStream;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    void testDetectJarArchive() throws ArchiveException {
        byte[] jarSignature = new byte[]{80, 75, 3, 4};
        InputStream in = new ByteArrayInputStream(jarSignature);
        assertEquals(ArchiveStreamFactory.ZIP, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectArArchive() throws ArchiveException {
        byte[] arSignature = new byte[]{33, 100, 101, 98, 105, 97, 110};
        InputStream in = new ByteArrayInputStream(arSignature);
        assertEquals(ArchiveStreamFactory.AR, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectCpioArchive() throws ArchiveException {
        byte[] cpioSignature = new byte[]{99, 112, 105, 111, 32, 110, 111, 99, 114, 99, 32};
        InputStream in = new ByteArrayInputStream(cpioSignature);
        assertEquals(ArchiveStreamFactory.CPIO, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectArjArchive() throws ArchiveException {
        byte[] arjSignature = new byte[]{10, 22, 4, 20};
        InputStream in = new ByteArrayInputStream(arjSignature);
        assertEquals(ArchiveStreamFactory.ARJ, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectDumpArchive() throws ArchiveException {
        byte[] dumpSignature = new byte[32];
        dumpSignature[0] = 116;
        dumpSignature[1] = 97;
        dumpSignature[2] = 112;
        dumpSignature[3] = 101;
        InputStream in = new ByteArrayInputStream(dumpSignature);
        assertEquals(ArchiveStreamFactory.DUMP, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectIOExceptionWhileReadingSignature() {
        InputStream in = mock(InputStream.class);
        when(in.markSupported()).thenReturn(true);
        when(in.mark(anyInt())).thenReturn(null);
        when(in.read(any(byte[].class))).thenThrow(new IOException("Mocked IOException"));
        assertThrows(ArchiveException.class, () -> ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectIOExceptionWhileReadingDumpSignature() {
        InputStream in = mock(InputStream.class);
        when(in.markSupported()).thenReturn(true);
        when(in.mark(anyInt())).thenReturn(null);
        when(in.read(any(byte[].class), eq(0), eq(ArchiveStreamFactory.DUMP_SIGNATURE_SIZE))).thenReturn(ArchiveStreamFactory.DUMP_SIGNATURE_SIZE);
        when(in.read(any(byte[].class), eq(ArchiveStreamFactory.DUMP_SIGNATURE_SIZE), anyInt())).thenThrow(new IOException("Mocked IOException"));
        assertThrows(ArchiveException.class, () -> ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectIOExceptionWhileReadingTarSignature() {
        InputStream in = mock(InputStream.class);
        when(in.markSupported()).thenReturn(true);
        when(in.mark(anyInt())).thenReturn(null);
        when(in.read(any(byte[].class), eq(0), eq(ArchiveStreamFactory.TAR_HEADER_SIZE))).thenReturn(ArchiveStreamFactory.TAR_HEADER_SIZE);
        when(in.read(any(byte[].class), eq(ArchiveStreamFactory.TAR_HEADER_SIZE), anyInt())).thenThrow(new IOException("Mocked IOException"));
        assertThrows(ArchiveException.class, () -> ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectJarMatchesButReturnsZip() throws ArchiveException {
        byte[] jarSignature = new byte[]{80, 75, 3, 4};
        InputStream in = new ByteArrayInputStream(jarSignature);
        assertEquals(ArchiveStreamFactory.ZIP, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectTarWithValidHeader() throws ArchiveException {
        byte[] tarSignature = new byte[ArchiveStreamFactory.TAR_HEADER_SIZE];
        InputStream in = new ByteArrayInputStream(tarSignature);
        assertEquals(ArchiveStreamFactory.TAR, ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectTarWithInvalidHeader() {
        byte[] tarSignature = new byte[ArchiveStreamFactory.TAR_HEADER_SIZE];
        InputStream in = new ByteArrayInputStream(tarSignature);
        assertThrows(ArchiveException.class, () -> ArchiveStreamFactory.detect(in));
    }

    @Test
    void testDetectTarWithIOExceptionInEntryProcessing() {
        byte[] tarSignature = new byte[ArchiveStreamFactory.TAR_HEADER_SIZE];
        InputStream in = mock(InputStream.class);
        when(in.markSupported()).thenReturn(true);
        when(in.mark(anyInt())).thenReturn(null);
        when(in.read(any(byte[].class))).thenReturn(ArchiveStreamFactory.TAR_HEADER_SIZE);
        when(in.reset()).thenReturn(null);
        assertThrows(ArchiveException.class, () -> ArchiveStreamFactory.detect(in));
    }
}
