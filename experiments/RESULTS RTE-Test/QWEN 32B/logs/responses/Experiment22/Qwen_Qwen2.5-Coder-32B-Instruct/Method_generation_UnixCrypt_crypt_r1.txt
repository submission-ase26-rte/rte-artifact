public static String crypt(final byte[] original, String salt) {
    if (salt == null) {
        salt = generateSalt();
    } else if (salt.length() < 2 || !CRYPT_SALT_PATTERN.matcher(salt).matches()) {
        throw new IllegalArgumentException("Invalid salt value: " + salt);
    }
    salt = salt.substring(0, 2);

    byte[] keyBytes = new byte[8];
    System.arraycopy(original, 0, keyBytes, 0, Math.min(original.length, 8));

    int[] keySchedule = desSetKey(keyBytes);
    int[] result = body(keySchedule, 0, 0);

    byte[] output = new byte[13];
    output[0] = salt.getBytes(StandardCharsets.UTF_8)[0];
    output[1] = salt.getBytes(StandardCharsets.UTF_8)[1];
    intToFourBytes(result[0], output, 2);
    intToFourBytes(result[1], output, 6);

    StringBuilder sb = new StringBuilder();
    for (int i = 0; i < 13; i++) {
        sb.append(SALT_CHARS[(output[i] & 0x3f)]);
    }

    return sb.toString();
}

private static String generateSalt() {
    SecureRandom random = new SecureRandom();
    byte[] saltBytes = new byte[2];
    random.nextBytes(saltBytes);
    StringBuilder salt = new StringBuilder();
    for (byte b : saltBytes) {
        salt.append(SALT_CHARS[(b & 0x3f)]);
    }
    return salt.toString();
}
