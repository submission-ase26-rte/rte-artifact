        optionGroup = new OptionGroup();

    }

    @Test
    public void testToStringWithSingleLongOption() {
        Option option = new Option(null, "longOption", false, "description");

    }

    public void testToStringWithMixedOptions() {
        Option option1 = new Option("a", null, false, "description1");

        optionGroup.addOption(option2);

    }

    }

    public void testToStringWithOnlyShortOptions() {
