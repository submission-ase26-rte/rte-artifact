    public String asPath() {
        if (node == null || node.getNodeType() == Node.DOCUMENT_NODE) {
            return "";
        }

        if (node.getNodeType() == Node.ELEMENT_NODE) {
            if (id != null) {
                return "id('" + id + "')";
            }

            String localName = node.getLocalName();
            if (localName == null) {
                localName = node.getNodeName();
                if (localName.indexOf(':') != -1) {
                    localName = localName.substring(localName.indexOf(':') + 1);
                }
            }

            String namespaceURI = getNamespaceURI();
            if (namespaceURI != null && !namespaceURI.isEmpty()) {
                String prefix = getNamespaceResolver().getPrefix(namespaceURI);
                if (prefix != null) {
                    return prefix + ":" + localName + "[" + getRelativePositionByQName() + "]";
                }
            }
            return "/" + localName + "[" + getRelativePositionByQName() + "]";
        } else if (node.getNodeType() == Node.TEXT_NODE || node.getNodeType() == Node.CDATA_SECTION_NODE) {
            return "/text()[" + getRelativePositionOfTextNode() + "]";
        } else if (node.getNodeType() == Node.PROCESSING_INSTRUCTION_NODE) {
            String target = ((ProcessingInstruction) node).getTarget();
            return "/processing-instruction('" + target + "')[" + getRelativePositionOfPI() + "]";
        }

        return "/text()[" + getRelativePositionOfTextNode() + "]";
    }
