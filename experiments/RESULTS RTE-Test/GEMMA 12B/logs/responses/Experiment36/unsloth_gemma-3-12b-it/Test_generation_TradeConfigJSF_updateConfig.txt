public class TradeConfigJSF_updateConfigTestGenerated {

    @Inject
    private TradeConfigJSF tradeConfigJSF;

    @Mock
    private ExternalContext mockExternalContext;

    @Mock
    private TradeDirectDBUtils mockTradeDirectDBUtils;

    @BeforeEach

    @Test
    public void testUpdateConfig_validInput() {
        tradeConfigJSF.runtimeMode = "Realtime";
        tradeConfigJSF.orderProcessingMode = "Asynch";
        tradeConfigJSF.webInterface = "JSP";
        tradeConfigJSF.maxUsers = 10;
        tradeConfigJSF.maxQuotes = 20;
        tradeConfigJSF.marketSummaryInterval = 60;
        tradeConfigJSF.primIterations = 5;
        tradeConfigJSF.publishQuotePriceChange = true;
        tradeConfigJSF.listQuotePriceChangeFrequency = 50;
        tradeConfigJSF.longRun = false;
        tradeConfigJSF.displayOrderAlerts = true;

        tradeConfigJSF.updateConfig();

        assertEquals("DayTrader Configuration Updated", tradeConfigJSF.getResult());
        assertEquals("Realtime", TradeConfig.getRunTimeModeNames()[TradeConfig.getRunTimeMode()]);
        assertEquals("Asynch", TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
        assertEquals("JSP", TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);

        assertEquals(50, TradeConfig.getListQuotePriceChangeFrequency());
        assertFalse(TradeConfig.getLongRun());
        assertTrue(TradeConfig.getDisplayOrderAlerts());
    }
