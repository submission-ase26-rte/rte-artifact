  @Test
  public void fetchRoutesForSpecification_validSpecification_returnsRoutes() {
    Location origin = new Location("originUnLocode");
    Location destination = new Location("destinationUnLocode");
    LocalDate arrivalDeadline = LocalDate.now();
    RouteSpecification routeSpecification =
        new RouteSpecification(origin, destination, arrivalDeadline);

    TransitPath transitPath1 = new TransitPath(new ArrayList<>());
    TransitPath transitPath2 = new TransitPath(new ArrayList<>());

    when(graphTraversalResource.queryParam("origin", "originUnLocode")
            .queryParam("destination", "destinationUnLocode")
            .request(MediaType.APPLICATION_JSON_TYPE)
            .get(Mockito.any()))
        .thenReturn(List.of(transitPath1, transitPath2));

    Itinerary itinerary1 = externalRoutingService.toItinerary(transitPath1);
    Itinerary itinerary2 = externalRoutingService.toItinerary(transitPath2);

    when(routeSpecification.isSatisfiedBy(Mockito.any(Itinerary.class)))
        .thenReturn(true);

    List<Itinerary> itineraries =
        externalRoutingService.fetchRoutesForSpecification(routeSpecification);

    assertEquals(2, itineraries.size());
  }
