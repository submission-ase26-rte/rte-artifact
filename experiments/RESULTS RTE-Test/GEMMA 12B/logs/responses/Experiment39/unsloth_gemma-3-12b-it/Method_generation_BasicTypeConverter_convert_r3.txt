import java.math.BigDecimal;
import java.math.BigInteger;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashSet;
import java.util.Iterator;
import java.util.List;
import java.util.Set;
import java.util.SortedSet;
import org.apache.commons.beanutils.Converter;
import org.apache.commons.beanutils.ConvertUtils;
import org.apache.commons.jxpath.JXPathInvalidAccessException;
import org.apache.commons.jxpath.JXPathTypeConversionException;
import org.apache.commons.jxpath.NodeSet;
import org.apache.commons.jxpath.Pointer;
import org.apache.commons.jxpath.util.TypeUtils;

public Object convert(final Object object, final Class toType) {
    if (object == null) {
        if (toType == Boolean.class) {
            return null;
        }
        if (toType == String.class) {
            return "";
        }
        if (toType == long.class) {
            return 0L;
        }
        if (toType == double.class) {
            return 0.0;
        }
        if (toType == float.class) {
            return 0.0f;
        }
        if (toType == int.class) {
            return 0;
        }
        if (toType == short.class) {
            return (short) 0;
        }
        if (toType == byte.class) {
            return (byte) 0;
        }
        if (toType == char.class) {
            return '\0';
        }
        if (toType == boolean.class) {
            return Boolean.FALSE;
        }

        if (toType == Integer.class) {
            return 0;
        }
        if (toType == Long.class) {
            return 0L;
        }
        if (toType == Double.class) {
            return 0.0;
        }
        if (toType == Float.class) {
            return 0.0f;
        }

        return null;
    }

    if (toType == String.class) {
        return object.toString();
    }

    if (Number.class.isAssignableFrom(toType)) {
        if (object instanceof Boolean) {
            return (Boolean) object ? 1 : 0;
        }
        if (object instanceof String) {
            try {
                return ConvertUtils.toNumber((String) object, toType);
            } catch (final Exception e) {
                throw new JXPathTypeConversionException((String) object, e);
            }
        }
        return object;
    }

    if (toType == Boolean.class) {
        if (object instanceof Number) {
            return ((Number) object).doubleValue() != 0.0;
        }
        if (object instanceof String) {
            return ConvertUtils.toBoolean((String) object);
        }
        return false;
    }

    if (toType.isArray()) {
        if (toType.getComponentType() == int.class) {
            if (object instanceof int[]) {
                return object;
            }
            if (object instanceof Integer[]) {
                int[] result = new int[((Integer[]) object).length];
                for (int i = 0; i < ((Integer[]) object).length; i++) {
                    result[i] = ((Integer[]) object)[i];
                }
                return result;
            }
            if (object instanceof List) {
                List list = (List) object;
                int[] result = new int[list.size()];
                for (int i = 0; i < list.size(); i++) {
                    result[i] = (Integer) list.get(i);
                }
                return result;
            }
        } else if (toType.getComponentType() == Object.class) {
            if (object instanceof Object[]) {
                return object;
            }
            if (object instanceof Collection) {
                Collection collection = (Collection) object;
                Object[] result = new Object[collection.size()];
                int i = 0;
                for (Iterator iterator = collection.iterator(); iterator.hasNext(); i++) {
                    result[i] = iterator.next();
                }
                return result;
            }
        }
        return null;
    }

    if (object instanceof Collection) {
        if (toType == Collection.class) {
            return new ArrayList<>((Collection) object);
        }
        if (toType == List.class) {
            return new ArrayList<>((Collection) object);
        }
        if (toType == Set.class) {
            return new HashSet<>((Collection) object);
        }
    }

    if (object instanceof Pointer) {
        Pointer pointer = (Pointer) object;
        Object value = pointer.getValue();
        if (value == null) {
            return "";
        }
        return convert(value, toType);
    }

    if (object instanceof NodeSet) {
        NodeSet nodeSet = (NodeSet) object;
        List values = nodeSet.getValues();
        if (values == null) {
            return "";
        }
        return convert(values, toType);
    }
    throw new JXPathTypeConversionException("Cannot convert " + object + " to " + toType);
}
