        void addPet_newPet_doesNotAddIfAlreadyExists() {
            Pet mockPet = new Pet();
            mockPet.setName("Buddy");
            owner.addPet(mockPet);
            owner.addPet(mockPet);
            assertEquals(1, owner.getPets().size());
        }

    @Test
        void addPet_newPet_addsToPetList() {
            Pet mockPet = new Pet();
            when(mockPet.isNew()).thenReturn(true);
            owner.addPet(mockPet);
            assertEquals(1, owner.getPets().size());
            assertEquals(mockPet, owner.getPets().get(0));
        }

    @Test
        void addPet_existingPet_doesNotAddToPetList() {
            Pet mockPet = new Pet();
            when(mockPet.isNew()).thenReturn(false);
            owner.addPet(mockPet);
            assertEquals(0, owner.getPets().size());
        }

    @Test
        void addPet_nullPet_doesNothing() {
            owner.addPet(null);
            assertEquals(0, owner.getPets().size());
        }

    @Test
        void addPet_newPet_addsPetWithNewName() {
            Pet mockPet1 = new Pet();
            mockPet1.setName("Buddy");
            Pet mockPet2 = new Pet();
            mockPet2.setName("Rocky");
            when(mockPet1.isNew()).thenReturn(true);
            when(mockPet2.isNew()).thenReturn(true);
            owner.addPet(mockPet1);
            owner.addPet(mockPet2);
            assertEquals(2, owner.getPets().size());
            assertEquals("Buddy", owner.getPets().get(0).getName());
            assertEquals("Rocky", owner.getPets().get(1).getName());
        }
