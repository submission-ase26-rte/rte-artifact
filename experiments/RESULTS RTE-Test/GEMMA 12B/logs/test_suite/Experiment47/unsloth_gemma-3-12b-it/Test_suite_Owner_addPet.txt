    void addPet_existingPet_doesNotAddToPetList() {

        assertEquals(0, owner.getPets().size());
    }

    @Test
    void addPet_nullPet_doesNothing() {
        owner.addPet(null);
        assertEquals(0, owner.getPets().size());
