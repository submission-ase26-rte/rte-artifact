import java.util.Stack;

import org.mockito.Mockito;
import static org.junit.jupiter.api.Assertions.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class PrecedingOrFollowingContext_nextNodeTestGenerated {
    private EvalContext mockEvalContext;
    private NodeTest mockNodeTest;
    private PrecedingOrFollowingContext target;

    @BeforeEach
    void setUp() {
        mockEvalContext = mock(EvalContext.class);
        mockNodeTest = mock(NodeTest.class);
        target = new PrecedingOrFollowingContext(mockEvalContext, mockNodeTest, false);
    }

    @AfterEach
    void tearDown() {
        mockEvalContext = null;
        mockNodeTest = null;
        target = null;
    }

    @Test
        void testNextNode_noStack_rootHasParent_pushIterator() {
            NodePointer root = Mockito.mock(NodePointer.class);
            when(root.isLeaf()).thenReturn(false);
            when(root.getParent()).thenReturn(Mockito.mock(NodePointer.class));
            when(mockEvalContext.getCurrentNodePointer()).thenReturn(root);
            NodeIterator iterator = Mockito.mock(NodeIterator.class);
            when(root.childIterator(null, false, root)).thenReturn(iterator);
            target.nextNode();
            verify(root).childIterator(null, false, root);
        }

    @Test
    void testNextNode_noStack_rootHasNoParent() {
        NodePointer root = Mockito.mock(NodePointer.class);
        when(root.isLeaf()).thenReturn(false);
        when(root.getParent()).thenReturn(null);
        when(mockEvalContext.getCurrentNodePointer()).thenReturn(root);
        assertFalse(target.nextNode());
    }

    @Test
    void testNextNode_reverse_rootHasParent_pushIterator() {
        NodePointer root = Mockito.mock(NodePointer.class);
        when(root.isLeaf()).thenReturn(false);
        when(root.getParent()).thenReturn(Mockito.mock(NodePointer.class));
        when(mockEvalContext.getCurrentNodePointer()).thenReturn(root);
        NodeIterator iterator = Mockito.mock(NodeIterator.class);
        when(root.childIterator(null, true, root)).thenReturn(iterator);
        target = new PrecedingOrFollowingContext(mockEvalContext, mockNodeTest, true);
        target.nextNode();
        verify(root).childIterator(null, true, root);
    }

    @Test
    void testNextNode_reverse_rootHasNoParent() {
        NodePointer root = Mockito.mock(NodePointer.class);
        when(root.isLeaf()).thenReturn(false);
        when(root.getParent()).thenReturn(null);
        when(mockEvalContext.getCurrentNodePointer()).thenReturn(root);
        target = new PrecedingOrFollowingContext(mockEvalContext, mockNodeTest, true);
        assertFalse(target.nextNode());
    }

}
