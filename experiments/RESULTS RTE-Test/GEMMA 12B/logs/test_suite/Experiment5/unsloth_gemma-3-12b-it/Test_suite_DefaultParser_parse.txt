@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class DefaultParser_parseTestGenerated {

    private DefaultParser parser;

    @BeforeEach
    void setUp() {
        parser = new DefaultParser();
    }

    @Test
    void testParseWithProperties() throws ParseException {
        Options options = new Options();
        options.addOption("p", true, "property option");

        Properties properties = new Properties();
        properties.setProperty("p", "propertyValue");

        CommandLine result = parser.parse(options, new String[]{}, properties, false);

        assertNotNull(result);
    }

    @Test
    void testParseWithEmptyOptions() throws ParseException {
        Options options = new Options();

        CommandLine result = parser.parse(options, new String[]{}, new Properties(), false);

        assertNotNull(result);
    }
