import org.eclipse.cargotracker.domain.model.voyage.VoyageRepository;
import org.eclipse.pathfinder.api.TransitEdge;
import org.eclipse.pathfinder.api.TransitPath;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.springframework.test.util.ReflectionTestUtils;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.logging.Level;
import java.util.logging.Logger;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.anyString;
import static org.mockito.Mockito.*;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class ExternalRoutingService_fetchRoutesForSpecificationTestGenerated {

    @Mock
    private Logger logger;

    @Mock
    private WebTarget graphTraversalResource;

    @Mock
    private LocationRepository locationRepository;

    @Mock
    private VoyageRepository voyageRepository;

    private ExternalRoutingService externalRoutingService;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        externalRoutingService = new ExternalRoutingService();
        ReflectionTestUtils.setField(externalRoutingService, "logger", logger);
        ReflectionTestUtils.setField(externalRoutingService, "graphTraversalResource", graphTraversalResource);
        ReflectionTestUtils.setField(externalRoutingService, "locationRepository", locationRepository);
        ReflectionTestUtils.setField(externalRoutingService, "voyageRepository", voyageRepository);
    }

    @Test
    public void testFetchRoutesForSpecification_withValidInput() {
        RouteSpecification routeSpecification = new RouteSpecification(
                new Location(UnLocode.of("USNYC")),
                new Location(UnLocode.of("DEHAM")),
                LocalDate.now()
        );

        TransitEdge transitEdge1 = mock(TransitEdge.class);
        TransitEdge transitEdge2 = mock(TransitEdge.class);
        TransitPath transitPath = new TransitPath(Arrays.asList(transitEdge1, transitEdge2));
        List<TransitPath> transitPaths = new ArrayList<>();
        transitPaths.add(transitPath);

        when(graphTraversalResource.queryParam(anyString(), anyString()).request(MediaType.APPLICATION_JSON_TYPE).get(any(GenericType.class)))
                .thenReturn(transitPaths);

        when(graphTraversalResource.queryParam(anyString(), anyString())).thenReturn(graphTraversalResource);
        when(transitEdge1.getEdgeId()).thenReturn("edge1");
        when(transitEdge2.getEdgeId()).thenReturn("edge2");

        List<Itinerary> itineraries = externalRoutingService.fetchRoutesForSpecification(routeSpecification);

        assertNotNull(itineraries);
        assertEquals(1, itineraries.size());
        assertEquals(2, itineraries.get(0).getLegs().size());
    }

    @Test
    public void testFetchRoutesForSpecification_withNoTransitPaths() {
        RouteSpecification routeSpecification = new RouteSpecification(
                new Location(UnLocode.of("USNYC")),
                new Location(UnLocode.of("DEHAM")),
                LocalDate.now()
        );

        when(graphTraversalResource.queryParam(anyString(), anyString()).request(MediaType.APPLICATION_JSON_TYPE).get(any(GenericType.class)))
                .thenReturn(new ArrayList<>());

        when(graphTraversalResource.queryParam(anyString(), anyString())).thenReturn(graphTraversalResource);

        List<Itinerary> itineraries = externalRoutingService.fetchRoutesForSpecification(routeSpecification);

        assertNotNull(itineraries);
        assertTrue(itineraries.isEmpty());
    }

    @Test
    public void testFetchRoutesForSpecification_withInvalidItinerary() {
        RouteSpecification routeSpecification = new RouteSpecification(
                new Location(UnLocode.of("USNYC")),
                new Location(UnLocode.of("DEHAM")),
                LocalDate.now()
        );

        TransitEdge transitEdge = mock(TransitEdge.class);
        TransitPath transitPath = new TransitPath(Arrays.asList(transitEdge));
        List<TransitPath> transitPaths = new ArrayList<>();
        transitPaths.add(transitPath);

        when(graphTraversalResource.queryParam(anyString(), anyString()).request(MediaType.APPLICATION_JSON_TYPE).get(any(GenericType.class)))
                .thenReturn(transitPaths);

        when(graphTraversalResource.queryParam(anyString(), anyString())).thenReturn(graphTraversalResource);
        when(transitEdge.getEdgeId()).thenReturn("edge1");

        List<Itinerary> itineraries = externalRoutingService.fetchRoutesForSpecification(routeSpecification);

        assertNotNull(itineraries);
        assertTrue(itineraries.isEmpty());

        verify(logger, times(1)).log(Level.FINE, "Received itinerary that did not satisfy the route specification");
    }

    @Test
    public void testFetchRoutesForSpecification_withNullRouteSpecification() {
        assertThrows(NullPointerException.class, () -> {
            externalRoutingService.fetchRoutesForSpecification(null);
        });
    }
}
