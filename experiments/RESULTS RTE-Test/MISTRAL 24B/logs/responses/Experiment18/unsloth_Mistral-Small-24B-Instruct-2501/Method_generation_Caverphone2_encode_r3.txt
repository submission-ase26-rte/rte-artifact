public String encode(final String source) {
    if (SoundexUtils.isEmpty(source)) {
        return "1111111111";
    }

    final StringBuilder encoded = new StringBuilder(10);
    final String lowerSource = source.toLowerCase();

    if (lowerSource.startsWith("wh")) {
        encoded.append('W');
    } else {
        encoded.append(getInitial(lowerSource));
    }

    int prev = 0;
    for (int i = 0; i < lowerSource.length(); i++) {
        final int current = getCode(lowerSource.charAt(i));
        if (current != prev) {
            encoded.append(current == 0 ? '1' : current);
        }
        prev = current;
    }

    while (encoded.length() < 10) {
        encoded.append('1');
    }

    return encoded.toString().substring(0, 10);
}

private char getInitial(final String source) {
    return getCode(source.charAt(0));
}

private int getCode(final char c) {
    switch (c) {
        case 'b': return 1;
        case 'c':
        case 'k':
        case 'q':
        case 'x':
        case 'z': return 2;
        case 'd':
        case 't': return 3;
        case 'l': return 4;
        case 'm':
        case 'n': return 5;
        case 'r': return 6;
        case 'a':
        case 'e':
        case 'i':
        case 'o':
        case 'u':
        case 'y': return 0;
        case 'f': return 8;
        case 'p': return 9;
        case 's': return 0;
        case 'j': return 10;
        case 'v': return 11;
        case 'w': return 12;
        case 'g': return 13;
        case 'h': return 14;
        case 'g':
            if (i > 0 && "aeiou".indexOf(lowerSource.charAt(i - 1)) >= 0) {
                return 13;
            } else {
                return 14;
            }
        case 'h':
            if (i > 0 && "aeiou".indexOf(lowerSource.charAt(i - 1)) >= 0) {
                return 15;
            } else {
                return 16;
            }
        case 't':
            if (i > 0 && "aeiou".indexOf(lowerSource.charAt(i - 1)) >= 0) {
                return 17;
            } else {
                return 18;
            }
        default: return 19;
    }
}
