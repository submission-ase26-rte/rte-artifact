public String nysiis(String str) {
    if (str == null) {
        return null;
    }

    int length = str.length();
    if (length == 0) {
        return "";
    }

    StringBuilder result = new StringBuilder();
    char[] chars = str.toCharArray();
    char prev = 0;
    char curr = 0;
    char next = 0;
    char aNext = 0;
    int i = 0;

    while (i < length) {
        prev = (i > 0) ? chars[i - 1] : 0;
        curr = chars[i];
        next = (i + 1 < length) ? chars[i + 1] : 0;
        aNext = (i + 2 < length) ? chars[i + 2] : 0;

        if (Character.isWhitespace(curr) || curr == '*') {
            i++;
            continue;
        }

        if (isVowel(curr)) {
            result.append('A');
        } else if (curr == 'H' && (!isVowel(prev) || !isVowel(next))) {
            result.append(prev);
        } else if (curr == 'W' && isVowel(prev)) {
            result.append(prev);
        } else {
            char[] transcode = transcodeRemaining(prev, curr, next, aNext);
            if (transcode.length > 0) {
                result.append(transcode[0]);
            }
        }

        i++;
    }

    String cleaned = SoundexUtils.clean(result.toString());

    if (isStrict() || cleaned.length() == 0) {
        return cleaned;
    }

    if (cleaned.length() == 1) {
        return cleaned + 'A';
    }

    return cleaned.substring(0, 4);
}
