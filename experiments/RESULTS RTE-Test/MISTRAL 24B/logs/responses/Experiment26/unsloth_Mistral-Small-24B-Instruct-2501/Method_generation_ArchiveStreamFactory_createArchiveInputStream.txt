public <I extends ArchiveInputStream<? extends ArchiveEntry>> I createArchiveInputStream(final String archiverName, final InputStream in, final String actualEncoding) throws ArchiveException {
    if (archiverName == null) {
        throw new IllegalArgumentException("archiverName cannot be null");
    }
    if (in == null) {
        throw new IllegalArgumentException("inputStream cannot be null");
    }

    String key = toKey(archiverName);
    ArchiveStreamProvider provider = getArchiveInputStreamProviders().get(key);
    if (provider == null) {
        throw new ArchiveException("no ArchiveStreamProvider found for " + key);
    }

    try {
        @SuppressWarnings("unchecked")
        I stream = (I) provider.createArchiveInputStream(archiverName, in, actualEncoding);
        return stream;
    } catch (IOException e) {
        throw new ArchiveException("Failed to create archive input stream for " + key, e);
    }
}
