@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class EventLogger_initTestGenerated {
    @Mock
    private CargoRepository cargoRepository;
    @Mock
    private LocationRepository locationRepository;
    @Mock
    private VoyageRepository voyageRepository;
    @InjectMocks
    private EventLogger eventLogger;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        ReflectionTestUtils.setField(eventLogger, "cargoRepository", cargoRepository);
        ReflectionTestUtils.setField(eventLogger, "locationRepository", locationRepository);
        ReflectionTestUtils.setField(eventLogger, "voyageRepository", voyageRepository);
    }

    @Test
    public void testInitWithNoCargo() {
        when(cargoRepository.findAll()).thenReturn(new ArrayList<>());
        eventLogger.init();
        List<SelectItem> trackingIds = eventLogger.getTrackingIds();
        assertNotNull(trackingIds);
        assertEquals(0, trackingIds.size());
    }

    @Test
    public void testInitWithNoLocations() {
        List<Cargo> cargos = new ArrayList<>();
        when(cargoRepository.findAll()).thenReturn(cargos);
        when(locationRepository.findAll()).thenReturn(new ArrayList<>());
        eventLogger.init();
        List<SelectItem> locationsList = eventLogger.getLocations();
        assertNotNull(locationsList);
        assertEquals(0, locationsList.size());
    }

    @Test
    public void testInitWithNoVoyages() {
        List<Cargo> cargos = new ArrayList<>();
        when(cargoRepository.findAll()).thenReturn(cargos);
        when(voyageRepository.findAll()).thenReturn(new ArrayList<>());
        eventLogger.init();
        List<SelectItem> voyagesList = eventLogger.getVoyages();
        assertNotNull(voyagesList);
        assertEquals(0, voyagesList.size());
    }

    @Test
    public void testInitWithMultipleVoyages() {
        Voyage voyage1 = new Voyage(new VoyageNumber("V01"), new Schedule());
        Voyage voyage2 = new Voyage(new VoyageNumber("V02"), new Schedule());
        List<Voyage> voyages = new ArrayList<>();
        voyages.add(voyage1);
        voyages.add(voyage2);
        when(voyageRepository.findAll()).thenReturn(voyages);
        eventLogger.init();
        List<SelectItem> voyagesList = eventLogger.getVoyages();
        assertNotNull(voyagesList);
        assertEquals(2, voyagesList.size());
        assertEquals("V01", voyagesList.get(0).getValue());
        assertEquals("V02", voyagesList.get(1).getValue());
    }

    @Test
    public void testInitWithUnroutedCargo() {
        TrackingId trackingId = new TrackingId("123");
        Cargo cargo = new Cargo(trackingId, new RouteSpecification());
        List<Cargo> cargos = new ArrayList<>();
        cargos.add(cargo);
        when(cargoRepository.findAll()).thenReturn(cargos);
        eventLogger.init();
        List<SelectItem> trackingIds = eventLogger.getTrackingIds();
        assertNotNull(trackingIds);
        assertEquals(0, trackingIds.size());
    }

    @Test
    public void testInitWithMultipleCargos() {
        TrackingId trackingId1 = new TrackingId("123");
        TrackingId trackingId2 = new TrackingId("456");
        Cargo cargo1 = new Cargo(trackingId1, new RouteSpecification());
        Cargo cargo2 = new Cargo(trackingId2, new RouteSpecification());
        List<Cargo> cargos = new ArrayList<>();
        cargos.add(cargo1);
        cargos.add(cargo2);
        when(cargoRepository.findAll()).thenReturn(cargos);
        eventLogger.init();
        List<SelectItem> trackingIds = eventLogger.getTrackingIds();
        assertNotNull(trackingIds);
        assertEquals(0, trackingIds.size());
    }

    @Test
    public void testInitWithMultipleLocations() {
        Location location1 = new Location(new UnLocode("USNYC"), "Location1");
        Location location2 = new Location(new UnLocode("USLAX"), "Location2");
        List<Location> locations = new ArrayList<>();
        locations.add(location1);
        locations.add(location2);
        when(locationRepository.findAll()).thenReturn(locations);
        eventLogger.init();
        List<SelectItem> locationsList = eventLogger.getLocations();
        assertNotNull(locationsList);
        assertEquals(2, locationsList.size());
        assertEquals("USNYC", locationsList.get(0).getValue());
        assertEquals("USLAX", locationsList.get(1).getValue());
        assertEquals("Location1 (USNYC)", locationsList.get(0).getLabel());
        assertEquals("Location2 (USLAX)", locationsList.get(1).getLabel());
    }

}
