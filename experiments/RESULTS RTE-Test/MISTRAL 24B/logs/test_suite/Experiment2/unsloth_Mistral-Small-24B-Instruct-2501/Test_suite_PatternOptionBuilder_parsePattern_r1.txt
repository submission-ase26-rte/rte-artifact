package org.apache.commons.cli;

import org.apache.commons.cli.Option;
import org.apache.commons.cli.Options;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;
import static org.junit.jupiter.api.Assertions.*;

    @Test
    public void testParsePattern_EmptyPattern() {
        String pattern = "";
        Options result = PatternOptionBuilder.parsePattern(pattern);
        assertTrue(result.getOptions().isEmpty());
    }

    @Test
    public void testParsePattern_NullPattern() {
        assertThrows(NullPointerException.class, () -> {
            PatternOptionBuilder.parsePattern(null);

    @Test
    public void testParsePattern_ValidPattern() {
        String pattern = "a!b:c%";
        Options result = PatternOptionBuilder.parsePattern(pattern);
        Option optionA = result.getOption("a");
        assertNotNull(optionA);
        assertTrue(optionA.isRequired());
        assertEquals(Option.UNINITIALIZED, optionA.getArgs());
        assertEquals(Option.UNLIMITED_VALUES, optionA.getArgs());
        Option optionB = result.getOption("b");
        assertNotNull(optionB);
        assertFalse(optionB.isRequired());
        assertEquals(Option.UNINITIALIZED, optionB.getArgs());
        assertEquals(Option.UNLIMITED_VALUES, optionB.getArgs());
        Option optionC = result.getOption("c");
        assertNotNull(optionC);
        assertFalse(optionC.isRequired());
        assertEquals(1, optionC.getArgs());
        assertEquals(Option.UNLIMITED_VALUES, optionC.getArgs());
    }
