class DOMNodePointer_asPathTestGenerated {

    private JXPathContext mockContext;

    @BeforeEach
    void setUp() {
        mockContext = Mockito.mock(JXPathContext.class);
    }

    @AfterEach
    void tearDown() {
        Mockito.reset(mockContext);
    }

    @Test
    void testAsPathWithId() {
        Document doc = Mockito.mock(Document.class);
        Element element = Mockito.mock(Element.class);
        Mockito.when(element.getNodeType()).thenReturn(Node.ELEMENT_NODE);
        Mockito.when(element.getAttributeNode("id")).thenReturn(null);
        DOMNodePointer pointer = new DOMNodePointer(element, Locale.getDefault(), "testId");
        assertEquals("id('testId')", pointer.asPath());
    }
