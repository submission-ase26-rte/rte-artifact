package org.apache.commons.jxpath.ri.axes;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.apache.commons.jxpath.ri.EvalContext;
import org.apache.commons.jxpath.ri.compiler.NodeTest;
import org.apache.commons.jxpath.ri.model.NodeIterator;
import org.apache.commons.jxpath.ri.model.NodePointer;

import java.util.Stack;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
@ExtendWith(MockitoExtension.class)
public class PrecedingOrFollowingContext_nextNodeTestGenerated {

    @Mock
    private EvalContext parentContext;

    @Mock
    private NodeTest nodeTest;

    @Mock
    private NodePointer rootNode;

    @Mock
    private NodePointer parentNode;

    @Mock
    private NodeIterator nodeIterator;

    private PrecedingOrFollowingContext precedingOrFollowingContext;

    @BeforeEach
    public void setUp() {
        precedingOrFollowingContext = new PrecedingOrFollowingContext(parentContext, nodeTest, false);
    }

    @AfterEach
    public void tearDown() {
        // Reset mocks (optional, but good practice)
        reset(parentContext, nodeTest, rootNode, parentNode, nodeIterator);
    }

    @Test
    void testNextNode_simpleSuccess() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.getParent()).thenReturn(parentNode);
        when(parentNode.childIterator(null, false, rootNode)).thenReturn(nodeIterator);
        when(nodeIterator.setPosition(1)).thenReturn(true);
        when(nodeIterator.getNodePointer()).thenReturn(rootNode);
        when(rootNode.testNode(nodeTest)).thenReturn(true);

        boolean result = precedingOrFollowingContext.nextNode();

        assertTrue(result);
        assertEquals(1, precedingOrFollowingContext.getCurrentPosition());
        verify(parentContext).getCurrentNodePointer();
        verify(rootNode).getParent();
        verify(parentNode).childIterator(null, false, rootNode);
        verify(nodeIterator).setPosition(1);
        verify(nodeIterator).getNodePointer();
        verify(rootNode).testNode(nodeTest);
    }

    @Test
    void testNextNode_noMatchFound() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.getParent()).thenReturn(parentNode);
        when(parentNode.childIterator(null, false, rootNode)).thenReturn(nodeIterator);
        when(nodeIterator.setPosition(1)).thenReturn(false);

        boolean result = precedingOrFollowingContext.nextNode();

        assertFalse(result);
    }

     @Test
    void testNextNode_reverseSuccess() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.getParent()).thenReturn(parentNode);
        when(parentNode.childIterator(null, true, rootNode)).thenReturn(nodeIterator);
        when(nodeIterator.setPosition(1)).thenReturn(true);
        when(nodeIterator.getNodePointer()).thenReturn(rootNode);
        when(rootNode.testNode(nodeTest)).thenReturn(true);

        precedingOrFollowingContext = new PrecedingOrFollowingContext(parentContext, nodeTest, true);
        boolean result = precedingOrFollowingContext.nextNode();

        assertTrue(result);
        assertEquals(1, precedingOrFollowingContext.getCurrentPosition());
        verify(parentContext).getCurrentNodePointer();
        verify(rootNode).getParent();
        verify(parentNode).childIterator(null, true, rootNode);
        verify(nodeIterator).setPosition(1);
        verify(nodeIterator).getNodePointer();
        verify(rootNode).testNode(nodeTest);
    }

    @Test
    void testNextNode_noParent() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.getParent()).thenReturn(null);

        boolean result = precedingOrFollowingContext.nextNode();

        assertFalse(result);
    }

    @Test
    void testNextNode_rootNode() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.isRoot()).thenReturn(true);

        boolean result = precedingOrFollowingContext.nextNode();

        assertFalse(result);
    }

    @Test
    void testNextNode_multipleLevels() {
        when(parentContext.getCurrentNodePointer()).thenReturn(rootNode);
        when(rootNode.getParent()).thenReturn(parentNode);
        when(parentNode.childIterator(null, false, rootNode)).thenReturn(nodeIterator);
        when(nodeIterator.setPosition(1)).thenReturn(true);
        when(nodeIterator.getNodePointer()).thenReturn(rootNode);
        when(rootNode.testNode(nodeTest)).thenReturn(false);
        when(nodeIterator.setPosition(2)).thenReturn(true);
        when(nodeIterator.getNodePointer()).thenReturn(rootNode);
        when(rootNode.testNode(nodeTest)).thenReturn(true);

        boolean result = precedingOrFollowingContext.nextNode();

        assertTrue(result);
    }
}
