public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        if (toType == Boolean.class || toType == Integer.class || toType == Long.class || toType == Double.class || toType == Float.class || toType == String.class) {
            return true;
        }
        return false;
    }

    if (toType == String.class) {
        return true;
    }

    if (toType.isPrimitive() || toType.isWrapperType()) {
        if (object instanceof Number) {
            return true;
        } else if (object instanceof Boolean) {
            return toType == Boolean.class;
        } else {
            return false;
        }
    }

    if (toType == Collection.class) {
        if (object instanceof Collection) {
            return true;
        } else if (object instanceof Object[]) {
            return true;
        } else {
            return false;
        }
    }

    if (object instanceof Array) {
        if (toType.getComponentType() != null) {
            if (toType.getComponentType().isPrimitive()) {
                return true;
            }
            if (object.getClass().getComponentType() == toType) {
                return true;
            }
        }
    }

    if (object instanceof NodeSet) {
        return Collection.class.isAssignableFrom(toType);
    }

    if (object instanceof Pointer) {
        return toType == String.class;
    }

    if (toType == Integer.class && object instanceof java.util.concurrent.atomic.AtomicBoolean) {
        return true;
    }

    if (toType == Boolean.class && (object instanceof Number || object instanceof java.util.concurrent.atomic.AtomicBoolean)) {
        return true;
    }

    if (toType == Double.class && object instanceof Number) {
        return true;
    }

    if (toType == java.util.concurrent.atomic.AtomicBoolean.class && (object instanceof Boolean || object instanceof Number)) {
        return true;
    }

    if (toType.isArray()) {
        if (object instanceof Collection) {
            return true;
        }
    }

    return false;
}
