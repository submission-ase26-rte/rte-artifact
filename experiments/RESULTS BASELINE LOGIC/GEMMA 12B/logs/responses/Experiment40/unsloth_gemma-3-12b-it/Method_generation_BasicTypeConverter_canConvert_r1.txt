public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        return toType.isAssignableFrom(Integer.class) || toType.isAssignableFrom(Long.class) || toType.isAssignableFrom(Double.class) || toType.isAssignableFrom(Float.class) || toType.isAssignableFrom(Boolean.class) || toType.isAssignableFrom(String.class);
    }

    if (toType == String.class) {
        return true;
    }

    if (toType.isPrimitive() || toType.isWrapperType()) {
        if (object instanceof Number) {
            return true;
        } else if (object instanceof Boolean) {
            return toType == Boolean.class;
        } else {
            return false;
        }
    }

    if (toType == Collection.class) {
        if (object instanceof Collection) {
            return true;
        } else if (object instanceof Object[]) {
            return true;
        } else {
            return false;
        }
    }

    if (object instanceof Array) {
        if (toType.getComponentType() != null) {
            if (toType.getComponentType().isPrimitive()) {
                return true;
            }
            if (object.getClass().getComponentType() == toType) {
                return true;
            }
        }
    }

    if (object instanceof NodeSet) {
        return Collection.class.isAssignableFrom(toType);
    }

    if (object instanceof Pointer) {
        return toType == String.class;
    }

    if (toType == Integer.class && object instanceof AtomicBoolean) {
        return true;
    }

    if (toType == Boolean.class && (object instanceof Number || object instanceof AtomicBoolean)) {
        return true;
    }

    if (toType == Double.class && object instanceof Number) {
        return true;
    }

    return false;
}
