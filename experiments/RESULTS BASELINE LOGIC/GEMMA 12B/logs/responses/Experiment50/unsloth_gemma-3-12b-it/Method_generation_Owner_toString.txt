public String toString() {
    return new org.springframework.core.style.ToStringCreator(this)
            .append("id", getId())
            .append("new", isNew())
            .append("lastName", getLastName())
            .append("firstName", getFirstName())
            .append("address", getAddress())
            .append("city", getCity())
            .append("telephone", getTelephone())
            .toString();
}
