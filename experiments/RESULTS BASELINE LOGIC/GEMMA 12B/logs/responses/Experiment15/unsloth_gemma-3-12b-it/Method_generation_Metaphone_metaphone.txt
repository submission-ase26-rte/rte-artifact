public String metaphone(final String txt) {
        if (txt == null) {
            return null;
        }
        if (txt.isEmpty()) {
            return "";
        }

        StringBuilder string = new StringBuilder(txt.toUpperCase());
        int wdsz = string.length();
        int maxCodeLen = getMaxCodeLen();

        StringBuilder metaphone = new StringBuilder();

        int index = 0;
        while (index < wdsz) {
            char currentChar = string.charAt(index);

            switch (currentChar) {
                case 'A':
                case 'E':
                case 'I':
                case 'O':
                case 'U':
                    if (index == 0) {
                        metaphone.append('A');
                    }
                    index++;
                    break;
                case 'B':
                    if (index == wdsz - 1 || string.charAt(index + 1) != 'B') {
                        metaphone.append("B");
                    }
                    index++;
                    break;
                case 'C':
                    if (index > 1 && isPreviousChar(string, index - 2, 'S') && isNextChar(string, index, 'I') && string.charAt(index + 1) == 'A') {
                        metaphone.append("X");
                    } else if (isNextChar(string, index, 'I') || isNextChar(string, index, 'E')) {
                        metaphone.append("S");
                    } else if (isNextChar(string, index, 'H')) {
                        if (index == 0 && wdsz >= 3 && string.charAt(2) == 'A') {
                            metaphone.append("K");
                        } else {
                            metaphone.append("K");
                        }
                    } else {
                        metaphone.append("K");
                    }
                    index++;
                    break;
                case 'D':
                    if (isNextChar(string, index, 'G') && isNextChar(string, index + 1, 'E')) {
                        metaphone.append("J");
                    } else {
                        metaphone.append("T");
                    }
                    index++;
                    break;
                case 'F':
                    metaphone.append("F");
                    index++;
                    break;
                case 'G':
                    if (isNextChar(string, index, 'H')) {
                        if (!isVowel(string, index + 1)) {
                            index += 2;
                            break;
                        }
                    }
                    if (isNextChar(string, index, 'N')) {
                        index++;
                        break;
                    }
                   if (isNextChar(string, index, 'I') || isNextChar(string, index, 'E')) {
                        metaphone.append("J");
                    } else {
                        metaphone.append("K");
                    }
                    index++;
                    break;
                case 'H':
                    if (index == 0 || isVowel(string, index - 1)) {
                        metaphone.append("H");
                    }
                    index++;
                    break;
                case 'J':
                    metaphone.append("J");
                    index++;
                    break;
                case 'K':
                    if (index == wdsz - 1 || string.charAt(index + 1) != 'K') {
                        metaphone.append("K");
                    }
                    index++;
                    break;
                case 'L':
                    metaphone.append("L");
                    index++;
                    break;
                case 'M':
                    metaphone.append("M");
                    index++;
                    break;
                case 'N':
                    metaphone.append("N");
                    index++;
                    break;
                case 'P':
                    if (isNextChar(string, index, 'H')) {
                        metaphone.append("F");
                    } else {
                        metaphone.append("P");
                    }
                    index++;
                    break;
                case 'Q':
                    metaphone.append("K");
                    index++;
                    break;
                case 'R':
                    metaphone.append("R");
                    index++;
                    break;
                case 'S':
                    if (isNextChar(string, index, 'H')) {
                        metaphone.append("X");
                    } else {
                        metaphone.append("S");
                    }
                    index++;
                    break;
                case 'T':
                    if (isNextChar(string, index, 'I') && isNextChar(string, index + 1, 'O')) {
                        metaphone.append("X");
                    } else if (isNextChar(string, index, 'H')) {
                        metaphone.append("0");
                    } else {
                        metaphone.append("T");
                    }
                    index++;
                    break;
                case 'V':
                    metaphone.append("F");
                    index++;
                    break;
                case 'W':
                    metaphone.append("W");
                    index++;
                    break;
                case 'X':
                    metaphone.append("S");
                    index++;
                    break;
                case 'Y':
                    if (index == 0) {
                        metaphone.append("Y");
                    }
                    index++;
                    break;
                case 'Z':
                    metaphone.append("S");
                    index++;
                    break;
                default:
                    index++;
            }

            if (metaphone.length() >= maxCodeLen) {
                break;
            }
        }

        return metaphone.toString();
    }
