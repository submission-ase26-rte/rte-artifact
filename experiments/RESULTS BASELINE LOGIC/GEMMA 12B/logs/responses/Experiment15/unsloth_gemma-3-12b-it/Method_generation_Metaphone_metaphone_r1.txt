public String metaphone(final String txt) {
        if (txt == null) {
            return null;
        }
        if (txt.isEmpty()) {
            return "";
        }

        txt = txt.toUpperCase();
        StringBuilder metaphone = new StringBuilder();
        int len = txt.length();
        int maxCodeLen = getMaxCodeLen();

        for (int i = 0; i < len; i++) {
            char c = txt.charAt(i);

            switch (c) {
                case 'A':
                case 'E':
                case 'I':
                case 'O':
                case 'U':
                    if (i == 0) {
                        metaphone.append('A');
                    }
                    break;
                case 'B':
                    if (i == len - 1 || txt.charAt(i + 1) != 'B') {
                        metaphone.append('B');
                    }
                    break;
                case 'C':
                    if (i > 1 && isPreviousChar(txt, i - 2, 'S') && isNextChar(txt, i, 'I') && txt.charAt(i + 1) == 'A') {
                        metaphone.append('X');
                    } else if (isNextChar(txt, i, 'I') || isNextChar(txt, i, 'E')) {
                        metaphone.append('S');
                    } else if (isNextChar(txt, i, 'H')) {
                        metaphone.append('K');
                    } else {
                        metaphone.append('K');
                    }
                    break;
                case 'D':
                    if (isNextChar(txt, i, 'G') && isNextChar(txt, i + 1, 'E')) {
                        metaphone.append('J');
                    } else {
                        metaphone.append('T');
                    }
                    break;
                case 'F':
                    metaphone.append('F');
                    break;
                case 'G':
                    if (isNextChar(txt, i, 'H')) {
                        if (!isVowel(txt, i + 1)) {
                            break;
                        }
                    }
                    if (isNextChar(txt, i, 'N')) {
                        break;
                    }
                    if (isNextChar(txt, i, 'I') || isNextChar(txt, i, 'E')) {
                        metaphone.append('J');
                    } else {
                        metaphone.append('K');
                    }
                    break;
                case 'H':
                    if (i == 0 || isVowel(txt, i - 1)) {
                        metaphone.append('H');
                    }
                    break;
                case 'J':
                    metaphone.append('J');
                    break;
                case 'K':
                    if (i == len - 1 || txt.charAt(i + 1) != 'K') {
                        metaphone.append('K');
                    }
                    break;
                case 'L':
                    metaphone.append('L');
                    break;
                case 'M':
                    metaphone.append('M');
                    break;
                case 'N':
                    metaphone.append('N');
                    break;
                case 'P':
                    if (isNextChar(txt, i, 'H')) {
                        metaphone.append('F');
                    } else {
                        metaphone.append('P');
                    }
                    break;
                case 'Q':
                    metaphone.append('K');
                    break;
                case 'R':
                    metaphone.append('R');
                    break;
                case 'S':
                    if (isNextChar(txt, i, 'H')) {
                        metaphone.append('X');
                    } else {
                        metaphone.append('S');
                    }
                    break;
                case 'T':
                    if (isNextChar(txt, i, 'I') && isNextChar(txt, i + 1, 'O')) {
                        metaphone.append('X');
                    } else if (isNextChar(txt, i, 'H')) {
                        metaphone.append('0');
                    } else {
                        metaphone.append('T');
                    }
                    break;
                case 'V':
                    metaphone.append('F');
                    break;
                case 'W':
                    metaphone.append('W');
                    break;
                case 'X':
                    metaphone.append('S');
                    break;
                case 'Y':
                    if (i == 0) {
                        metaphone.append('Y');
                    }
                    break;
                case 'Z':
                    metaphone.append('S');
                    break;
            }

            if (metaphone.length() >= maxCodeLen) {
                break;
            }
        }

        return metaphone.toString();
    }
