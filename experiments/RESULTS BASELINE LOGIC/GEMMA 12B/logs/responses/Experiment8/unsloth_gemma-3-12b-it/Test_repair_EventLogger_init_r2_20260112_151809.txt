@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class EventLogger_initTestGenerated {

  @Inject
  private EventLogger eventLogger;

  private LocationRepository locationRepository;

  @Inject
  private VoyageRepository voyageRepository;

  @Inject
  private ApplicationEvents applicationEvents;

  @BeforeEach
  public void setUp() {
    MockitoAnnotations.openMocks(this);
    eventLogger = new EventLogger();
    ReflectionTestUtils.setField(eventLogger, "cargoRepository", cargoRepository);
    ReflectionTestUtils.setField(eventLogger, "locationRepository", locationRepository);
    ReflectionTestUtils.setField(eventLogger, "voyageRepository", voyageRepository);
    ReflectionTestUtils.setField(eventLogger, "applicationEvents", applicationEvents);
  }

  @AfterEach

  }

  @Test
  public void testInit_validData() {
    List<Cargo> cargos = new ArrayList<>();
    Cargo cargo1 = new Cargo(new TrackingId("TRACK1"), null);
    cargo1.getDelivery().setTransportStatus(TransportStatus.CLAIMED);
    Cargo cargo2 = new Cargo(new TrackingId("TRACK2"), null);
    cargos.add(cargo1);
    cargos.add(cargo2);

    List<Location> locations = new ArrayList<>();
    Location location1 = new Location(new UnLocode("UNLOC1"), "Location 1");
    locations.add(location1);

    List<Voyage> voyages = new ArrayList<>();
    Voyage voyage1 = new Voyage(new org.eclipse.cargotracker.domain.model.voyage.VoyageNumber("VOYAGE1"), null);
    voyages.add(voyage1);

    when(cargoRepository.findAll()).thenReturn(cargos);
    when(locationRepository.findAll()).thenReturn(locations);
    when(voyageRepository.findAll()).thenReturn(voyages);

    eventLogger.init();

    assertNotNull(eventLogger.getTrackingIds());
    assertEquals(1, eventLogger.getTrackingIds().size());

    assertNotNull(eventLogger.getLocations());
    assertEquals(1, eventLogger.getLocations().size());

    assertNotNull(eventLogger.getVoyages());
    assertEquals(1, eventLogger.getVoyages().size());

    assertEquals("TRACK2", eventLogger.getTrackingIds().get(0).getValue());
    assertEquals("Location 1 (UNLOC1)", eventLogger.getLocations().get(0).getValue());
    assertEquals("VOYAGE1", eventLogger.getVoyages().get(0).getValue());
  }

  @Test
  public void testInit_noCargo() {
