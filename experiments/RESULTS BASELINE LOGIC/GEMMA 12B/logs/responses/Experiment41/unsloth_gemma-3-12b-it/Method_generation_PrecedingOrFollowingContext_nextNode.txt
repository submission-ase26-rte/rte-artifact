public boolean nextNode() {
    if (rootNode == null || rootNode.isRoot()) {
        return false;
    }

    NodeIterator iterator = rootNode.childIterator(nodeTest, reverse, rootNode);
    if (iterator == null) {
        return false;
    }

    if (!setPosition(1)) {
        return false;
    }

    NodePointer nextNode = iterator.getNodePointer();
    if (nextNode != null && nextNode.testNode(nodeTest)) {
        currentNodePointer = nextNode;
        return true;
    }

    int position = 1;
    while (setPosition(++position)) {
        nextNode = iterator.getNodePointer();
        if (nextNode != null && nextNode.testNode(nodeTest)) {
            currentNodePointer = nextNode;
            return true;
        }
    }

    return false;
}
