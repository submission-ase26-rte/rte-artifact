@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public CommandLine parse(final Options options, final String[] arguments, final Properties properties, final boolean stopAtNonOption) throws ParseException {
        cmd = new CommandLine();
        this.options = options;
        requiredOptions = new ArrayList<>();
        for (Option option : options.getOptions()) {
            if (option.isRequired()) {
                requiredOptions.add(option);
            }
        }
        processProperties(properties);
        List args = new ArrayList(Arrays.asList(arguments));
        int pos = 0;
        while (!args.isEmpty()) {
            String arg = (String) args.get(0);
            args.remove(0);
            if (arg.startsWith("-")) {
                Option option = options.getOption(arg);
                if (option != null) {
                    try {
                        processOption(arg, options.iterator());
                    } catch (ParseException e) {
                        throw e;
                    }
                } else {
                    throw new ParseException("Unrecognized option: " + arg);
                }
            } else {
                cmd.addArg(arg);
            }
        }
        checkRequiredOptions();
        return cmd;
    }
