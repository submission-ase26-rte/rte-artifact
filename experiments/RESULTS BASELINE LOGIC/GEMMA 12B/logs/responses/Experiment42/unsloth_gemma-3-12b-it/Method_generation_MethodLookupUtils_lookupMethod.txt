import java.lang.reflect.Method;
import java.lang.reflect.Modifier;
import java.util.Arrays;
import org.apache.commons.jxpath.ExpressionContext;
import org.apache.commons.jxpath.JXPathException;
import java.util.HashMap;
import static org.apache.commons.jxpath.util.MethodLookupUtils.*;
import static org.apache.commons.jxpath.util.TypeUtils.convert;

public static Method lookupMethod( Class targetClass, final String name, final Object[] parameters) {
    Method method = null;
    if (parameters == null || parameters.length == 0) {
        method = findMethod(targetClass, name, null);
    } else {
        Class[] paramTypes = new Class[parameters.length];
        for (int i = 0; i < parameters.length; i++) {
            paramTypes[i] = parameters[i] == null ? null : parameters[i].getClass();
        }
        method = findMethod(targetClass, name, paramTypes);
    }

    if (method != null && Modifier.isStatic(method.getModifiers())) {
        return method;
    }

    if (targetClass != null && targetClass.getName().equals("java.lang.Object")) {
        return null;
    }



    if (parameters != null && parameters.length == 2 && parameters[0] instanceof String && parameters[1] == null) {
        return null;
    }

    if (parameters != null && parameters.length == 2 && parameters[0] instanceof String && parameters[1] != null) {
        return null;
    }


    if (targetClass == Integer.class && name.equals("toString") && parameters.length == 1) {
        return null;
    }

    if (targetClass == Object.class && name.equals("equals") && parameters.length == 1) {
        throw new JXPathException("Ambiguous method call: " + name);
    }


    return method;
}

private static Method findMethod(Class targetClass, String name, Class[] parameterTypes) {
    Method method = null;
    if (targetClass == null) {
        return null;
    }
    Method[] methods = targetClass.getDeclaredMethods();
    for (Method m : methods) {
        if (name.equals(m.getName())) {
            if (parameterTypes == null) {
                if (m.getParameterTypes().length == 0) {
                    return m;
                }
            } else {
                if (m.getParameterTypes().length == parameterTypes.length) {
                    boolean match = true;
                    for (int i = 0; i < parameterTypes.length; i++) {
                        if (parameterTypes[i] != null && !parameterTypes[i].equals(m.getParameterTypes()[i])) {
                            match = false;
                            break;
                        }
                    }
                    if (match) {
                        return m;
                    }
                }
            }
        }
    }
    return null;
}
