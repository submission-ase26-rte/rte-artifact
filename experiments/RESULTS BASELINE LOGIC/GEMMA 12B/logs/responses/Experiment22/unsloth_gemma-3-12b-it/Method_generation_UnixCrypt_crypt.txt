public static String crypt(final byte[] original, String salt) {
    if (salt == null) {
        SecureRandom random = new SecureRandom();
        byte[] saltBytes = new byte[8];
        random.nextBytes(saltBytes);
        salt = bytesToHex(saltBytes);
    }

    if (salt.length() < 2) {
        throw new IllegalArgumentException("Salt must be at least two characters long");
    }

    if (!Pattern.matches("[a-zA-Z0-9./+]", salt)) {
        throw new IllegalArgumentException("Salt contains invalid characters");
    }

    String result = UnixCrypt.crypt(original, salt, true);
    return result;
}

private static String bytesToHex(byte[] bytes) {
    StringBuilder sb = new StringBuilder();
    for (byte b : bytes) {
        sb.append(String.format("%02x", b));
    }
    return sb.toString();
