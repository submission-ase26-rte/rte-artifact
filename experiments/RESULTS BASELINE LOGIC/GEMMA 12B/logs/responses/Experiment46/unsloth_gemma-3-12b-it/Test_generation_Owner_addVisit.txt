public class Owner_addVisitTestGenerated {

    private Owner owner;
    private Pet pet;
    private Visit visit;

    @BeforeEach
    public void setUp() {
        owner = new Owner();
        owner.setAddress("123 Main St");
        owner.setCity("Anytown");
        owner.setTelephone("555-1212");

        pet = new Pet();
        pet.setBirthDate(LocalDate.now());
        pet.setType(PetType.DOG);
        pet.setName("Buddy");
        Set<Visit> visits = new LinkedHashSet<>();
        pet.setVisits(visits);

        visit = new Visit();
        visit.setDate(LocalDate.now());
        visit.setDescription("Annual checkup");
    }

    @AfterEach
    public void tearDown() {
        owner = null;
        pet = null;
        visit = null;
    }

    @Test
    public void testAddVisitValid() {
        owner.getPets().add(pet);
        pet.addVisit(visit);
        assertNotNull(pet.getVisits());
        assertEquals(1, pet.getVisits().size());
    }

    @Test
    public void testAddVisitPetNotFound() {
        assertThrows(IllegalArgumentException.class, () -> {
            owner = new Owner();
            owner.addVisit(999, visit);
        });
    }

    @Test
    public void testAddVisitNullPetId() {
        assertThrows(IllegalArgumentException.class, () -> {
            owner = new Owner();
            owner.addVisit(null, visit);
        });
    }

    @Test
    public void testAddVisitNullVisit() {
        owner.getPets().add(pet);
        assertThrows(IllegalArgumentException.class, () -> {
            owner.addVisit(pet.getId(), null);
        });
    }

    @Test
    public void testAddVisitExistingPet() {
        owner.getPets().add(pet);
        pet.addVisit(visit);
        Visit addedVisit = null;
        for (Visit v : pet.getVisits()) {
            if (v.getDescription().equals("Annual checkup")) {
                addedVisit = v;
                break;
            }
        }
        assertNotNull(addedVisit);
        assertEquals("Annual checkup", addedVisit.getDescription());
    }
}
