  @Test
  public void testFetchRoutesForSpecification_validSpecification() {
    Location origin = new Location("origin");
    Location destination = new Location("destination");
    LocalDate arrivalDeadline = LocalDate.now();
    RouteSpecification routeSpecification =
        new RouteSpecification(origin, destination, arrivalDeadline);

    List<TransitPath> transitPaths = new ArrayList<>();
    TransitPath transitPath1 = Mockito.mock(TransitPath.class);
    when(transitPath1.getArrivalDate()).thenReturn(LocalDate.now().plusDays(1));
    TransitPath transitPath2 = Mockito.mock(TransitPath.class);
    when(transitPath2.getArrivalDate()).thenReturn(LocalDate.now().plusDays(2));
    transitPaths.add(transitPath1);
    transitPaths.add(transitPath2);

    WebTarget graphTraversalResource = externalRoutingService.graphTraversalResource;
    when(graphTraversalResource.queryParam("origin", origin.getUnLocode().getIdString())
            .queryParam("destination", destination.getUnLocode().getIdString())
            .request(MediaType.APPLICATION_JSON_TYPE)
            .get(new GenericType<List<TransitPath>>() {}))
        .thenReturn(transitPaths);

    List<Itinerary> itineraries =
        externalRoutingService.fetchRoutesForSpecification(routeSpecification);

    assertEquals(2, itineraries.size());
    assertTrue(itineraries.get(0).getVoyageNumbers().size() > 0);
    assertTrue(itineraries.get(1).getVoyageNumbers().size() > 0);
  }
