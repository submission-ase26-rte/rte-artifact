package org.apache.commons.compress.archivers.ar;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class ArArchiveInputStream_matchesTestGenerated {

    @BeforeEach
    public void setUp() {
        // Setup code if needed
    }

    @Test
    public void testMatchesValidSignature() {
        byte[] validSignature = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e, 0x0a};
        assertTrue(ArArchiveInputStream.matches(validSignature, 8));
    }

    @Test
    public void testMatchesInvalidSignature() {
        byte[] invalidSignature = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e, 0x0b};
        assertFalse(ArArchiveInputStream.matches(invalidSignature, 8));
    }

    @Test
    public void testMatchesTooShortSignature() {
        byte[] tooShortSignature = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e};
        assertFalse(ArArchiveInputStream.matches(tooShortSignature, 7));
    }

    @Test
    public void testMatchesLengthZero() {
        byte[] emptySignature = new byte[]{};
        assertFalse(ArArchiveInputStream.matches(emptySignature, 0));
    }

    @Test
    public void testMatchesNullSignature() {
        assertFalse(ArArchiveInputStream.matches(null, 8));
    }

    @Test
    public void testMatchesLengthLessThanSignatureLength() {
        byte[] validSignature = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e, 0x0a};
        assertFalse(ArArchiveInputStream.matches(validSignature, 7));
    }

    @Test
    public void testMatchesLengthGreaterThanSignatureLength() {
        byte[] validSignature = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e, 0x0a};
        assertTrue(ArArchiveInputStream.matches(validSignature, 9));
    }

    @Test
    public void testMatchesEmptySignature() {
        byte[] emptySignature = new byte[]{};
        assertFalse(ArArchiveInputStream.matches(emptySignature, 8));
    }

    @Test
    public void testMatchesSignatureWithExtraBytes() {
        byte[] signatureWithExtraBytes = new byte[]{0x21, 0x3c, 0x61, 0x72, 0x63, 0x68, 0x3e, 0x0a, 0x00};
        assertTrue(ArArchiveInputStream.matches(signatureWithExtraBytes, 8));
    }
}
