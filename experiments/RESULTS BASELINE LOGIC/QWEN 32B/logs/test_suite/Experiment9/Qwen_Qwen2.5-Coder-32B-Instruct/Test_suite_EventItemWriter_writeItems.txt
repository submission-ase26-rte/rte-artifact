@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class EventItemWriter_writeItemsTestGenerated {

    @Mock
    private JobContext jobContext;

    @Mock
    private ApplicationEvents applicationEvents;

    @InjectMocks
    private EventItemWriter eventItemWriter;

    private Path archiveFile;

    @BeforeEach
    void setUp() throws IOException {
        MockitoAnnotations.openMocks(this);
        ReflectionTestUtils.setField(eventItemWriter, "ARCHIVE_DIRECTORY", "target/test-archives");
        Files.createDirectories(Path.of("target/test-archives"));
        archiveFile = Path.of("target/test-archives/archive_testJob_1.csv");
        when(jobContext.getProperties().getProperty("target/test-archives")).thenReturn("target/test-archives");
        when(jobContext.getJobName()).thenReturn("testJob");
        when(jobContext.getInstanceId()).thenReturn(1L);
    }

    @AfterEach
    void tearDown() throws IOException {
        Files.deleteIfExists(archiveFile);
        Files.deleteIfExists(Path.of("target/test-archives"));
    }

    @Test
    void testWriteItemsWithEmptyList() throws Exception {
        List<Object> items = Collections.emptyList();

        eventItemWriter.writeItems(items);

        assertTrue(Files.exists(archiveFile));
        List<String> lines = Files.readAllLines(archiveFile);
        assertEquals(0, lines.size());
        verify(applicationEvents, never()).receivedHandlingEventRegistrationAttempt(any());
    }

    @Test
    void testWriteItemsWithNullList() throws Exception {
        List<Object> items = null;

        Exception exception = assertThrows(NullPointerException.class, () -> {
            eventItemWriter.writeItems(items);
        });

        assertNotNull(exception);
        assertFalse(Files.exists(archiveFile));
        verify(applicationEvents, never()).receivedHandlingEventRegistrationAttempt(any());
    }
}
