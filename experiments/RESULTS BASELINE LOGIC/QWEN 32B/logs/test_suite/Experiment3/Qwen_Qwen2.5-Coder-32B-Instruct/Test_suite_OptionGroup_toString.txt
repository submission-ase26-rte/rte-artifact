package org.apache.commons.cli;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import java.util.Collection;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class OptionGroup_toStringTestGenerated {

    private OptionGroup optionGroup;

    @BeforeEach
    public void setUp() {
        optionGroup = new OptionGroup();
    }

    @AfterEach
    public void tearDown() {
        optionGroup = null;
    }

    @Test
    public void testToStringWithNoOptions() {
        assertEquals("[]", optionGroup.toString());
    }

    @Test
    public void testToStringWithOneOptionShortOpt() {
        Option option = new Option("a", "Option A");
        optionGroup.addOption(option);
        assertEquals("[-a Option A]", optionGroup.toString());
    }

    @Test
    public void testToStringWithOneOptionLongOpt() {
        Option option = new Option("a", "alpha", false, "Option A");
        optionGroup.addOption(option);
        assertEquals("[--alpha Option A]", optionGroup.toString());
    }

    @Test
    public void testToStringWithMultipleOptions() {
        Option option1 = new Option("a", "Option A");
        Option option2 = new Option("b", "Option B");
        optionGroup.addOption(option1);
        optionGroup.addOption(option2);
        assertEquals("[-a Option A, -b Option B]", optionGroup.toString());
    }

    @Test
    public void testToStringWithMixedShortAndLongOptions() {
        Option option1 = new Option("a", "Option A");
        Option option2 = new Option("b", "beta", false, "Option B");
        optionGroup.addOption(option1);
        optionGroup.addOption(option2);
        assertEquals("[-a Option A, --beta Option B]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullOptionDescription() {
        Option option = new Option("a", (String) null);
        optionGroup.addOption(option);
        assertEquals("[-a]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullOptionOpt() {
        Option option = new Option(null, "alpha", false, "Option A");
        optionGroup.addOption(option);
        assertEquals("[--alpha Option A]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullOptionOptAndLongOpt() {
        Option option = new Option(null, (String) null, false, "Option A");
        optionGroup.addOption(option);
        assertEquals("[--null Option A]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullOptionDescriptionAndOpt() {
        Option option = new Option(null, (String) null);
        optionGroup.addOption(option);
        assertEquals("[--null]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullOption() {
        optionGroup.addOption(null);
        assertEquals("[]", optionGroup.toString());
    }

    @Test
    public void testToStringWithMultipleNullOptions() {
        optionGroup.addOption(null);
        optionGroup.addOption(null);
        assertEquals("[]", optionGroup.toString());
    }

    @Test
    public void testToStringWithNullAndValidOptions() {
        Option option = new Option("a", "Option A");
        optionGroup.addOption(option);
        optionGroup.addOption(null);
        assertEquals("[-a Option A]", optionGroup.toString());
    }
}
