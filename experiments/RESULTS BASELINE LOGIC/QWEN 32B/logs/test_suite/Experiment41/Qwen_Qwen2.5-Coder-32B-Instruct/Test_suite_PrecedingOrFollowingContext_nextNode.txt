package org.apache.commons.jxpath.ri.axes;

import static org.junit.jupiter.api.Assertions.*;
import org.apache.commons.jxpath.ri.EvalContext;
import org.apache.commons.jxpath.ri.compiler.NodeTest;
import org.apache.commons.jxpath.ri.model.NodeIterator;
import org.apache.commons.jxpath.ri.model.NodePointer;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mockito;

import java.util.Stack;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class PrecedingOrFollowingContext_nextNodeTestGenerated {

    private PrecedingOrFollowingContext context;
    private EvalContext parentContext;
    private NodeTest nodeTest;
    private NodePointer nodePointer;
    private NodeIterator nodeIterator;

    @BeforeEach
    void setUp() {
        parentContext = Mockito.mock(EvalContext.class);
        nodeTest = Mockito.mock(NodeTest.class);
        nodePointer = Mockito.mock(NodePointer.class);
        nodeIterator = Mockito.mock(NodeIterator.class);
        context = new PrecedingOrFollowingContext(parentContext, nodeTest, false);
    }

    @Test
    void testNextNode_InitialSetup() {
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(null);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_WithParent() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(true);
        Mockito.when(nodeIterator.getNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.testNode(nodeTest)).thenReturn(true);

        assertTrue(context.nextNode());
    }

    @Test
    void testNextNode_WithParentAndChild() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        NodePointer child = Mockito.mock(NodePointer.class);
        NodeIterator childIterator = Mockito.mock(NodeIterator.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(true);
        Mockito.when(nodeIterator.getNodePointer()).thenReturn(child);
        Mockito.when(child.isLeaf()).thenReturn(false);
        Mockito.when(child.childIterator(null, false, null)).thenReturn(childIterator);
        Mockito.when(childIterator.setPosition(1)).thenReturn(true);
        Mockito.when(childIterator.getNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.testNode(nodeTest)).thenReturn(true);

        assertTrue(context.nextNode());
    }

    @Test
    void testNextNode_NoMatchingNode() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(true);
        Mockito.when(nodeIterator.getNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.testNode(nodeTest)).thenReturn(false);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_ReverseOrder() {
        context = new PrecedingOrFollowingContext(parentContext, nodeTest, true);
        NodePointer parent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, true, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(true);
        Mockito.when(nodeIterator.getNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.testNode(nodeTest)).thenReturn(true);

        assertTrue(context.nextNode());
    }

    @Test
    void testNextNode_EmptyStack() {
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(null);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_WithMultipleParents() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        NodePointer grandParent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.getParent()).thenReturn(grandParent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(false);
        Mockito.when(grandParent.childIterator(null, false, parent)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(true);
        Mockito.when(nodeIterator.getNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.testNode(nodeTest)).thenReturn(true);

        assertTrue(context.nextNode());
    }

    @Test
    void testNextNode_WithNullNodePointer() {
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(null);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_WithNullParent() {
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(null);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_WithNullNodeIterator() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(null);

        assertFalse(context.nextNode());
    }

    @Test
    void testNextNode_WithEmptyNodeIterator() {
        NodePointer parent = Mockito.mock(NodePointer.class);
        Mockito.when(parentContext.getCurrentNodePointer()).thenReturn(nodePointer);
        Mockito.when(nodePointer.getParent()).thenReturn(parent);
        Mockito.when(parent.childIterator(null, false, nodePointer)).thenReturn(nodeIterator);
        Mockito.when(nodeIterator.setPosition(1)).thenReturn(false);

        assertFalse(context.nextNode());
    }
}
