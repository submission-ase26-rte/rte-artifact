package org.apache.commons.cli;

import static org.junit.jupiter.api.Assertions.assertEquals;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class Option_toStringTestGenerated {

    private Option option;

    @BeforeEach
    public void setUp() {
        option = new Option("o", "option", false, "description");
    }

    @Test
    public void testToStringNoLongOptionNoArgsNoDeprecated() {
        assertEquals("[ Option o :: description :: class java.lang.String ]", option.toString());
    }

    @Test
    public void testToStringWithLongOptionNoArgsNoDeprecated() {
        option.setLongOpt("longOption");
        assertEquals("[ Option o longOption :: description :: class java.lang.String ]", option.toString());
    }

    @Test
    public void testToStringNoLongOptionWithArgsNoDeprecated() {
        option.setArgs(1);
        assertEquals("[ Option o [ARG...] :: description :: class java.lang.String ]", option.toString());
    }

    @Test
    public void testToStringNoLongOptionWithOptionalArgNoDeprecated() {
        option.setOptionalArg(true);
        assertEquals("[ Option o [ARG] :: description :: class java.lang.String ]", option.toString());
    }

    @Test
    public void testToStringWithCustomType() {
        option.setType(Integer.class);
        assertEquals("[ Option o :: description :: class java.lang.Integer ]", option.toString());
    }
}
