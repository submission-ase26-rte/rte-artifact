package org.apache.commons.compress.archivers.cpio;

import static org.junit.jupiter.api.Assertions.assertEquals;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class CpioArchiveEntry_setInodeTestGenerated {

    private CpioArchiveEntry entry;

    @BeforeEach
    public void setUp() {
        entry = new CpioArchiveEntry("testFile");
    }

    @Test
    public void testSetInode_PositiveValue() {
        entry.setInode(12345L);
        assertEquals(12345L, entry.getInode());
    }

    @Test
    public void testSetInode_ZeroValue() {
        entry.setInode(0L);
        assertEquals(0L, entry.getInode());
    }

    @Test
    public void testSetInode_NegativeValue() {
        entry.setInode(-12345L);
        assertEquals(-12345L, entry.getInode());
    }

    @Test
    public void testSetInode_MaxValue() {
        entry.setInode(Long.MAX_VALUE);
        assertEquals(Long.MAX_VALUE, entry.getInode());
    }

    @Test
    public void testSetInode_MinValue() {
        entry.setInode(Long.MIN_VALUE);
        assertEquals(Long.MIN_VALUE, entry.getInode());
    }
}
