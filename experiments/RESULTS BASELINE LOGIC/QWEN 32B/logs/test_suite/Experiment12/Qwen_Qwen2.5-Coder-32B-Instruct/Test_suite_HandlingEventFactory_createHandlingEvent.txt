    @Test
    public void testCreateHandlingEventWithVoyage() throws CannotCreateHandlingEventException {
        LocalDateTime registrationTime = LocalDateTime.now();
        LocalDateTime completionTime = LocalDateTime.now().plusHours(1);
        TrackingId trackingId = new TrackingId("XYZ123");
        VoyageNumber voyageNumber = new VoyageNumber("VOY123");
        UnLocode unlocode = new UnLocode("USNYC");
        HandlingEvent.Type type = HandlingEvent.Type.LOAD;

        Cargo cargo = mock(Cargo.class);
        Voyage voyage = mock(Voyage.class);
        Location location = mock(Location.class);

        when(cargoRepository.find(trackingId)).thenReturn(cargo);
        when(voyageRepository.find(voyageNumber)).thenReturn(voyage);
        when(locationRepository.find(unlocode)).thenReturn(location);

        HandlingEvent handlingEvent = handlingEventFactory.createHandlingEvent(
                registrationTime,
                completionTime,
                trackingId,
                voyageNumber,
                unlocode,
                type
        );

        assertNotNull(handlingEvent);
        assertEquals(cargo, handlingEvent.getCargo());
        assertEquals(completionTime, handlingEvent.getCompletionTime());
        assertEquals(registrationTime, handlingEvent.getRegistrationTime());
        assertEquals(type, handlingEvent.getType());
        assertEquals(location, handlingEvent.getLocation());
        assertEquals(voyage, handlingEvent.getVoyage());
    }
