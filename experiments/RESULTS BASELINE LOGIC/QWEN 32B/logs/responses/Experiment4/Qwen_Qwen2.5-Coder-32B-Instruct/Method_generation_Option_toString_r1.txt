public String toString() {
    StringBuilder buf = new StringBuilder("[ Option ");
    buf.append(getOpt());
    if (getLongOpt() != null) {
        buf.append(" ").append(getLongOpt());
    }
    if (getArgs() > 0) {
        buf.append(" [ARG");
        if (getArgs() > 1 || isOptionalArg()) {
            buf.append("...");
        }
        buf.append("]");
    } else if (getArgName() != null && getArgName().equals("DEPRECATED")) {
        buf.append(" ").append(getArgName());
    }
    buf.append(" :: ").append(getDescription());
    buf.append(" :: ").append(getType() != null ? getType().getName() : DEFAULT_TYPE.getName()).append(" ]");
    return buf.toString();
}
