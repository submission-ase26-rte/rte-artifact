public CommandLine parse(final Options options, final String[] arguments, final Properties properties, final boolean stopAtNonOption) throws ParseException {
    if (options == null) {
        throw new ParseException("options is null");
    }

    this.options = options;
    this.stopAtNonOption = stopAtNonOption;
    this.cmd = new CommandLine.Builder().build();
    this.expectedOpts = new ArrayList<>(options.getOptions());

    handleProperties(properties);

    for (String arg : arguments) {
        handleToken(arg);
    }

    checkRequiredOptions();
    checkRequiredArgs();

    return cmd;
}

private void handleProperties(final Properties properties) {
    if (properties == null) {
        return;
    }

    Enumeration<?> propertyNames = properties.propertyNames();
    while (propertyNames.hasMoreElements()) {
        String propertyName = (String) propertyNames.nextElement();
        String propertyValue = properties.getProperty(propertyName);
        Option opt = options.getOption(propertyName);
        if (opt != null && opt.hasArg() && !opt.isRequired()) {
            cmd.addOption(opt);
            cmd.addValueForOption(opt, propertyValue);
        }
    }
}

private void handleToken(final String token) {
    if (isOption(token)) {
        currentToken = token;
        currentOption = options.getOption(stripLeadingAndTrailingQuotes ? stripLeadingAndTrailingQuotes(token) : token);
        if (currentOption == null) {
            throw new ParseException("Unrecognized option: " + token);
        }
        if (currentOption.hasArg()) {
            String arg = getNextArgument();
            cmd.addOption(currentOption);
            cmd.addValueForOption(currentOption, arg);
        } else {
            cmd.addOption(currentOption);
        }
    } else {
        if (stopAtNonOption) {
            throw new ParseException("Encountered non-option argument: " + token);
        }
        cmd.addArg(token);
    }
}

private String getNextArgument() {
    if (currentToken != null && currentToken.startsWith("--")) {
        return "";
    }
    return "";
}

private void checkRequiredOptions() throws ParseException {
    List<String> missingOptions = new ArrayList<>();
    for (String requiredOption : options.getRequiredOptions()) {
        if (!cmd.hasOption(requiredOption)) {
            missingOptions.add(requiredOption);
        }
    }
    if (!missingOptions.isEmpty()) {
        throw new ParseException("Missing required options: " + String.join(", ", missingOptions));
    }

    for (Object groupObj : options.getOptionGroups()) {
        OptionGroup group = (OptionGroup) groupObj;
        if (group.isRequired() && group.getSelected() == null) {
            throw new ParseException("Option group " + group + " does not contain a selected option");
        }
    }
}

private void checkRequiredArgs() throws ParseException {
    for (Option option : cmd.getOptions()) {
        if (option.hasArgs() && option.isRequired() && option.getValues() == null) {
            throw new ParseException("Missing argument for option: " + option.getOpt());
        }
    }
}

private boolean isOption(final String token) {
    return isShortOption(token) || isLongOption(token);
}

private boolean isShortOption(final String token) {
    return token.length() == 2 && token.charAt(0) == '-';
}

private boolean isLongOption(final String token) {
    return token.startsWith("--");
}

private String stripLeadingAndTrailingQuotes(String token) {
    if (token.length() > 1 && token.charAt(0) == '"' && token.charAt(token.length() - 1) == '"') {
        return token.substring(1, token.length() - 1);
    }
    return token;
}
