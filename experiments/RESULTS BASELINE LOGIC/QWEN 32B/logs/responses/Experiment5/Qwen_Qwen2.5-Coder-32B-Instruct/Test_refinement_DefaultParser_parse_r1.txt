package org.apache.commons.cli;

import java.util.Properties;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;
import org.junit.jupiter.api.function.Executable;

import static org.junit.jupiter.api.Assertions.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class DefaultParser_parseTestGenerated {
    private DefaultParser parser;
    private Options options;
    private Properties properties;

    @BeforeEach
    public void setUp() {
        parser = new DefaultParser();
        options = new Options();
        properties = new Properties();
    }

    @AfterEach
    public void tearDown() {
        parser = null;
        options = null;
        properties = null;
    }

    @Test
    public void testParse_WithOptionGroup_NoSelection() {
        OptionGroup group = new OptionGroup();
        group.addOption(new Option("a", "alpha", false, "alpha option"));
        group.addOption(new Option("b", "beta", false, "beta option"));
        group.setRequired(true);
        options.addOptionGroup(group);
        Executable executable = () -> parser.parse(options, new String[]{}, properties, false);
        assertThrows(ParseException.class, executable);
    }

    @Test
    public void testParse_WithNullOptions() {
        Executable executable = () -> parser.parse(null, new String[]{}, properties, false);
        assertThrows(ParseException.class, executable);
    }
}
