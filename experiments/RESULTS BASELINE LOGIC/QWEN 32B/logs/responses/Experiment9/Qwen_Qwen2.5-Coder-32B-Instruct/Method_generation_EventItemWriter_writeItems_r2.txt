public void writeItems(List<Object> items) throws Exception {
    if (items == null) {
        throw new NullPointerException("Items list cannot be null");
    }

    String archivePath = ARCHIVE_DIRECTORY + "/" + jobContext.getJobName() + "_" + jobContext.getInstanceId() + ".csv";
    Path archiveFile = Path.of(archivePath);
    Files.createDirectories(archiveFile.getParent());

    if (items.isEmpty()) {
        Files.createFile(archiveFile);
        return;
    }

    List<String> lines = items.stream()
        .map(item -> (HandlingEventRegistrationAttempt) item)
        .map(attempt -> DateConverter.toString(attempt.getCompletionTime()) + "," + attempt.getTrackingId().getId() + "," + 
                         attempt.getVoyageNumber().getVoyageNumber() + "," + attempt.getEventType() + "," + 
                         attempt.getLocation().getUnLocode().getId())
        .toList();

    Files.write(archiveFile, lines);

    for (Object item : items) {
        applicationEvents.receivedHandlingEventRegistrationAttempt((HandlingEventRegistrationAttempt) item);
    }
}
