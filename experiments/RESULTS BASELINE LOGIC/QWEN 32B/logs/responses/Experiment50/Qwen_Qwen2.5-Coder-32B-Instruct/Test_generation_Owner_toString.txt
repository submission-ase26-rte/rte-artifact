package org.springframework.samples.petclinic.owner;

import java.util.ArrayList;
import java.util.List;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.springframework.core.style.ToStringCreator;
import org.springframework.test.util.ReflectionTestUtils;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.mockito.Mockito.*;

public class Owner_toStringTestGenerated {

    @Mock
    private List<Pet> pets;

    @InjectMocks
    private Owner owner;

    private AutoCloseable closeable;

    @BeforeEach
    public void setUp() {
        closeable = MockitoAnnotations.openMocks(this);
        ReflectionTestUtils.setField(owner, "address", "123 Main St");
        ReflectionTestUtils.setField(owner, "city", "Springfield");
        ReflectionTestUtils.setField(owner, "telephone", "555-1234");
        ReflectionTestUtils.setField(owner, "pets", new ArrayList<>());
    }

    @AfterEach
    public void tearDown() throws Exception {
        closeable.close();
    }

    @Test
    public void testToStringWithAllFieldsSet() {
        ReflectionTestUtils.setField(owner, "id", 1);
        when(owner.isNew()).thenReturn(false);
        when(owner.getLastName()).thenReturn("Doe");
        when(owner.getFirstName()).thenReturn("John");

        String expected = new ToStringCreator(owner)
                .append("id", 1)
                .append("new", false)
                .append("lastName", "Doe")
                .append("firstName", "John")
                .append("address", "123 Main St")
                .append("city", "Springfield")
                .append("telephone", "555-1234")
                .toString();

        assertEquals(expected, owner.toString());
    }

    @Test
    public void testToStringWithNullFields() {
        ReflectionTestUtils.setField(owner, "id", null);
        when(owner.isNew()).thenReturn(true);
        when(owner.getLastName()).thenReturn(null);
        when(owner.getFirstName()).thenReturn(null);

        String expected = new ToStringCreator(owner)
                .append("id", null)
                .append("new", true)
                .append("lastName", null)
                .append("firstName", null)
                .append("address", "123 Main St")
                .append("city", "Springfield")
                .append("telephone", "555-1234")
                .toString();

        assertEquals(expected, owner.toString());
    }

    @Test
    public void testToStringWithEmptyFields() {
        ReflectionTestUtils.setField(owner, "id", 0);
        when(owner.isNew()).thenReturn(false);
        when(owner.getLastName()).thenReturn("");
        when(owner.getFirstName()).thenReturn("");

        String expected = new ToStringCreator(owner)
                .append("id", 0)
                .append("new", false)
                .append("lastName", "")
                .append("firstName", "")
                .append("address", "123 Main St")
                .append("city", "Springfield")
                .append("telephone", "555-1234")
                .toString();

        assertEquals(expected, owner.toString());
    }
}
