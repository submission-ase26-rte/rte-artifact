public static String md5Crypt(final byte[] keyBytes, final String salt, final String prefix, final Random random) {
    if (keyBytes == null) {
        throw new NullPointerException("Key bytes must not be null");
    }
    if (salt == null) {
        salt = getRandomSalt(8);
    } else if (!salt.startsWith(prefix)) {
        throw new IllegalArgumentException("Salt must start with the prefix: " + prefix);
    } else {
        String saltPart = salt.substring(prefix.length());
        if (saltPart.length() < 2 || saltPart.length() > 8 || !saltPart.matches("[a-zA-Z0-9./]+")) {
            throw new IllegalArgumentException("Salt must be between 2 and 8 characters and only contain a-z, A-Z, 0-9, '.', '/'");
        }
    }

    StringBuilder saltBuf = new StringBuilder(salt.substring(prefix.length()));
    if (saltBuf.length() > 8) {
        saltBuf.setLength(8);
    }

    StringBuilder buffer = new StringBuilder(prefix);
    buffer.append(saltBuf);

    MessageDigest md = getMd5Digest();
    md.update(keyBytes);
    md.update(buffer.toString().getBytes(StandardCharsets.UTF_8));

    byte[] finalState = md.digest();

    md.reset();
    md.update(keyBytes);
    md.update(saltBuf.toString().getBytes(StandardCharsets.UTF_8));
    md.update(keyBytes);

    byte[] keyBytesCopy = keyBytes.clone();
    int keyBytesLength = keyBytesCopy.length;
    while (keyBytesLength > 0) {
        md.update(keyBytesCopy, 0, keyBytesLength > 16 ? 16 : keyBytesLength);
        keyBytesLength -= 16;
    }

    keyBytesCopy = new byte[16];
    Arrays.fill(keyBytesCopy, (byte) 0);
    keyBytesLength = keyBytes.length;
    while (keyBytesLength > 0) {
        if ((keyBytesLength & 1) != 0) {
            md.update(keyBytesCopy, 0, 1);
        } else {
            md.update(keyBytes, 0, 1);
        }
        keyBytesLength >>= 1;
    }

    finalState = md.digest();

    for (int i = keyBytes.length; i > 0; i >>= 1) {
        if ((i & 1) != 0) {
            md.update(finalState, 0, 1);
        } else {
            md.update(keyBytes, 0, 1);
        }
    }

    finalState = md.digest();

    for (int i = 0; i < 1000; i++) {
        md.reset();
        if ((i & 1) != 0) {
            md.update(keyBytes);
        } else {
            md.update(finalState);
        }

        if (i % 3 != 0) {
            md.update(saltBuf.toString().getBytes(StandardCharsets.UTF_8));
        }

        if (i % 7 != 0) {
            md.update(keyBytes);
        }

        if ((i & 1) != 0) {
            md.update(finalState);
        } else {
            md.update(keyBytes);
        }

        finalState = md.digest();
    }

    byte[] pBytes = new byte[16];
    byte[] sBytes = new byte[16];
    for (int i = 0; i < 16; i++) {
        pBytes[i] = finalState[i];
        sBytes[i] = finalState[16 - i - 1];
    }

    for (int i = 0; i < 4; i++) {
        b64from24bit(pBytes[i * 3 + 0], pBytes[i * 3 + 1], pBytes[i * 3 + 2], 4, buffer);
    }

    b64from24bit(pBytes[12], pBytes[13], sBytes[11], 3, buffer);

    return buffer.toString();
}
