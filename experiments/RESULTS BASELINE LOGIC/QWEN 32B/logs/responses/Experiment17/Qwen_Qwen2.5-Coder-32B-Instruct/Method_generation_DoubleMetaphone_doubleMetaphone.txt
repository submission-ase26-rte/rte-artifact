public String doubleMetaphone(String value, final boolean alternate) {
    if (value == null) {
        return null;
    }
    value = cleanInput(value);
    if (value.isEmpty()) {
        return "";
    }
    DoubleMetaphoneResult result = new DoubleMetaphoneResult(maxCodeLen);
    boolean slavoGermanic = isSlavoGermanic(value);
    int index = 0;
    if (value.startsWith("GN")) {
        index = 2;
        result.append('K', 'N');
    } else if (value.startsWith("KN")) {
        index = 2;
        result.append('N', 'N');
    } else if (value.startsWith("W")) {
        if (value.charAt(1) == 'R') {
            index = 2;
        } else if (isVowel(value.charAt(1))) {
            index = 1;
        }
        result.append('W', 'W');
    } else if (value.startsWith("X")) {
        index = 1;
        result.append('S', 'S');
    }
    while (index < value.length() && result.length() < maxCodeLen) {
        char ch = value.charAt(index);
        switch (ch) {
            case 'A':
            case 'E':
            case 'I':
            case 'O':
            case 'U':
            case 'Y':
                index = handleAEIOUY(result, index);
                break;
            case 'B':
                result.append('P');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'B') ? index + 2 : index + 1;
                break;
            case 'Ç':
                result.append('S');
                index++;
                break;
            case 'C':
                index = handleC(value, result, index);
                break;
            case 'D':
                index = handleD(value, result, index);
                break;
            case 'F':
                result.append('F');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'F') ? index + 2 : index + 1;
                break;
            case 'G':
                index = handleG(value, result, index, slavoGermanic);
                break;
            case 'H':
                index = handleH(value, result, index);
                break;
            case 'J':
                index = handleJ(value, result, index, slavoGermanic);
                break;
            case 'K':
                result.append('K');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'K') ? index + 2 : index + 1;
                break;
            case 'L':
                index = handleL(value, result, index);
                break;
            case 'M':
                result.append('M');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'M') ? index + 2 : index + 1;
                break;
            case 'N':
                result.append('N');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'N') ? index + 2 : index + 1;
                break;
            case 'Ñ':
                result.append('N');
                index++;
                break;
            case 'P':
                index = handleP(value, result, index);
                break;
            case 'Q':
                result.append('K');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'Q') ? index + 2 : index + 1;
                break;
            case 'R':
                index = handleR(value, result, index, slavoGermanic);
                break;
            case 'S':
                index = handleS(value, result, index, slavoGermanic);
                break;
            case 'T':
                index = handleT(value, result, index);
                break;
            case 'V':
                result.append('F');
                index = (index + 1 < value.length() && value.charAt(index + 1) == 'V') ? index + 2 : index + 1;
                break;
            case 'W':
                index = handleW(value, result, index);
                break;
            case 'X':
                index = handleX(value, result, index);
                break;
            case 'Z':
                index = handleZ(value, result, index, slavoGermanic);
                break;
            default:
                index++;
                break;
        }
    }
    return alternate ? result.getAlternate() : result.getPrimary();
}
