public void printHelp(final PrintWriter pw, final int width, final String cmdLineSyntax, final String header, final Options options, final int leftPad, final int descPad, final String footer, final boolean autoUsage) {
    if (cmdLineSyntax == null) {
        throw new IllegalArgumentException("cmdLineSyntax is required");
    }
    if (cmdLineSyntax.isEmpty()) {
        throw new IllegalArgumentException("cmdLineSyntax is required");
    }
    if (options == null) {
        throw new NullPointerException("options is required");
    }
    if (width <= 0) {
        throw new IllegalStateException("width must be positive");
    }

    printWriter = pw;
    printWrapped(printWriter, width, defaultLeftPad, header);
    if (autoUsage) {
        printUsage(printWriter, width, cmdLineSyntax);
    }
    printOptions(printWriter, width, options, leftPad, descPad);
    printWrapped(printWriter, width, defaultLeftPad, footer);
}
