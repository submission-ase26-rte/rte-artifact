public static Option create(final String opt) throws IllegalArgumentException {
    if (opt == null || opt.isEmpty()) {
        throw new IllegalArgumentException("opt cannot be null or empty");
    }
    if (opt.length() > 1) {
        throw new IllegalArgumentException("opt cannot be more than one character");
    }
    if (!Character.isLetterOrDigit(opt.charAt(0))) {
        throw new IllegalArgumentException("opt must be a letter or digit");
    }
    Option option = new Option(opt, hasArg, description);
    option.setLongOpt(longOption);
    option.setRequired(required);
    option.setType(type);
    option.setValueSeparator(valueSeparator);
    option.setArgName(argName);
    reset();
    return option;
}
