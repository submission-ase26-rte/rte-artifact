import java.time.LocalDate;

import java.util.List;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.Mockito;
import org.mockito.junit.jupiter.MockitoExtension;
import org.springframework.samples.petclinic.model.Pet;
import org.springframework.samples.petclinic.model.Visit;
import org.springframework.util.Assert;

import static org.junit.jupiter.api.Assertions.assertThrows;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
public class Owner_addVisitTestGenerated {

    @Mock
    private Pet petMock;

    @Mock
    private Visit visitMock;

    @InjectMocks
    private Owner owner;

    @BeforeEach
    public void setUp() {
        List<Pet> pets = new ArrayList<>();
        pets.add(petMock);
        ReflectionTestUtils.setField(owner, "pets", pets);
    }

    @AfterEach
    public void tearDown() {
        reset(petMock, visitMock);
    }

    @Test
    public void testAddVisitWithValidPetIdAndVisit() {
        when(petMock.getId()).thenReturn(1);
        owner.addVisit(1, visitMock);
        verify(petMock, times(1)).addVisit(visitMock);
    }

    @Test
    public void testAddVisitWithNullPetId() {
        assertThrows(IllegalArgumentException.class, () -> owner.addVisit(null, visitMock));
    }

    @Test
    public void testAddVisitWithNullVisit() {
        assertThrows(IllegalArgumentException.class, () -> owner.addVisit(1, null));
    }

    @Test
    public void testAddVisitWithInvalidPetId() {
        when(petMock.getId()).thenReturn(2);
        assertThrows(IllegalArgumentException.class, () -> owner.addVisit(1, visitMock));
    }

    @Test
    public void testAddVisitWithEmptyPetsList() {
        ReflectionTestUtils.setField(owner, "pets", new ArrayList<>());
        assertThrows(IllegalArgumentException.class, () -> owner.addVisit(1, visitMock));
    }
