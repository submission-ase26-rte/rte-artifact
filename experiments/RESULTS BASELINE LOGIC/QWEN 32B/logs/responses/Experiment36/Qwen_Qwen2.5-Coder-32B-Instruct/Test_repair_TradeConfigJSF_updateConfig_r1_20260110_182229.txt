package com.ibm.websphere.samples.daytrader.web.jsf;

import com.ibm.websphere.samples.daytrader.util.TradeConfig;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.mockito.MockedStatic;
import org.springframework.test.util.ReflectionTestUtils;

import javax.faces.context.ExternalContext;
import javax.servlet.http.HttpSession;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.any;
import static org.mockito.ArgumentMatchers.anyString;
import static org.mockito.Mockito.*;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class TradeConfigJSF_updateConfigTestGenerated {

    @Mock
    private ExternalContext context;

    @Mock
    private HttpSession httpSession;

    @Mock
    private TradeDirectDBUtils dbUtils;

    @InjectMocks
    private TradeConfigJSF tradeConfigJSF;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
        ReflectionTestUtils.setField(tradeConfigJSF, "dbUtils", dbUtils);
        ReflectionTestUtils.setField(tradeConfigJSF, "context", context);
        tradeConfigJSF.setOrderProcessingModeList(new String[]{"SYNCH", "ASYNCH", "ASYNCH_2PHASE"});
        tradeConfigJSF.setWebInterfaceList(new String[]{"JSP", "JSP_Images", "JSP_Images_HTTP2"});
        tradeConfigJSF.setRuntimeModeList(new String[]{"EJB3", "DIRECT", "SESSION_TO_DIRECT"});
    }

    @AfterEach
    public void tearDown() {
        MockitoAnnotations.openMocks(this).close();
    }

    @Test
    public void testUpdateConfigWithValidInputs() {
        tradeConfigJSF.setRuntimeMode("DIRECT");
        tradeConfigJSF.setOrderProcessingMode("ASYNCH");
        tradeConfigJSF.setWebInterface("JSP_Images");
        tradeConfigJSF.setMaxUsers(100);
        ReflectionTestUtils.setField(tradeConfigJSF, "maxQuotes", 1000);
        tradeConfigJSF.setMarketSummaryInterval(30);
        tradeConfigJSF.setPrimIterations(1000);
        tradeConfigJSF.setPublishQuotePriceChange(true);
        tradeConfigJSF.setListQuotePriceChangeFrequency(20);
        tradeConfigJSF.setDisplayOrderAlerts(true);
        tradeConfigJSF.setLongRun(true);

        tradeConfigJSF.updateConfig();

        assertEquals("DIRECT", TradeConfig.getRunTimeModeNames()[TradeConfig.getRunTimeMode()]);
        assertEquals("ASYNCH", TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
        assertEquals("JSP_Images", TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
        assertEquals(100, TradeConfig.getMAX_USERS());
        assertEquals(1000, TradeConfig.getMAX_QUOTES());
        assertEquals(30, TradeConfig.getMarketSummaryInterval());
        assertEquals(1000, TradeConfig.getPrimIterations());
        assertTrue(TradeConfig.getPublishQuotePriceChange());
        assertEquals(20, TradeConfig.getListQuotePriceChangeFrequency());
        assertTrue(TradeConfig.getDisplayOrderAlerts());
        assertTrue(TradeConfig.getLongRun());
        assertEquals("DayTrader Configuration Updated", tradeConfigJSF.getResult());
    }

    @Test
    public void testUpdateConfigWithNullOrderProcessingMode() {
        tradeConfigJSF.setOrderProcessingMode(null);

        tradeConfigJSF.updateConfig();

        assertNotNull(TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
    }

    @Test
    public void testUpdateConfigWithInvalidOrderProcessingMode() {
        tradeConfigJSF.setOrderProcessingMode("INVALID_MODE");

        try (MockedStatic<Log> mockedLog = mockStatic(Log.class)) {
            tradeConfigJSF.updateConfig();
            mockedLog.verify(() -> Log.error(any(Exception.class), anyString(), anyString(), anyString()));
        }
    }

    @Test
    public void testUpdateConfigWithNullWebInterface() {
        tradeConfigJSF.setWebInterface(null);

        tradeConfigJSF.updateConfig();

        assertNotNull(TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
    }

    @Test
    public void testUpdateConfigWithInvalidWebInterface() {
        tradeConfigJSF.setWebInterface("INVALID_INTERFACE");

        try (MockedStatic<Log> mockedLog = mockStatic(Log.class)) {
            tradeConfigJSF.updateConfig();
            mockedLog.verify(() -> Log.error(any(Exception.class), anyString(), anyString(), anyString()));
        }
    }

    @Test
    public void testUpdateConfigWithDefaultValues() {
        tradeConfigJSF.updateConfig();

        assertNotNull(TradeConfig.getRunTimeModeNames()[TradeConfig.getRunTimeMode()]);
        assertNotNull(TradeConfig.getOrderProcessingModeNames()[TradeConfig.getOrderProcessingMode()]);
        assertNotNull(TradeConfig.getWebInterfaceNames()[TradeConfig.getWebInterface()]);
        assertTrue(TradeConfig.getMAX_USERS() > 0);
        assertTrue(TradeConfig.getMAX_QUOTES() > 0);
        assertTrue(TradeConfig.getMarketSummaryInterval() > 0);
        assertTrue(TradeConfig.getPrimIterations() > 0);
        assertNotNull(TradeConfig.getPublishQuotePriceChange());
        assertTrue(TradeConfig.getListQuotePriceChangeFrequency() >= 0);
        assertNotNull(TradeConfig.getDisplayOrderAlerts());
        assertNotNull(TradeConfig.getLongRun());
        assertEquals("DayTrader Configuration Updated", tradeConfigJSF.getResult());
    }
}
