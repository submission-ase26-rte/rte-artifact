public boolean canConvert(final Object object, final Class toType) {
    if (object == null) {
        return true;
    }
    if (toType.isAssignableFrom(object.getClass())) {
        return true;
    }
    if (toType == Boolean.class) {
        return object instanceof Number || object instanceof String;
    }
    if (Number.class.isAssignableFrom(toType)) {
        return object instanceof Boolean || object instanceof Number || object instanceof String;
    }
    if (toType == String.class) {
        return true;
    }
    if (toType.isArray()) {
        if (object.getClass().isArray()) {
            return toType.getComponentType().isAssignableFrom(object.getClass().getComponentType());
        }
        if (object instanceof Collection) {
            return true;
        }
    }
    if (Collection.class.isAssignableFrom(toType)) {
        return object.getClass().isArray() || object instanceof Collection;
    }
    if (object.getClass().isArray()) {
        int length = Array.getLength(object);
        if (length == 1) {
            return canConvert(Array.get(object, 0), toType);
        }
        return true; // Allow conversion from any array to a collection
    }
    Converter converter = ConvertUtils.lookup(toType);
    return converter != null;
}
