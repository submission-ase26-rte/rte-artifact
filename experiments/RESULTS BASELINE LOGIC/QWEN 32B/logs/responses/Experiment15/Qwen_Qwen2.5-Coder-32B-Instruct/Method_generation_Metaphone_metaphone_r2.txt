public String metaphone(final String txt) {
    if (txt == null || txt.isEmpty()) {
        return "";
    }

    StringBuilder inwd = new StringBuilder(txt.toLowerCase());
    StringBuilder metaph = new StringBuilder();

    // Handle initial characters
    if (inwd.length() > 1) {
        if (inwd.charAt(0) == 'k' && inwd.charAt(1) == 'n') {
            inwd.delete(0, 1);
        } else if (inwd.charAt(0) == 'x' && (inwd.charAt(1) == 'c' || inwd.charAt(1) == 'h')) {
            inwd.setCharAt(0, 's');
        } else if (inwd.charAt(0) == 'w' && inwd.charAt(1) == 'h') {
            inwd.setCharAt(0, 'h');
        } else if (inwd.charAt(0) == 'h' && !isVowel(inwd, 1)) {
            inwd.delete(0, 1);
        } else if (inwd.charAt(0) == 'g' && inwd.charAt(1) == 'n' && !isVowel(inwd, 2)) {
            inwd.delete(0, 2);
        } else if (inwd.charAt(0) == 'p' && inwd.charAt(1) == 'n' && !isVowel(inwd, 2)) {
            inwd.delete(0, 2);
        } else if (inwd.charAt(0) == 'a' && inwd.charAt(1) == 'e') {
            inwd.setCharAt(0, 'e');
        } else if (inwd.charAt(0) == 'g' && inwd.charAt(1) == 'y' && isVowel(inwd, 2)) {
            inwd.setCharAt(0, 'j');
        } else if (inwd.charAt(0) == 'x') {
            inwd.setCharAt(0, 's');
        }
    }

    int wdsz = inwd.length();
    int n = 0;

    while (metaph.length() < getMaxCodeLen() && n < wdsz) {
        char next;
        switch (inwd.charAt(n)) {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                if (n == 0) {
                    metaph.append(inwd.charAt(n));
                }
                break;
            case 'b':
                if (n != wdsz - 1 && inwd.charAt(n + 1) != 'b') {
                    metaph.append('b');
                }
                break;
            case 'c':
                if (regionMatch(inwd, n, "cia")) {
                    metaph.append('x');
                } else if (regionMatch(inwd, n, "ch")) {
                    metaph.append('x');
                } else if (regionMatch(inwd, n, "sch")) {
                    metaph.append('x');
                } else if (n > 0 && (inwd.charAt(n - 1) == 's' || inwd.charAt(n - 1) == 'z')) {
                    metaph.append('s');
                } else {
                    metaph.append('k');
                }
                break;
            case 'd':
                if (regionMatch(inwd, n, "dge") || regionMatch(inwd, n, "dgi") || regionMatch(inwd, n, "dgy")) {
                    metaph.append('j');
                } else if (regionMatch(inwd, n, "tch")) {
                    // do nothing
                } else {
                    metaph.append('t');
                }
                break;
            case 'g':
                if (regionMatch(inwd, n, "gh")) {
                    // do nothing
                } else if (regionMatch(inwd, n, "gn") && (n + 2 >= wdsz || !isVowel(inwd, n + 2))) {
                    // do nothing
                } else if (regionMatch(inwd, n, "gn") && (n + 2 < wdsz && isVowel(inwd, n + 2))) {
                    metaph.append('j');
                } else if (regionMatch(inwd, n, "gi") || regionMatch(inwd, n, "gy")) {
                    metaph.append('j');
                } else if (regionMatch(inwd, n, "g")) {
                    metaph.append('k');
                }
                break;
            case 'h':
                if ((n == 0 || isVowel(inwd, n - 1)) && isVowel(inwd, n + 1)) {
                    metaph.append('h');
                }
                break;
            case 'j':
                metaph.append('j');
                break;
            case 'k':
                if (n == 0 || inwd.charAt(n - 1) != 'c') {
                    metaph.append('k');
                }
                break;
            case 'l':
                metaph.append('l');
                break;
            case 'm':
                metaph.append('m');
                break;
            case 'n':
                metaph.append('n');
                break;
            case 'p':
                if (isNextChar(inwd, n, 'h')) {
                    metaph.append('f');
                } else {
                    metaph.append('p');
                }
                break;
            case 'q':
                metaph.append('k');
                break;
            case 'r':
                metaph.append('r');
                break;
            case 's':
                if (regionMatch(inwd, n, "sh") || regionMatch(inwd, n, "sio") || regionMatch(inwd, n, "sia")) {
                    metaph.append('x');
                } else {
                    metaph.append('s');
                }
                break;
            case 't':
                if (regionMatch(inwd, n, "tch")) {
                    // do nothing
                } else if (regionMatch(inwd, n, "th")) {
                    metaph.append('0');
                } else {
                    metaph.append('t');
                }
                break;
            case 'v':
                metaph.append('f');
                break;
            case 'w':
                if (n == 0 && isVowel(inwd, n + 1)) {
                    metaph.append('w');
                } else if (!isPreviousChar(inwd, n, 'h')) {
                    metaph.append('w');
                }
                break;
            case 'x':
                if (n == 0) {
                    metaph.append('s');
                } else {
                    metaph.append('ks');
                }
                break;
            case 'y':
                if (isVowel(inwd, n + 1)) {
                    metaph.append('y');
                }
                break;
            case 'z':
                metaph.append('s');
                break;
            default:
                break;
        }
        n++;
    }

    return metaph.toString();
}
