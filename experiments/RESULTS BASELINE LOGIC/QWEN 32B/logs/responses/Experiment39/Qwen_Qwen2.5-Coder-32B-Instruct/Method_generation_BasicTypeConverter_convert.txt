public Object convert(final Object object, final Class toType) {
    if (object == null) {
        if (toType.isPrimitive()) {
            throw new JXPathTypeConversionException("Cannot convert null to primitive type " + toType);
        }
        return null;
    }
    if (toType.isInstance(object)) {
        return object;
    }
    if (object instanceof Collection) {
        Collection<?> collection = (Collection<?>) object;
        if (toType.isArray()) {
            Object array = Array.newInstance(toType.getComponentType(), collection.size());
            int index = 0;
            for (Object element : collection) {
                Array.set(array, index++, convert(element, toType.getComponentType()));
            }
            return array;
        } else if (Collection.class.isAssignableFrom(toType)) {
            Collection<Object> result = allocateCollection(toType);
            for (Object element : collection) {
                result.add(convert(element, Object.class));
            }
            return unmodifiableCollection(result);
        } else if (collection.size() == 1) {
            return convert(collection.iterator().next(), toType);
        } else {
            return "";
        }
    } else if (object.getClass().isArray()) {
        int length = Array.getLength(object);
        if (toType.isArray()) {
            Object array = Array.newInstance(toType.getComponentType(), length);
            for (int i = 0; i < length; i++) {
                Array.set(array, i, convert(Array.get(object, i), toType.getComponentType()));
            }
            return array;
        } else if (Collection.class.isAssignableFrom(toType)) {
            Collection<Object> result = allocateCollection(toType);
            for (int i = 0; i < length; i++) {
                result.add(convert(Array.get(object, i), Object.class));
            }
            return unmodifiableCollection(result);
        } else if (length == 1) {
            return convert(Array.get(object, 0), toType);
        } else {
            return "";
        }
    } else if (toType == String.class) {
        return object.toString();
    } else if (object instanceof Boolean && Number.class.isAssignableFrom(toType)) {
        return ((Boolean) object) ? Integer.valueOf(1) : Integer.valueOf(0);
    } else if (object instanceof Number && toType == Boolean.class) {
        return ((Number) object).intValue() != 0 ? Boolean.TRUE : Boolean.FALSE;
    } else if (object instanceof String) {
        return convertStringToPrimitive(object, toType);
    } else {
        Converter converter = ConvertUtils.lookup(toType);
        if (converter != null) {
            return converter.convert(toType, object);
        }
        throw new JXPathTypeConversionException("Cannot convert " + object.getClass() + " to " + toType);
    }
}
