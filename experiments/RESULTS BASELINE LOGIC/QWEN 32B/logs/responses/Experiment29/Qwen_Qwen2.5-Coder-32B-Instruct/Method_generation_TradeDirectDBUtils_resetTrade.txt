public RunStatsDataBean resetTrade(boolean deleteAll) throws Exception {
    RunStatsDataBean runStatsDataBean = new RunStatsDataBean();
    Connection conn = null;
    try {
        conn = datasource.getConnection();
        conn.setAutoCommit(false);
        if (deleteAll) {
            String[] deleteSQLs = {
                "DELETE FROM holdingejb",
                "DELETE FROM orderejb",
                "DELETE FROM quoteejb",
                "DELETE FROM accountejb",
                "DELETE FROM accountprofileejb"
            };
            for (String sql : deleteSQLs) {
                PreparedStatement stmt = getStatement(conn, sql);
                stmt.executeUpdate();
            }
            runStatsDataBean.setNewUserCount(0);
        } else {
            String[] selectSQLs = {
                "SELECT count(*) FROM accountejb",
                "SELECT count(*) FROM accountprofileejb",
                "SELECT count(*) FROM quoteejb",
                "SELECT count(*) FROM holdingejb",
                "SELECT count(*) FROM orderejb",
                "SELECT sum(openbalance) FROM accountejb",
                "SELECT sum(balance) FROM accountejb"
            };
            String[] updateSQLs = {
                "UPDATE accountejb SET balance = openbalance",
                "UPDATE orderejb SET orderstatus = 'closed'"
            };
            int index = 0;
            for (String sql : selectSQLs) {
                PreparedStatement stmt = getStatement(conn, sql);
                ResultSet rs = stmt.executeQuery();
                if (rs.next()) {
                    switch (index) {
                        case 0:
                            runStatsDataBean.setTradeUserCount(rs.getInt(1));
                            break;
                        case 1:
                            runStatsDataBean.setNewUserCount(rs.getInt(1));
                            break;
                        case 2:
                            runStatsDataBean.setTradeStockCount(rs.getInt(1));
                            break;
                        case 3:
                            runStatsDataBean.setHoldingCount(rs.getInt(1));
                            break;
                        case 4:
                            runStatsDataBean.setOrderCount(rs.getInt(1));
                            break;
                        case 5:
                            runStatsDataBean.setSumLoginCount(rs.getInt(1));
                            break;
                        case 6:
                            runStatsDataBean.setSumLogoutCount(rs.getInt(1));
                            break;
                    }
                }
                index++;
            }
            for (String sql : updateSQLs) {
                PreparedStatement stmt = getStatement(conn, sql);
                stmt.executeUpdate();
            }
        }
        conn.commit();
    } catch (SQLException e) {
        if (conn != null) {
            conn.rollback();
        }
        Log.error(e, "TradeDirectDBUtils:resetTrade -- failed");
        throw e;
    } finally {
        if (conn != null) {
            conn.close();
        }
    }
    return runStatsDataBean;
}
