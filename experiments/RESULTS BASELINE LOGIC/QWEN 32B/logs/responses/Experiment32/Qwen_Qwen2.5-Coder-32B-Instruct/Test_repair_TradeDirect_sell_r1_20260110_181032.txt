package com.ibm.websphere.samples.daytrader.impl.direct;

import com.ibm.websphere.samples.daytrader.beans.MarketSummaryDataBean;
import com.ibm.websphere.samples.daytrader.beans.OrderDataBean;
import com.ibm.websphere.samples.daytrader.beans.QuoteDataBean;
import com.ibm.websphere.samples.daytrader.beans.AccountDataBean;
import com.ibm.websphere.samples.daytrader.beans.HoldingDataBean;
import com.ibm.websphere.samples.daytrader.beans.AccountProfileDataBean;
import com.ibm.websphere.samples.daytrader.util.TradeConfig;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.springframework.test.util.ReflectionTestUtils;

import javax.jms.JMSException;
import javax.sql.DataSource;
import javax.transaction.UserTransaction;
import java.math.BigDecimal;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.*;

import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class TradeDirect_sellTestGenerated {

    @Mock
    private DataSource datasource;

    @Mock
    private UserTransaction txn;

    @Mock
    private Connection conn;

    @Mock
    private PreparedStatement ps;

    @Mock
    private ResultSet rs;

    @Mock
    private AccountDataBean accountData;

    @Mock
    private HoldingDataBean holdingData;

    @Mock
    private QuoteDataBean quoteData;

    @Mock
    private OrderDataBean orderData;

    @InjectMocks
    private TradeDirect tradeDirect;

    @BeforeEach
    public void setUp() throws Exception {
        MockitoAnnotations.openMocks(this);
        ReflectionTestUtils.setField(tradeDirect, "inSession", false);
        when(datasource.getConnection()).thenReturn(conn);
        when(conn.prepareStatement(anyString())).thenReturn(ps);
        when(ps.executeQuery()).thenReturn(rs);
        when(rs.next()).thenReturn(true);
        when(rs.getDouble(anyString())).thenReturn(1.0);
        when(rs.getBigDecimal(anyString())).thenReturn(BigDecimal.ONE);
        when(rs.getInt(anyString())).thenReturn(1);
        when(rs.getString(anyString())).thenReturn("test");
        when(rs.getObject(anyString())).thenReturn(BigDecimal.ONE);
        when(accountData.getAccountID()).thenReturn(1);
        when(holdingData.getQuoteID()).thenReturn("test");
        when(holdingData.getQuantity()).thenReturn(1.0);
        when(quoteData.getPrice()).thenReturn(BigDecimal.ONE);
        when(quoteData.getOrderFee()).thenReturn(BigDecimal.ONE);
        when(tradeDirect.createOrder(any(), any(), any(), anyString(), anyDouble())).thenReturn(orderData);
        when(tradeDirect.getOrderData(conn, 1)).thenReturn(orderData);
    }

    @Test
    public void testSell_SynchronousMode() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.SYNCH);
        assertNotNull(result);
        verify(tradeDirect).completeOrder(conn, 1);
    }

    @Test
    public void testSell_AsynchronousMode() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.ASYNCH);
        assertNotNull(result);
        verify(tradeDirect).completeOrderAsync(1, true);
    }

    @Test
    public void testSell_Asynchronous2PhaseMode() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        when(txn.begin()).thenReturn(null);
        when(txn.commit()).thenReturn(null);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.ASYNCH_2PHASE);
        assertNotNull(result);
        verify(tradeDirect).queueOrder(1, true);
        verify(txn).begin();
        verify(txn).commit();
    }

    @Test
    public void testSell_AccountNotFound() throws Exception {
        when(tradeDirect.getAccountData(conn, "testUser")).thenReturn(null);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.SYNCH);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
    }

    @Test
    public void testSell_HoldingNotFound() throws Exception {
        when(tradeDirect.getHoldingData(conn, 1)).thenReturn(null);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.SYNCH);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
    }

    @Test
    public void testSell_QuoteNotFound() throws Exception {
        when(tradeDirect.getQuoteData(conn, "test")).thenReturn(null);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.SYNCH);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
    }

    @Test
    public void testSell_ExceptionDuringCompleteOrder() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        doThrow(new SQLException()).when(tradeDirect).completeOrder(conn, 1);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.SYNCH);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
    }

    @Test
    public void testSell_ExceptionDuringQueueOrder() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        doThrow(new JMSException("test")).when(tradeDirect).queueOrder(1, true);
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.ASYNCH_2PHASE);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
    }

    @Test
    public void testSell_ExceptionDuringTransaction() throws Exception {
        ReflectionTestUtils.setField(tradeDirect, "inGlobalTxn", false);
        when(txn.begin()).thenReturn(null);
        doThrow(new SQLException()).when(tradeDirect).createOrder(any(), any(), any(), anyString(), anyDouble());
        OrderDataBean result = tradeDirect.sell("testUser", 1, TradeConfig.ASYNCH_2PHASE);
        assertNotNull(result);
        assertEquals("cancelled", result.getOrderStatus());
        verify(txn).rollback();
    }
}
