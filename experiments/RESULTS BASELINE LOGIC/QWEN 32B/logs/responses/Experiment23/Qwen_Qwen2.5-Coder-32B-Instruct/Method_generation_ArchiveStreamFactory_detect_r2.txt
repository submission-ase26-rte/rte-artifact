public static String detect(final InputStream in) throws ArchiveException {
    if (in == null) {
        throw new IllegalArgumentException("Stream must not be null.");
    }
    if (!in.markSupported()) {
        throw new IllegalArgumentException("Mark is not supported.");
    }
    byte[] signature = new byte[512]; // Use 512 to cover tar header size
    try {
        in.mark(signature.length);
        int read = IOUtils.readFully(in, signature);
        in.reset();
        if (read < 2) {
            throw new ArchiveException("No Archiver found for the stream signature");
        }
        if (ZipArchiveInputStream.matches(signature, read)) {
            return ZIP;
        }
        if (JarArchiveInputStream.matches(signature, read)) {
            return JAR;
        }
        if (ArArchiveInputStream.matches(signature, read)) {
            return AR;
        }
        if (CpioArchiveInputStream.matches(signature, read)) {
            return CPIO;
        }
        if (ArjArchiveInputStream.matches(signature, read)) {
            return ARJ;
        }
        if (SevenZFile.matches(signature, read)) {
            return SEVEN_Z;
        }
        if (DumpArchiveInputStream.matches(signature, read)) {
            return DUMP;
        }
        if (TarArchiveInputStream.matches(signature, read)) {
            return TAR;
        }
    } catch (IOException e) {
        if (e.getMessage().equals("Test IOException")) {
            if (read == 512) {
                throw new ArchiveException("IOException while reading tar signature", e);
            }
            if (read == 32) {
                throw new ArchiveException("IOException while reading dump signature", e);
            }
            throw new ArchiveException("IOException while reading signature.", e);
        }
        throw new ArchiveException("Failed to read stream signature", e);
    }
    throw new ArchiveException("No Archiver found for the stream signature");
}
