package org.apache.commons.compress.archivers;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;
import org.apache.commons.compress.archivers.ArchiveException;
import org.apache.commons.compress.archivers.ArchiveStreamFactory;
import org.apache.commons.compress.archivers.jar.JarArchiveEntry;
import org.apache.commons.compress.archivers.jar.JarArchiveOutputStream;
import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
import org.apache.commons.compress.archivers.tar.TarArchiveOutputStream;
import org.apache.commons.compress.utils.IOUtils;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;

import static org.junit.jupiter.api.Assertions.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
public class ArchiveStreamFactory_detectTestGenerated {
    private ArchiveStreamFactory factory;
    private ByteArrayOutputStream byteArrayOutputStream;

    @BeforeEach
    public void setUp() {
        factory = new ArchiveStreamFactory();
        byteArrayOutputStream = new ByteArrayOutputStream();
    }

    @AfterEach
    public void tearDown() throws IOException {
        byteArrayOutputStream.close();
    }

    private void createZipArchive() throws IOException {
        ZipOutputStream zipOutputStream = new ZipOutputStream(byteArrayOutputStream);
        ZipEntry zipEntry = new ZipEntry("test.txt");
        zipOutputStream.putNextEntry(zipEntry);
        zipOutputStream.write("Hello, World!".getBytes(StandardCharsets.UTF_8));
        zipOutputStream.closeEntry();
        zipOutputStream.close();
    }

    private void createJarArchive() throws IOException {
        JarArchiveOutputStream jarOutputStream = new JarArchiveOutputStream(byteArrayOutputStream);
        JarArchiveEntry jarEntry = new JarArchiveEntry("test.txt");
        jarOutputStream.putArchiveEntry(jarEntry);
        jarOutputStream.write("Hello, World!".getBytes(StandardCharsets.UTF_8));
        jarOutputStream.closeArchiveEntry();
        jarOutputStream.close();
    }

    private void createTarArchive() throws IOException {
        TarArchiveOutputStream tarOutputStream = new TarArchiveOutputStream(byteArrayOutputStream);
        TarArchiveEntry entry = new TarArchiveEntry("test.txt");
        entry.setSize("Hello, World!".getBytes(StandardCharsets.UTF_8).length);
        tarOutputStream.putArchiveEntry(entry);
        tarOutputStream.write("Hello, World!".getBytes(StandardCharsets.UTF_8));
        tarOutputStream.closeArchiveEntry();
        tarOutputStream.close();
    }

    @Test
    public void testDetectJar() throws IOException, ArchiveException {
        createJarArchive();
        ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(byteArrayOutputStream.toByteArray());
        assertEquals(ArchiveStreamFactory.JAR, factory.detect(byteArrayInputStream));
    }

    @Test
    public void testDetectInvalidSignature() throws IOException {
        byte[] invalidSignature = new byte[10];
        ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(invalidSignature);
        Exception exception = assertThrows(ArchiveException.class, () -> factory.detect(byteArrayInputStream));
        assertEquals("No Archiver found for the stream signature", exception.getMessage());
    }

    @Test
    public void testDetectNullInputStream() {
        Exception exception = assertThrows(IllegalArgumentException.class, () -> factory.detect(null));
        assertEquals("Stream must not be null.", exception.getMessage());
    }

    @Test
    public void testDetectNonMarkSupportedInputStream() throws IOException {
        InputStream nonMarkSupportedInputStream = new ByteArrayInputStream(new byte[0]) {
            @Override
            public boolean markSupported() {
                return false;
            }
        };
        Exception exception = assertThrows(IllegalArgumentException.class, () -> factory.detect(nonMarkSupportedInputStream));
        assertEquals("Mark is not supported.", exception.getMessage());
    }

    @Test
    public void testDetectIOExceptionWhileReadingSignature() throws IOException {
        InputStream inputStream = new ByteArrayInputStream(new byte[0]) {
            @Override
            public int read(byte[] b) throws IOException {
                throw new IOException("Test IOException");
            }
        };
        Exception exception = assertThrows(ArchiveException.class, () -> factory.detect(inputStream));
        assertEquals("IOException while reading signature.", exception.getMessage());
    }

    @Test
    public void testDetectIOExceptionWhileReadingTarSignature() throws IOException {
        InputStream inputStream = new ByteArrayInputStream(new byte[0]) {
            @Override
            public int read(byte[] b) throws IOException {
                if (b.length == 512) { // Tar header size
                    throw new IOException("Test IOException");
                }
                return super.read(b);
            }
        };
        Exception exception = assertThrows(ArchiveException.class, () -> factory.detect(inputStream));
        assertEquals("IOException while reading tar signature", exception.getMessage());
    }
}
