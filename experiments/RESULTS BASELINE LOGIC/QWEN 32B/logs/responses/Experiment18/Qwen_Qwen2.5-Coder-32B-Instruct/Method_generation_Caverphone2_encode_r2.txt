public String encode(final String source) {
    if (SoundexUtils.isEmpty(source)) {
        return "1111111111";
    }
    String txt = source.toLowerCase().replaceAll("[^a-z]", "");
    txt = txt.replaceAll("^cough$", "kof");
    txt = txt.replaceAll("^rough$", "rof");
    txt = txt.replaceAll("^tough$", "tof");
    txt = txt.replaceAll("^enough$", "enof");
    txt = txt.replaceAll("^gn", "n");
    txt = txt.replaceAll("^ph", "fh");
    txt = txt.replaceAll("b", "p");
    txt = txt.replaceAll("c|q|x", "k");
    txt = txt.replaceAll("v", "f");
    txt = txt.replaceAll("dg", "g");
    txt = txt.replaceAll("tio|tia", "sio");
    txt = txt.replaceAll("d", "t");
    txt = txt.replaceAll("z", "s");
    txt = txt.replaceAll("^[aeiou]", "a");
    txt = txt.replaceAll("[aeiou]", "");
    txt = txt.replaceAll("j", "y");
    txt = txt.replaceAll("^y", "a");
    txt = txt.replaceAll("y", "3");
    txt = txt.replaceAll("3gh3", "3kh3");
    txt = txt.replaceAll("gh", "22");
    txt = txt.replaceAll("g", "k");
    txt = txt.replaceAll("s", "5");
    txt = txt.replaceAll("t", "7");
    txt = txt.replaceAll("p", "b");
    txt = txt.replaceAll("k$", "");
    txt = txt.replaceAll("f$", "");
    txt = txt.replaceAll("m$", "n");
    txt = txt.replaceAll("n$", "");
    txt = txt.replaceAll("w3", "w3");
    txt = txt.replaceAll("wh3", "w3h3");
    txt = txt.replaceAll("w$", "3");
    txt = txt.replaceAll("r3", "r3");
    txt = txt.replaceAll("r$", "");
    txt = txt.replaceAll("l3", "l3");
    txt = txt.replaceAll("l$", "");
    txt = txt.length() > 10 ? txt.substring(0, 10) : txt;
    while (txt.length() < 10) {
        txt += "1";
    }
    return txt.toUpperCase();
}
