import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class EventItemWriter_writeItemsTestGenerated {
    @Mock
    private JobContext jobContext;
    @Mock
    private ApplicationEvents applicationEvents;
    @Mock
    private HandlingEventRegistrationAttempt attempt;
    private EventItemWriter eventItemWriter;
    private AutoCloseable closeable;

    @BeforeEach
    void setUp() throws Exception {
        closeable = MockitoAnnotations.openMocks(this);
        eventItemWriter = new EventItemWriter();
        ReflectionTestUtils.setField(eventItemWriter, "jobContext", jobContext);
        ReflectionTestUtils.setField(eventItemWriter, "applicationEvents", applicationEvents);
    }

    @AfterEach
    void tearDown() throws Exception {
        closeable.close();
    }

    @Test
    void testWriteItemsWithEmptyList() throws Exception {
        // Setup
        List<Object> items = Collections.emptyList();
        Properties properties = new Properties();
        String tempDir = System.getProperty("java.io.tmpdir");
        properties.setProperty("archive_directory", tempDir);
        when(jobContext.getProperties()).thenReturn(properties);
        when(jobContext.getJobName()).thenReturn("testJob");
        when(jobContext.getInstanceId()).thenReturn(123L);
        // Execute
        eventItemWriter.writeItems(items);
        // Verify
        verify(applicationEvents, never()).receivedHandlingEventRegistrationAttempt(any());
        Path filePath = Path.of(tempDir, "archive_testJob_123.csv");
        assertFalse(Files.exists(filePath));
    }

    @Test
    void testWriteItemsWithNonEmptyList() throws Exception {
        // Setup
        List<Object> items = Collections.singletonList(attempt);
        LocalDateTime registrationTime = LocalDateTime.now(ZoneId.systemDefault());
        LocalDateTime completionTime = registrationTime.plusHours(1);
        Properties properties = new Properties();
        String tempDir = System.getProperty("java.io.tmpdir");
        properties.setProperty("archive_directory", tempDir);
        when(jobContext.getProperties()).thenReturn(properties);
        when(jobContext.getJobName()).thenReturn("testJob");
        when(jobContext.getInstanceId()).thenReturn(123L);
        when(attempt.getRegistrationTime()).thenReturn(registrationTime);
        when(attempt.getCompletionTime()).thenReturn(completionTime);
        when(attempt.getTrackingId()).thenReturn(new TrackingId("tracking123"));
        when(attempt.getVoyageNumber()).thenReturn(new VoyageNumber("voyage456"));
        when(attempt.getUnLocode()).thenReturn(new UnLocode("ABC123"));
        when(attempt.getType()).thenReturn(HandlingEvent.Type.valueOf("UNKNOWN"));
        // Execute
        eventItemWriter.writeItems(items);
        // Verify
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(attempt);
        Path filePath = Path.of(tempDir, "archive_testJob_123.csv");
        assertTrue(Files.exists(filePath));
        assertTrue(Files.readString(filePath).contains("tracking123"));
    }

    @Test
    void testWriteItemsWithNullList() throws Exception {
        // Setup
        List<Object> items = null;
        Properties properties = new Properties();
        String tempDir = System.getProperty("java.io.tmpdir");
        properties.setProperty("archive_directory", tempDir);
        when(jobContext.getProperties()).thenReturn(properties);
        when(jobContext.getJobName()).thenReturn("testJob");
        when(jobContext.getInstanceId()).thenReturn(123L);
        // Execute & Verify
        assertDoesNotThrow(() -> eventItemWriter.writeItems(items));
        Path filePath = Path.of(tempDir, "archive_testJob_123.csv");
        assertFalse(Files.exists(filePath));
    }

    @Test
    void testWriteItemsWithNullAttemptInList() throws Exception {
        // Setup
        List<Object> items = Collections.singletonList(null);
        Properties properties = new Properties();
        String tempDir = System.getProperty("java.io.tmpdir");
        properties.setProperty("archive_directory", tempDir);
        when(jobContext.getProperties()).thenReturn(properties);
        when(jobContext.getJobName()).thenReturn("testJob");
        when(jobContext.getInstanceId()).thenReturn(123L);
        // Execute & Verify
        assertDoesNotThrow(() -> eventItemWriter.writeItems(items));
        Path filePath = Path.of(tempDir, "archive_testJob_123.csv");
        assertFalse(Files.exists(filePath));
    }

}
