        // Setup
        Cargo cargo = mock(Cargo.class);

        when(cargo.getItinerary()).thenReturn(itinerary);

        when(cargo.getTrackingId()).thenReturn(new TrackingId("TRACK123"));
        List<Cargo> cargos = new ArrayList<>();

        when(cargoRepository.findAll()).thenReturn(cargos);
        Location location = new Location(new UnLocode("USNYC"), "New York");
        List<Location> locations = new ArrayList<>();
        locations.add(location);
        when(locationRepository.findAll()).thenReturn(locations);
        Voyage voyage = new Voyage(new VoyageNumber("VA001"), null);
        List<Voyage> voyages = new ArrayList<>();
        voyages.add(voyage);
        when(voyageRepository.findAll()).thenReturn(voyages);
        // Execute
        eventLogger.init();
        // Verify
        assertEquals(1, eventLogger.getTrackingIds().size());
        assertEquals("TRACK123", eventLogger.getTrackingIds().get(0).getLabel());
        assertEquals(1, eventLogger.getLocations().size());

        assertEquals("VA001", eventLogger.getVoyages().get(0).getLabel());
