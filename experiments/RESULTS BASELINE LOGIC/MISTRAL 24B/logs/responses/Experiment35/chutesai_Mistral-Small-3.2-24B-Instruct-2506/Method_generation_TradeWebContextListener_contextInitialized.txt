public void contextInitialized(ServletContextEvent event) {
    ServletContext servletContext = event.getServletContext();
    InputStream inputStream = servletContext.getResourceAsStream("/properties/daytrader.properties");
    Properties properties = new Properties();

    try {
        if (inputStream != null) {
            properties.load(inputStream);
        }

        String envRuntimeMode = System.getenv("RUNTIME_MODE");
        String envOrderProcessingMode = System.getenv("ORDER_PROCESSING_MODE");
        String envMaxUsers = System.getenv("MAX_USERS");
        String envMaxQuotes = System.getenv("MAX_QUOTES");
        String envPublishQuotes = System.getenv("PUBLISH_QUOTES");
        String envDisplayOrderAlerts = System.getenv("DISPLAY_ORDER_ALERTS");
        String envWebInterface = System.getenv("WEB_INTERFACE");
        String envListQuotePriceChangeFrequency = System.getenv("LIST_QUOTE_PRICE_CHANGE_FREQUENCY");

        int runtimeMode = properties.getProperty("runtimeMode") != null ? Integer.parseInt(properties.getProperty("runtimeMode")) :
                         (envRuntimeMode != null ? Integer.parseInt(envRuntimeMode) : 1);
        int orderProcessingMode = properties.getProperty("orderProcessingMode") != null ? Integer.parseInt(properties.getProperty("orderProcessingMode")) :
                                (envOrderProcessingMode != null ? Integer.parseInt(envOrderProcessingMode) : 1);
        int maxUsers = properties.getProperty("maxUsers") != null ? Integer.parseInt(properties.getProperty("maxUsers")) :
                     (envMaxUsers != null ? Integer.parseInt(envMaxUsers) : 100);
        int maxQuotes = properties.getProperty("maxQuotes") != null ? Integer.parseInt(properties.getProperty("maxQuotes")) :
                      (envMaxQuotes != null ? Integer.parseInt(envMaxQuotes) : 1000);
        boolean publishQuotePriceChange = properties.getProperty("publishQuotePriceChange") != null ? Boolean.parseBoolean(properties.getProperty("publishQuotePriceChange")) :
                                        (envPublishQuotes != null ? Boolean.parseBoolean(envPublishQuotes) : true);
        boolean displayOrderAlerts = properties.getProperty("displayOrderAlerts") != null ? Boolean.parseBoolean(properties.getProperty("displayOrderAlerts")) :
                                  (envDisplayOrderAlerts != null ? Boolean.parseBoolean(envDisplayOrderAlerts) : true);
        int webInterface = properties.getProperty("webInterface") != null ? Integer.parseInt(properties.getProperty("webInterface")) :
                         (envWebInterface != null ? Integer.parseInt(envWebInterface) : 1);
        int listQuotePriceChangeFrequency = properties.getProperty("listQuotePriceChangeFrequency") != null ? Integer.parseInt(properties.getProperty("listQuotePriceChangeFrequency")) :
                                          (envListQuotePriceChangeFrequency != null ? Integer.parseInt(envListQuotePriceChangeFrequency) : 60);
        int primIterations = properties.getProperty("primIterations") != null ? Integer.parseInt(properties.getProperty("primIterations")) : 1000;
        int marketSummaryInterval = properties.getProperty("marketSummaryInterval") != null ? Integer.parseInt(properties.getProperty("marketSummaryInterval")) : 300;
        boolean longRun = properties.getProperty("longRun") != null ? Boolean.parseBoolean(properties.getProperty("longRun")) : false;

        TradeConfig.setRunTimeMode(runtimeMode);
        TradeConfig.setOrderProcessingMode(orderProcessingMode);
        TradeConfig.setMAX_USERS(maxUsers);
        TradeConfig.setMAX_QUOTES(maxQuotes);
        TradeConfig.setPublishQuotePriceChange(publishQuotePriceChange);
        TradeConfig.setDisplayOrderAlerts(displayOrderAlerts);
        TradeConfig.setWebInterface(webInterface);
        TradeConfig.setListQuotePriceChangeFrequency(listQuotePriceChangeFrequency);
        TradeConfig.setPrimIterations(primIterations);
        TradeConfig.setMarketSummaryInterval(marketSummaryInterval);
        TradeConfig.setLongRun(longRun);
    } catch (Exception e) {
        Log.trace("TradeWebContextListener contextInitialized -- initializing TradeDirect");
    }

    Log.trace("TradeWebContextListener contextInitialized -- initializing TradeDirect");
}
