package org.apache.commons.cli;

import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.BeforeEach;
import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class Option_toStringTestGenerated {

    private Option option;

    @BeforeEach
    void setUp() {
        this.option = new Option("o");
    }

    @Test
    void testToString_BasicOption() {
        option = Option.builder("o").desc("test description").build();
        String result = option.toString();
        assertTrue(result.contains("Option o"));
        assertTrue(result.contains(":: test description"));
        assertTrue(result.contains(":: null ]"));
    }

    @Test
    void testToString_WithLongOption() {
        option = Option.builder("o").longOpt("longOpt").desc("test description").build();
        String result = option.toString();
        assertTrue(result.contains("Option o"));
        assertTrue(result.contains("longOpt"));
        assertTrue(result.contains(":: test description"));
    }

    @Test
    void testToString_WithArg() {
        option = Option.builder("o").argName("test description").hasArg(true).build();
        String result = option.toString();
        assertTrue(result.contains("Option o [ARG]"));
    }

    @Test
    void testToString_WithArgs() {
        option = Option.builder("o").numberOfArgs(2).build();
        String result = option.toString();
        assertTrue(result.contains("Option o [ARG...]"));
    }

    @Test
    void testToString_WithCustomType() {
        option = Option.builder("o").type(String.class).desc("test description").build();
        String result = option.toString();
        assertTrue(result.contains("class java.lang.String ]"));
    }

    @Test
    void testToString_WithNullDescription() {
        option = Option.builder("o").desc(null).build();
        String result = option.toString();
        assertTrue(result.contains(":: null"));
    }

    @Test
    void testToString_WithEmptyDescription() {
        option = Option.builder("o").desc("").build();
        String result = option.toString();
        assertTrue(result.contains(":: ]"));
    }

    @Test
    void testToString_WithNullType() {
        option = Option.builder("o").type(null).desc("test description").build();
        String result = option.toString();
        assertTrue(result.contains(":: null ]"));
    }
}
