public CommandLine parse(final Options options, final String[] arguments, final Properties properties, final boolean stopAtNonOption) throws ParseException {
    if (options == null) {
        throw new NullPointerException();
    }

    this.options = options;
    this.cmd = new CommandLine();
    this.requiredOptions = new ArrayList();

    if (arguments != null) {
        List<String> flattened = Arrays.asList(flatten(options, arguments, stopAtNonOption));
        ListIterator<String> iter = flattened.listIterator();
        while (iter.hasNext()) {
            String arg = iter.next();
            processOption(arg, iter);
            if (stopAtNonOption && !arg.startsWith("-")) {
                break;
            }
        }
    }

    if (properties != null) {
        processProperties(properties);
    }

    checkRequiredOptions();

    for (Option opt : options.helpOptions()) {
        opt.clearValues();
    }

    for (OptionGroup group : options.getOptionGroups()) {
        group.setSelected(null);
    }

    return cmd;
}
