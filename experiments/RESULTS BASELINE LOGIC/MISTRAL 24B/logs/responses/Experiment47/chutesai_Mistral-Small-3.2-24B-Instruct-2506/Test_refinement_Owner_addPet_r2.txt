package org.springframework.samples.petclinic.owner;

import java.util.ArrayList;
import java.util.List;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.Timeout;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

import static org.junit.jupiter.api.Assertions.assertThrows;
import static org.mockito.Mockito.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class Owner_addPetTestGenerated {
    @Mock
    private List<Pet> pets;
    @InjectMocks
    private Owner owner;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.openMocks(this);
        when(owner.getPets()).thenReturn(pets);
    }

    @Test
    void addPet_WhenPetIsNew_ShouldAddToPets() {
        // Arrange
        Pet newPet = mock(Pet.class);
        when(newPet.isNew()).thenReturn(true);
        // Act
        owner.addPet(newPet);
        // Assert
        verify(pets).add(newPet);
    }

    @Test
    void addPet_WhenPetIsNotNew_ShouldNotAddToPets() {
        // Arrange
        Pet existingPet = mock(Pet.class);
        when(existingPet.isNew()).thenReturn(false);
        // Act
        owner.addPet(existingPet);
        // Assert
        verify(pets, never()).add(existingPet);
    }

    @Test
    void addPet_WhenPetIsNull_ShouldThrowNullPointerException() {
        // Act & Assert
        assertThrows(NullPointerException.class, () -> owner.addPet(null));
    }

    @Test
    void addPet_WhenPetsIsNull_ShouldNotThrowException() {
        // Arrange
        when(owner.getPets()).thenReturn(null);
        Pet newPet = mock(Pet.class);
        when(newPet.isNew()).thenReturn(true);
        // Act
        owner.addPet(newPet);
        // Assert
        // No exception should be thrown and no operation should be performed
        verifyNoInteractions(newPet);
    }

    @Test
    void addPet_WhenPetIsNewAndPetsIsNull_ShouldNotThrowException() {
        // Arrange
        when(owner.getPets()).thenReturn(null);
        Pet newPet = mock(Pet.class);
        when(newPet.isNew()).thenReturn(true);
        // Act
        owner.addPet(newPet);
        // Assert
        verifyNoInteractions(pets);
    }

    @Test
    void addPet_WhenPetIsNewAndPetsIsEmpty_ShouldAddToPets() {
        // Arrange
        Pet newPet = mock(Pet.class);
        when(newPet.isNew()).thenReturn(true);
        when(pets).thenReturn(new ArrayList<>());
        // Act
        owner.addPet(newPet);
        // Assert
        verify(pets).add(newPet);
    }
}
