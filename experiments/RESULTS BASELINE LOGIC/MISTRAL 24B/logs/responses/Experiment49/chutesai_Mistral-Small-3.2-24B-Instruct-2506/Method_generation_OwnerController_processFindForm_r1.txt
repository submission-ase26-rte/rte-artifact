@PostMapping("/owners")
public String processFindForm(@RequestParam(defaultValue = "1") int page, Owner owner, BindingResult result, Model model) {
    if (result.hasErrors()) {
        return "owners/ownersList";
    }

    String lastName = owner.getLastName();

    if (lastName == null || lastName.isEmpty()) {
        return addPaginationModel(page, model, findPaginatedForOwnersLastName(page, ""));
    }

    Page<Owner> ownersPage = findPaginatedForOwnersLastName(page, lastName);

    if (ownersPage.getTotalElements() == 0) {
        result.rejectValue("lastName", "notFound", "not found");
        return "owners/findOwners";
    } else if (ownersPage.getTotalElements() == 1) {
        return "redirect:/owners/" + ownersPage.getContent().get(0).getId();
    } else {
        return addPaginationModel(page, model, ownersPage);
    }
}
