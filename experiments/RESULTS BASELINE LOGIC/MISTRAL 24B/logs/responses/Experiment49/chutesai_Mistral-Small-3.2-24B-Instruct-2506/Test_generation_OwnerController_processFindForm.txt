package org.springframework.samples.petclinic.owner;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageImpl;
import org.springframework.data.domain.PageRequest;
import org.springframework.ui.ExtendedModel;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.MapBindingResult;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class OwnerController_processFindFormTestGenerated {

    @Mock
    private Model model;

    @Mock
    private BindingResult bindingResult;

    @InjectMocks
    private OwnerController ownerController;

    private Owner owner;
    private Page<Owner> ownersPage;

    @BeforeEach
    void setUp() {
        owner = new Owner();
        ownersPage = new PageImpl<>(new ArrayList<>(), PageRequest.of(0, 10), 0);
    }

    @Test
    void testProcessFindForm_EmptyLastName_ReturnsAllRecords() {
        when(ownerController.findPaginatedForOwnersLastName(1, "")).thenReturn(ownersPage);

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("owners/ownersList", viewName);
        verify(ownerController).findPaginatedForOwnersLastName(1, "");
    }

    @Test
    void testProcessFindForm_NoOwnersFound_ReturnsFindOwnersWithError() {
        when(ownerController.findPaginatedForOwnersLastName(1, "Doe")).thenReturn(new PageImpl<>(Collections.emptyList()));

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("owners/findOwners", viewName);
        verify(bindingResult).rejectValue("lastName", "notFound", "not found");
    }

    @Test
    void testProcessFindForm_SingleOwnerFound_RedirectsToOwnerDetails() {
        Owner testOwner = new Owner();
        testOwner.setId(1L);
        testOwner.setLastName("Doe");
        Page<Owner> singleOwnerPage = new PageImpl<>(Collections.singletonList(testOwner), PageRequest.of(0, 10), 1);

        when(ownerController.findPaginatedForOwnersLastName(1, "Doe")).thenReturn(singleOwnerPage);

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("redirect:/owners/1", viewName);
    }

    @Test
    void testProcessFindForm_MultipleOwnersFound_ReturnsOwnersListWithPagination() {
        Owner owner1 = new Owner();
        owner1.setId(1L);
        owner1.setLastName("Doe");
        Owner owner2 = new Owner();
        owner2.setId(2L);
        owner2.setLastName("Doe");
        Page<Owner> multipleOwnersPage = new PageImpl<>(List.of(owner1, owner2), PageRequest.of(0, 10), 2);

        when(ownerController.findPaginatedForOwnersLastName(1, "Doe")).thenReturn(multipleOwnersPage);

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("owners/ownersList", viewName);
        verify(ownerController).addPaginationModel(1, model, multipleOwnersPage);
    }

    @Test
    void testProcessFindForm_NullOwner_HandlesGracefully() {
        String viewName = ownerController.processFindForm(1, null, bindingResult, model);

        assertEquals("owners/ownersList", viewName);
    }

    @Test
    void testProcessFindForm_NullLastName_HandlesGracefully() {
        owner.setLastName(null);
        when(ownerController.findPaginatedForOwnersLastName(1, "")).thenReturn(ownersPage);

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("owners/ownersList", viewName);
    }

    @Test
    void testProcessFindForm_EmptyLastName_HandlesGracefully() {
        owner.setLastName("");
        when(ownerController.findPaginatedForOwnersLastName(1, "")).thenReturn(ownersPage);

        String viewName = ownerController.processFindForm(1, owner, bindingResult, model);

        assertEquals("owners/ownersList", viewName);
    }
}
