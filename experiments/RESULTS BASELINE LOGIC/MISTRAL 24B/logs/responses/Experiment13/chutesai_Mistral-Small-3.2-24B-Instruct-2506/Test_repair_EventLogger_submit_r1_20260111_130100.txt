package org.eclipse.cargotracker.interfaces.handling.mobile;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.time.LocalDateTime;
import java.util.List;
import jakarta.faces.context.FacesContext;
import jakarta.faces.model.SelectItem;
import org.eclipse.cargotracker.application.ApplicationEvents;
import org.eclipse.cargotracker.domain.model.cargo.CargoRepository;
import org.eclipse.cargotracker.domain.model.handling.HandlingEvent;
import org.eclipse.cargotracker.domain.model.location.LocationRepository;
import org.eclipse.cargotracker.domain.model.voyage.VoyageRepository;
import org.eclipse.cargotracker.interfaces.handling.HandlingEventRegistrationAttempt;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.junit.jupiter.api.Timeout;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class EventLogger_submitTestGenerated {

    @Mock
    private CargoRepository cargoRepository;

    @Mock
    private LocationRepository locationRepository;

    @Mock
    private VoyageRepository voyageRepository;

    @Mock
    private ApplicationEvents applicationEvents;

    @Mock
    private FacesContext context;

    @InjectMocks
    private EventLogger eventLogger;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    void submit_WithVoyageRequiredEvent_ShouldCreateAttemptWithVoyage() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation("USNYC");
        eventLogger.setVoyageNumber("VA001");
        eventLogger.setEventType(HandlingEvent.Type.LOAD);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act
        eventLogger.submit();

        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getVoyageNumber() != null &&
                attempt.getVoyageNumber().toString().equals("VA001")
            )
        );
    }

    @Test
    void submit_WithNonVoyageRequiredEvent_ShouldCreateAttemptWithoutVoyage() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation("USNYC");
        eventLogger.setVoyageNumber("VA001");
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act
        eventLogger.submit();

        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getVoyageNumber() == null
            )
        );
    }

    @Test
    void submit_WithNullTrackingId_ShouldThrowException() {
        // Arrange
        eventLogger.setTrackingId(null);
        eventLogger.setLocation("USNYC");
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act & Assert
        assertThrows(IllegalArgumentException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithNullLocation_ShouldThrowException() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation(null);
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act & Assert
        assertThrows(IllegalArgumentException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithNullEventType_ShouldThrowException() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation("USNYC");
        eventLogger.setEventType(null);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act & Assert
        assertThrows(IllegalArgumentException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithNullCompletionTime_ShouldUseCurrentTime() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation("USNYC");
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setCompletionTime(null);

        // Act
        eventLogger.submit();

        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getCompletionTime() != null
            )
        );
    }

    @Test
    void submit_WithValidInputs_ShouldAddFacesMessage() {
        // Arrange
        eventLogger.setTrackingId("TST001");
        eventLogger.setLocation("USNYC");
        eventLogger.setEventType(HandlingEvent.Type.RECEIVE);
        eventLogger.setCompletionTime(LocalDateTime.now());

        // Act
        eventLogger.submit();

        // Assert
        verify(context).addMessage(eq(null), argThat(message ->
            message.getSummary().equals("Event submitted")
        ));
    }
}
