package org.eclipse.cargotacker.interfaces.handling.mobile;

import jakarta.faces.application.FacesMessage;

import org.eclipse.cargotracker.application.ApplicationEvents;

import org.eclipse.cargotracker.domain.model.location.LocationRepository;
import org.eclipse.cargotracker.domain.model.voyage.VoyageRepository;

import org.eclipse.cargotracker.interfaces.handling.mobile.EventLogger;
import org.junit.jupiter.api.BeforeEach;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.any;
import static org.mockito.ArgumentMatchers.argThat;
import static org.mockito.ArgumentMatchers.eq;
import static org.mockito.Mockito.*;

@Timeout(value = 2, threadMode = Timeout.ThreadMode.SEPARATE_THREAD)
class EventLogger_submitTestGenerated {
    @Mock
    private CargoRepository cargoRepository;
    @Mock
    private LocationRepository locationRepository;
    @Mock
    private VoyageRepository voyageRepository;
    @Mock
    private ApplicationEvents applicationEvents;
    @Mock
    private FacesContext context;
    @Mock
    private FacesMessage facesMessage;
    @InjectMocks
    private EventLogger eventLogger;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.openMocks(this);
        eventLogger = new EventLogger();
        ReflectionTestUtils.setField(eventLogger, "cargoRepository", cargoRepository);
        ReflectionTestUtils.setField(eventLogger, "locationRepository", locationRepository);
        ReflectionTestUtils.setField(eventLogger, "voyageRepository", voyageRepository);
        ReflectionTestUtils.setField(eventLogger, "applicationEvents", applicationEvents);
    }

    @Test
    void submit_WithNullTrackingId_ShouldThrowException() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", null);
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        // Act & Assert
        assertThrows(IllegalArgumentException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithNullLocation_ShouldThrowException() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", null);
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        // Act & Assert
        assertThrows(IllegalArgumentException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithNullEventType_ShouldThrowException() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", null);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        // Act & Assert
        assertThrows(NullPointerException.class, () -> eventLogger.submit());
    }

    @Test
    void submit_WithVoyageRequiredEvent_ShouldCreateAttemptWithVoyage() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "voyageNumber", "VA001");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.LOAD);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        // Act
        eventLogger.submit();
        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getVoyageNumber() != null &&
                attempt.getVoyageNumber().equals("VA001")
            )
        );
    }

    @Test
    void submit_WithNonVoyageRequiredEvent_ShouldCreateAttemptWithoutVoyage() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "voyageNumber", "VA001");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        // Act
        eventLogger.submit();
        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getVoyageNumber() == null
            )
        );
    }

    @Test
    void submit_WithNullCompletionTime_ShouldUseCurrentTime() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", null);
        // Act
        eventLogger.submit();
        // Assert
        verify(applicationEvents).receivedHandlingEventRegistrationAttempt(
            argThat(attempt ->
                attempt.getCompletionTime() != null
            )
        );
    }

    @Test
    void submit_WithValidInputs_ShouldAddFacesMessage() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        when(facesMessage.getSummary()).thenReturn("Event submitted");
        // Act
        eventLogger.submit();
        // Assert
        verify(context).addMessage(eq(null), argThat(message ->
            message.getSummary().equals("Event submitted")
        ));
    }

    @Test
    void submit_ShouldAddFacesMessageWithCurrentInstance() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        when(facesMessage.getSummary()).thenReturn("Event submitted");
        // Act
        eventLogger.submit();
        // Assert
        verify(context).addMessage(
            any(),
            argThat(message -> "Event submitted".equals(message.getSummary()))
        );
    }

    @Test
    void submit_ShouldAddFacesMessageWithCurrentInstanceAndEmptyDetail() {
        // Arrange
        ReflectionTestUtils.setField(eventLogger, "trackingId", "TST001");
        ReflectionTestUtils.setField(eventLogger, "location", "USNYC");
        ReflectionTestUtils.setField(eventLogger, "eventType", HandlingEvent.Type.RECEIVE);
        ReflectionTestUtils.setField(eventLogger, "completionTime", LocalDateTime.now());
        when(facesMessage.getSummary()).thenReturn("Event submitted");
        when(facesMessage.getDetail()).thenReturn("");
        // Act
        eventLogger.submit();
        // Assert
        verify(context).addMessage(
            any(),
            argThat(message -> "Event submitted".equals(message.getSummary()) && message.getDetail().isEmpty())
        );
    }
}
