public String encode(final String source) {
    if (SoundexUtils.isEmpty(source)) {
        return "AAAAAAAAAAA";
    }

    StringBuilder encoded = new StringBuilder();

    // Handle null or empty string
    if (source == null || source.isEmpty()) {
        return "AAAAAAAAAAA";
    }

    // Process the input string
    String input = source.toUpperCase();
    int length = input.length();

    // Remove non-alphabetic characters
    StringBuilder cleaned = new StringBuilder();
    for (int i = 0; i < length; i++) {
        char c = input.charAt(i);
        if (Character.isLetter(c)) {
            cleaned.append(c);
        }
    }

    // Special cases
    if (cleaned.length() == 0) {
        return "AAAAAAAAAAA";
    }

    // Handle initial 'GN'
    if (cleaned.length() > 1 && cleaned.charAt(0) == 'G' && cleaned.charAt(1) == 'N') {
        encoded.append('2');
        cleaned.deleteCharAt(0);
        cleaned.deleteCharAt(0);
    }

    // Handle initial 'K' followed by 'W'
    if (cleaned.length() > 1 && cleaned.charAt(0) == 'K' && cleaned.charAt(1) == 'W') {
        encoded.append("KW");
        cleaned.deleteCharAt(0);
        cleaned.deleteCharAt(0);
    }

    // Handle initial 'WH'
    if (cleaned.length() > 1 && cleaned.charAt(0) == 'W' && cleaned.charAt(1) == 'H') {
        encoded.append("WH");
        cleaned.deleteCharAt(0);
        cleaned.deleteCharAt(0);
    }

    // Handle initial 'X'
    if (cleaned.length() > 0 && cleaned.charAt(0) == 'X') {
        encoded.append('S');
        cleaned.deleteCharAt(0);
    }

    // Process remaining characters
    for (int i = 0; i < cleaned.length(); i++) {
        char c = cleaned.charAt(i);

        // Handle consonants
        if (c == 'B' || c == 'C' || c == 'D' || c == 'F' || c == 'G' ||
            c == 'H' || c == 'J' || c == 'K' || c == 'L' || c == 'M' ||
            c == 'N' || c == 'P' || c == 'Q' || c == 'R' || c == 'S' ||
            c == 'T' || c == 'V' || c == 'W' || c == 'X' || c == 'Y' || c == 'Z') {

            // Handle 'C' followed by 'H'
            if (i < cleaned.length() - 1 && c == 'C' && cleaned.charAt(i + 1) == 'H') {
                encoded.append('K');
                i++;
                continue;
            }

            // Handle 'C' followed by 'K'
            if (i < cleaned.length() - 1 && c == 'C' && cleaned.charAt(i + 1) == 'K') {
                encoded.append('K');
                i++;
                continue;
            }

            // Handle 'D' followed by 'G'
            if (i < cleaned.length() - 1 && c == 'D' && cleaned.charAt(i + 1) == 'G') {
                encoded.append('J');
                i++;
                continue;
            }

            // Handle 'N' followed by 'G'
            if (i < cleaned.length() - 1 && c == 'N' && cleaned.charAt(i + 1) == 'G') {
                encoded.append('N');
                i++;
                continue;
            }

            // Handle 'S' followed by 'H'
            if (i < cleaned.length() - 1 && c == 'S' && cleaned.charAt(i + 1) == 'H') {
                encoded.append('SK');
                i++;
                continue;
            }

            // Handle 'T' followed by 'H'
            if (i < cleaned.length() - 1 && c == 'T' && cleaned.charAt(i + 1) == 'H') {
                encoded.append('T');
                i++;
                continue;
            }

            // Handle 'W' followed by 'H'
            if (i < cleaned.length() - 1 && c == 'W' && cleaned.charAt(i + 1) == 'H') {
                encoded.append('F');
                i++;
                continue;
            }

            // Default consonant handling
            encoded.append(c);
        }

        // Handle vowels
        if (c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U') {
            encoded.append(c);
        }
    }

    // Handle final 'H', 'W', or 'Y'
    if (encoded.length() > 0 && (encoded.charAt(encoded.length() - 1) == 'H' ||
                                encoded.charAt(encoded.length() - 1) == 'W' ||
                                encoded.charAt(encoded.length() - 1) == 'Y')) {
        encoded.setCharAt(encoded.length() - 1, 'A');
    }

    // Pad with zeros to make length 10
    while (encoded.length() < 10) {
        encoded.append('0');
    }

    return encoded.toString().substring(0, 10);
}
