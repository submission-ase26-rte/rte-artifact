public <O extends ArchiveOutputStream<? extends ArchiveEntry>> O createArchiveOutputStream(final String archiverName, final OutputStream out, final String actualEncoding) throws ArchiveException {
    if (archiverName == null) {
        throw new IllegalArgumentException("Archiver name must not be null");
    }
    if (out == null) {
        throw new IllegalArgumentException("OutputStream must not be null");
    }

    if (SEVEN_Z.equals(archiverName)) {
        throw new StreamingNotSupportedException("Streaming is not supported for seven-zip archives");
    }

    final String key = toKey(archiverName);
    final SortedMap<String, ArchiveStreamProvider> providers = getArchiveOutputStreamProviders();

    if (providers.containsKey(key)) {
        return providers.get(key).createArchiveOutputStream(archiverName, out, actualEncoding);
    }

    throw new ArchiveException("No provider found for archiver: " + archiverName);
}
