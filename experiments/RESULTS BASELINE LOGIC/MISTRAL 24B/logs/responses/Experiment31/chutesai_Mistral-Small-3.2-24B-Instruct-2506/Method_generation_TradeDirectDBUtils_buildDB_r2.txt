public void buildDB(java.io.PrintWriter out, java.io.InputStream ddlFile) throws Exception {
    out.println("<HEAD><BR><EM> TradeBuildDB: Building DayTrader Database...</EM><BR> This operation will take several minutes. Please wait...</HEAD>");
    out.println("<BODY>");

    if (ddlFile != null) {
        try {
            Object[] sqlBuffer = parseDDLToBuffer(ddlFile);
            if (sqlBuffer.length == 0) {
                out.println("<BR>TradeBuildDB: **** Parsing DDL file returned empty buffer, please check that a valid DB specific DDL file is available and retry ****</BR></BODY>");
                return;
            }

            if (!recreateDBTables(sqlBuffer, out)) {
                out.println("<BR>TradeBuildDB: **** Unable to create tables for the specified database ****</BR></BODY>");
                return;
            }
        } catch (IOException e) {
            Log.error(e, "Error parsing DDL file");
            out.println("<BR>TradeBuildDB: **** Unable to parse DDL file for the specified database ****</BR></BODY>");
            return;
        }
    } else {
        out.println("<BR>TradeBuildDB: **** Creating " + TradeConfig.getMAX_QUOTES() + " Quotes ****</BR>");

        RunStatsDataBean runStatsDataBean = resetTrade(true);
        if (runStatsDataBean == null) {
            throw new Exception("Error resetting trade");
        }

        for (int i = 0; i < TradeConfig.getMAX_QUOTES(); i++) {
            String symbol = TradeConfig.rndSymbol();
            String companyName = TradeConfig.rndFullName();
            BigDecimal price = new BigDecimal(TradeConfig.rndPrice());
            try {
                ts.createQuote(symbol, companyName, price);
            } catch (Exception e) {
                Log.error(e, "Error creating quote");
                if (i >= TradeConfig.getMAX_QUOTES() - 10) {
                    throw e;
                }
            }
            out.println("<BR>");
        }
    }
    out.println("</BODY>");
}
