package org.eclipse.cargotracker.application.internal;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.time.LocalDateTime;
import java.util.logging.Logger;

import jakarta.inject.Inject;
import org.eclipse.cargotracker.application.ApplicationEvents;
import org.eclipse.cargotracker.domain.model.cargo.TrackingId;
import org.eclipse.cargotracker.domain.model.handling.CannotCreateHandlingEventException;
import org.eclipse.cargotracker.domain.model.handling.HandlingEvent;
import org.eclipse.cargotracker.domain.model.handling.HandlingEventFactory;
import org.eclipse.cargotracker.domain.model.handling.HandlingEventRepository;
import org.eclipse.cargotracker.domain.model.location.UnLocode;
import org.eclipse.cargotracker.domain.model.voyage.VoyageNumber;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

@ExtendWith(MockitoExtension.class)
class DefaultHandlingEventService_registerHandlingEventTestGenerated {

    @Mock
    private ApplicationEvents applicationEvents;

    @Mock
    private HandlingEventRepository handlingEventRepository;

    @Mock
    private HandlingEventFactory handlingEventFactory;

    @Mock
    private Logger logger;

    @InjectMocks
    private DefaultHandlingEventService defaultHandlingEventService;

    private TrackingId trackingId;
    private VoyageNumber voyageNumber;
    private UnLocode unLocode;
    private LocalDateTime completionTime;

    @BeforeEach
    void setUp() {
        trackingId = new TrackingId("TEST1234567");
        voyageNumber = new VoyageNumber("VOY123");
        unLocode = new UnLocode("USNYC");
        completionTime = LocalDateTime.now();
    }

    @Test
    void registerHandlingEvent_ShouldCreateAndStoreEvent() throws CannotCreateHandlingEventException {
        HandlingEvent mockEvent = mock(HandlingEvent.class);
        when(handlingEventFactory.createHandlingEvent(any(), any(), any(), any(), any(), any()))
                .thenReturn(mockEvent);

        defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, HandlingEvent.Type.LOAD);

        verify(handlingEventFactory).createHandlingEvent(any(), any(), eq(trackingId), eq(voyageNumber), eq(unLocode), eq(HandlingEvent.Type.LOAD));
        verify(handlingEventRepository).store(mockEvent);
        verify(applicationEvents).cargoWasHandled(mockEvent);
        verify(logger).info("Registered handling event");
    }

    @Test
    void registerHandlingEvent_ShouldThrowExceptionWhenEventCreationFails() {
        when(handlingEventFactory.createHandlingEvent(any(), any(), any(), any(), any(), any()))
                .thenThrow(new CannotCreateHandlingEventException("Error creating handling event"));

        assertThrows(CannotCreateHandlingEventException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, HandlingEvent.Type.LOAD);
        });

        verify(handlingEventFactory).createHandlingEvent(any(), any(), eq(trackingId), eq(voyageNumber), eq(unLocode), eq(HandlingEvent.Type.LOAD));
        verifyNoMoreInteractions(handlingEventRepository, applicationEvents, logger);
    }

    @Test
    void registerHandlingEvent_WithNullCompletionTime_ShouldThrowException() {
        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(null, trackingId, voyageNumber, unLocode, HandlingEvent.Type.LOAD);
        });
    }

    @Test
    void registerHandlingEvent_WithNullTrackingId_ShouldThrowException() {
        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, null, voyageNumber, unLocode, HandlingEvent.Type.LOAD);
        });
    }

    @Test
    void registerHandlingEvent_WithNullVoyageNumber_ShouldThrowException() {
        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, null, unLocode, HandlingEvent.Type.LOAD);
        });
    }

    @Test
    void registerHandlingEvent_WithNullUnLocode_ShouldThrowException() {
        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, null, HandlingEvent.Type.LOAD);
        });
    }

    @Test
    void registerHandlingEvent_WithNullHandlingEventType_ShouldThrowException() {
        assertThrows(NullPointerException.class, () -> {
            defaultHandlingEventService.registerHandlingEvent(completionTime, trackingId, voyageNumber, unLocode, null);
        });
    }
}
