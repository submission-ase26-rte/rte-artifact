public String asPath() {
    if (id != null) {
        return "id('" + id + "')";
    }

    if (node.getNodeType() == Node.DOCUMENT_NODE || node.getNodeType() == Node.COMMENT_NODE) {
        return "";
    }

    NodePointer parent = getParentPointer();
    if (parent != null && !(parent instanceof DOMNodePointer)) {
        return parent.asPath();
    }

    String parentPath = parent != null ? parent.asPath() : "";

    switch (node.getNodeType()) {
        case Node.ELEMENT_NODE:
            String localName = getLocalName(node);
            String namespaceURI = getNamespaceURI();
            if (namespaceURI != null) {
                NamespaceResolver resolver = getNamespaceResolver();
                if (resolver != null) {
                    String prefix = resolver.getPrefix(namespaceURI);
                    if (prefix != null) {
                        localName = prefix + ":" + localName;
                    } else {
                        return (parentPath.isEmpty() ? "/" : parentPath) + "node()[" + getRelativePositionOfElement() + "]";
                    }
                }
            }
            int position = getRelativePositionByQName();
            String nodePath = localName + "[" + position + "]";
            return (parentPath.isEmpty() ? "/" : parentPath) + nodePath;

        case Node.TEXT_NODE:
        case Node.CDATA_SECTION_NODE:
            return (parentPath.isEmpty() ? "/" : parentPath) + "text()[" + getRelativePositionOfTextNode() + "]";

        case Node.PROCESSING_INSTRUCTION_NODE:
            ProcessingInstruction pi = (ProcessingInstruction) node;
            return (parentPath.isEmpty() ? "/" : parentPath) + "processing-instruction('" + pi.getTarget() + "')[" + getRelativePositionOfPI() + "]";

        default:
            return "";
    }
}
