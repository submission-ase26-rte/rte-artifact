public String doubleMetaphone(String value, final boolean alternate) {
    if (value == null) {
        return null;
    }

    if (value.isEmpty()) {
        return "";
    }

    DoubleMetaphoneResult result = new DoubleMetaphoneResult(maxCodeLen);
    int index = 0;
    int len = value.length();

    value = cleanInput(value);

    boolean slavoGermanic = isSlavoGermanic(value);
    boolean scandinavian = isSlavoGermanic(value);

    while (index < len && result.getCurrentCodeSize() < maxCodeLen) {
        char[] chars = value.toCharArray();
        char currentChar = charAt(value, index);

        if (currentChar == 'A' || currentChar == 'E' || currentChar == 'I' || currentChar == 'O' || currentChar == 'U' || currentChar == 'Y') {
            index = handleAEIOUY(result, index);
        } else if (currentChar == 'B') {
            index = handleB(value, result, index);
        } else if (currentChar == 'Ç' || currentChar == 'C') {
            index = handleC(value, result, index);
        } else if (currentChar == 'D') {
            index = handleD(value, result, index);
        } else if (currentChar == 'G') {
            index = handleG(value, result, index, slavoGermanic);
        } else if (currentChar == 'F') {
            index = handleF(value, result, index);
        } else if (currentChar == 'H') {
            index = handleH(value, result, index);
        } else if (currentChar == 'J') {
            index = handleJ(value, result, index, slavoGermanic);
        } else if (currentChar == 'K') {
            index = handleK(value, result, index);
        } else if (currentChar == 'P') {
            index = handleP(value, result, index);
        } else if (currentChar == 'Q') {
            index = handleQ(value, result, index);
        } else if (currentChar == 'R') {
            index = handleR(value, result, index, slavoGermanic);
        } else if (currentChar == 'S') {
            index = handleS(value, result, index, slavoGermanic);
        } else if (currentChar == 'T') {
            index = handleT(value, result, index);
        } else if (currentChar == 'V') {
            index = handleV(value, result, index);
        } else if (currentChar == 'W' || currentChar == 'Y') {
            index = handleW(value, result, index);
        } else if (currentChar == 'X') {
            index = handleX(value, result, index);
        } else if (currentChar == 'Z') {
            index = handleZ(value, result, index, slavoGermanic);
        } else {
            index++;
        }
    }

    return alternate ? result.getAlternate() : result.getPrimary();
}
