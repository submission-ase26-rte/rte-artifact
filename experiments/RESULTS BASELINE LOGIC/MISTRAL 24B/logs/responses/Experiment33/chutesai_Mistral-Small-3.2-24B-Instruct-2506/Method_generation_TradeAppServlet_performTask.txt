public void performTask(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
    String action = req.getParameter("action");
    HttpSession session = req.getSession();
    String userID = (String) session.getAttribute("uidBean");

    if (action == null) {
        tsAction.doWelcome(getServletContext(), req, resp, "");
        return;
    }

    switch (action) {
        case "login":
            String uid = req.getParameter("uid");
            String passwd = req.getParameter("passwd");
            tsAction.doLogin(getServletContext(), req, resp, uid, passwd);
            break;
        case "register":
            String userId = req.getParameter("user id");
            String password = req.getParameter("passwd");
            String confirmPassword = req.getParameter("confirm passwd");
            String fullName = req.getParameter("Full Name");
            String creditCard = req.getParameter("Credit Card Number");
            String money = req.getParameter("money");
            String email = req.getParameter("email");
            String address = req.getParameter("snail mail");
            tsAction.doRegister(getServletContext(), req, resp, userId, password, confirmPassword, fullName, creditCard, money, email, address);
            break;
        case "quotes":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            String symbols = req.getParameter("symbols");
            tsAction.doQuotes(getServletContext(), req, resp, userID, symbols);
            break;
        case "buy":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            String symbol = req.getParameter("symbol");
            String quantity = req.getParameter("quantity");
            tsAction.doBuy(getServletContext(), req, resp, userID, symbol, quantity);
            break;
        case "sell":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            String holdingID = req.getParameter("holdingID");
            tsAction.doSell(getServletContext(), req, resp, userID, Integer.parseInt(holdingID));
            break;
        case "portfolio":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            tsAction.doPortfolio(getServletContext(), req, resp, userID, "");
            break;
        case "logout":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            tsAction.doLogout(getServletContext(), req, resp, userID);
            break;
        case "home":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            tsAction.doHome(getServletContext(), req, resp, userID, "Ready to Trade");
            break;
        case "update_profile":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            String password = req.getParameter("password");
            String cpassword = req.getParameter("cpassword");
            String fullname = req.getParameter("fullname");
            String address = req.getParameter("address");
            String creditcard = req.getParameter("creditcard");
            String email = req.getParameter("email");
            tsAction.doAccountUpdate(getServletContext(), req, resp, userID, password, cpassword, fullname, address, creditcard, email);
            break;
        case "mksummary":
            if (userID == null) {
                tsAction.doWelcome(getServletContext(), req, resp, "User Not Logged in");
                return;
            }
            tsAction.doMarketSummary(getServletContext(), req, resp, userID);
            break;
        default:
            tsAction.doWelcome(getServletContext(), req, resp, "TradeAppServlet: Invalid Action" + action);
            break;
    }
}
