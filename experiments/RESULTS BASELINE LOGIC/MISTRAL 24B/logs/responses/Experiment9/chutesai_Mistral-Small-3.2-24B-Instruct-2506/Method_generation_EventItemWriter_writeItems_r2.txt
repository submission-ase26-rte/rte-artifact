public void writeItems(List<Object> items) throws Exception {
    if (items == null || items.isEmpty()) {
        return;
    }

    Properties properties = jobContext.getProperties();
    String archiveDirectory = properties.getProperty("archive_directory", System.getProperty("java.io.tmpdir"));
    String jobName = jobContext.getJobName();
    long instanceId = jobContext.getInstanceId();
    String fileName = String.format("archive_%s_%d.csv", jobName, instanceId);
    Path filePath = Path.of(archiveDirectory, fileName);

    for (Object item : items) {
        if (item == null) {
            continue;
        }

        HandlingEventRegistrationAttempt attempt = (HandlingEventRegistrationAttempt) item;
        applicationEvents.receivedHandlingEventRegistrationAttempt(attempt);

        String csvLine = String.format("%s,%s,%s,%s,%s,%s",
            attempt.getTrackingId(),
            attempt.getVoyageNumber(),
            attempt.getUnLocode(),
            attempt.getType(),
            DateConverter.toString(attempt.getRegistrationTime()),
            DateConverter.toString(attempt.getCompletionTime()));

        Files.createDirectories(filePath.getParent());
        Files.writeString(filePath, csvLine + System.lineSeparator(), StandardOpenOption.CREATE, StandardOpenOption.APPEND);
    }
}
