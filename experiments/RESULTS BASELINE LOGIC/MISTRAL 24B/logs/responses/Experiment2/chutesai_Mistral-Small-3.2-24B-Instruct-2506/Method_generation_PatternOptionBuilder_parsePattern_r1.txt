public static Options parsePattern(final String pattern) {
    if (pattern == null) {
        throw new NullPointerException();
    }

    Options options = new Options();
    if (pattern.isEmpty()) {
        return options;
    }

    for (int i = 0; i < pattern.length(); i++) {
        if (Character.isLetter(pattern.charAt(i))) {
            Option.Builder builder = Option.builder();
            builder.withLongOpt(null);
            builder.withArgName(null);
            builder.withDescription(null);
            builder.withValueSeparator('=');
            builder.withOptionalArg(false);
            builder.withRequired(false);
            builder.withType(null);

            int j = i + 1;
            if (j < pattern.length()) {
                char next = pattern.charAt(j);
                if (next == '!') {
                    builder.withRequired(true);
                    i = j;
                } else if (isValueCode(next)) {
                    Class<?> type = getValueType(next);
                    builder.withType(type);
                    i = j;
                }
            }

            Option option = builder.create(pattern.charAt(i));
            options.addOption(option);
        }
    }

    return options;
}
