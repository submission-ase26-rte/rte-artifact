public static Options parsePattern(final String pattern) {
    if (pattern == null) {
        throw new NullPointerException();
    }

    Options options = new Options();
    if (pattern.isEmpty()) {
        return options;
    }

    for (int i = 0; i < pattern.length(); i++) {
        char c = pattern.charAt(i);
        if (Character.isLetter(c)) {
            Option.Builder builder = Option.builder()
                .withLongOpt(null)
                .withArgName(null)
                .withDescription(null)
                .withValueSeparator('=')
                .withOptionalArg(false)
                .withRequired(false)
                .withType(null);

            int j = i + 1;
            if (j < pattern.length()) {
                char next = pattern.charAt(j);
                if (next == '!') {
                    builder.withRequired(true);
                    i = j;
                } else if (isValueCode(next)) {
                    Class<?> type = getValueType(next);
                    if (type != null) {
                        builder.withType(type);
                        if (type == File[].class) {
                            builder.withConverter(unsupported());
                        }
                        i = j;
                    }
                }
            }

            Option option = builder.create(c);
            options.addOption(option);
        }
    }

    return options;
}
